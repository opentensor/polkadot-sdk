143/// Recovery handle that fails regularly to simulate unavailable povs.
144pub struct FailingRecoveryHandle {
145	overseer_handle: OverseerHandle,
146	counter: u32,
147	failed_hashes: HashSet<CandidateHash>,
148}
149
150impl FailingRecoveryHandle {
151	/// Create a new FailingRecoveryHandle
152	pub fn new(overseer_handle: OverseerHandle) -> Self {
153		Self { overseer_handle, counter: 0, failed_hashes: Default::default() }
154	}
155}
156
157#[async_trait::async_trait]
158impl RecoveryHandle for FailingRecoveryHandle {
159	async fn send_recovery_msg(
160		&mut self,
161		message: AvailabilityRecoveryMessage,
162		origin: &'static str,
163	) {
164		let AvailabilityRecoveryMessage::RecoverAvailableData(ref receipt, _, _, _, _) = message;
165		let candidate_hash = receipt.hash();
166
167		// For every 3rd block we immediately signal unavailability to trigger
168		// a retry. The same candidate is never failed multiple times to ensure progress.
169		if self.counter % 3 == 0 && self.failed_hashes.insert(candidate_hash) {
170			tracing::info!(target: LOG_TARGET, ?candidate_hash, "Failing pov recovery.");
171
172			let AvailabilityRecoveryMessage::RecoverAvailableData(_, _, _, _, back_sender) =
173				message;
174			back_sender
175				.send(Err(RecoveryError::Unavailable))
176				.expect("Return channel should work here.");
177		} else {
178			self.overseer_handle.send_msg(message, origin).await;
179		}
180		self.counter += 1;
181	}
182}

194/// Starts a `ServiceBuilder` for a full service.
195///
196/// Use this macro if you don't actually need the full service, but just the builder in order to
197/// be able to perform chain operations.
198pub fn new_partial(
199	config: &mut Configuration,
200	enable_import_proof_record: bool,
201) -> Result<Service, sc_service::Error> {
202	let heap_pages = config
203		.executor
204		.default_heap_pages
205		.map_or(DEFAULT_HEAP_ALLOC_STRATEGY, |h| HeapAllocStrategy::Static { extra_pages: h as _ });
206
207	let executor = WasmExecutor::builder()
208		.with_execution_method(config.executor.wasm_method)
209		.with_onchain_heap_alloc_strategy(heap_pages)
210		.with_offchain_heap_alloc_strategy(heap_pages)
211		.with_max_runtime_instances(config.executor.max_runtime_instances)
212		.with_runtime_cache_size(config.executor.runtime_cache_size)
213		.build();
214
215	let (client, backend, keystore_container, task_manager) =
216		sc_service::new_full_parts_record_import::<Block, RuntimeApi, _>(
217			config,
218			None,
219			executor,
220			enable_import_proof_record,
221		)?;
222	let client = Arc::new(client);
223
224	let (block_import, slot_based_handle) =
225		SlotBasedBlockImport::new(client.clone(), client.clone());
226	let block_import = ParachainBlockImport::new(block_import, backend.clone());
227
228	let transaction_pool = Arc::from(
229		sc_transaction_pool::Builder::new(
230			task_manager.spawn_essential_handle(),
231			client.clone(),
232			config.role.is_authority().into(),
233		)
234		.with_options(config.transaction_pool.clone())
235		.with_prometheus(config.prometheus_registry())
236		.build(),
237	);
238
239	let slot_duration = sc_consensus_aura::slot_duration(&*client)?;
240	let import_queue = cumulus_client_consensus_aura::import_queue::<AuthorityPair, _, _, _, _, _>(
241		ImportQueueParams {
242			block_import: block_import.clone(),
243			client: client.clone(),
244			create_inherent_data_providers: move |_, ()| async move {
245				let timestamp = sp_timestamp::InherentDataProvider::from_system_time();
246
247				let slot =
248					sp_consensus_aura::inherents::InherentDataProvider::from_timestamp_and_slot_duration(
249						*timestamp,
250						slot_duration,
251					);
252
253				Ok((slot, timestamp))
254			},
255			spawner: &task_manager.spawn_essential_handle(),
256			registry: None,
257			telemetry: None,
258		},
259	)?;
260
261	let params = PartialComponents {
262		backend,
263		client,
264		import_queue,
265		keystore_container,
266		task_manager,
267		transaction_pool,
268		select_chain: (),
269		other: (block_import, slot_based_handle),
270	};
271
272	Ok(params)
273}

275async fn build_relay_chain_interface(
276	relay_chain_config: Configuration,
277	parachain_prometheus_registry: Option<&Registry>,
278	collator_key: Option<CollatorPair>,
279	collator_options: CollatorOptions,
280	task_manager: &mut TaskManager,
281) -> RelayChainResult<Arc<dyn RelayChainInterface + 'static>> {
282	let relay_chain_node = match collator_options.relay_chain_mode {
283		cumulus_client_cli::RelayChainMode::Embedded => polkadot_test_service::new_full(
284			relay_chain_config,
285			if let Some(ref key) = collator_key {
286				polkadot_service::IsParachainNode::Collator(key.clone())
287			} else {
288				polkadot_service::IsParachainNode::Collator(CollatorPair::generate().0)
289			},
290			None,
291			polkadot_service::CollatorOverseerGen,
292		)
293		.map_err(|e| RelayChainError::Application(Box::new(e) as Box<_>))?,
294		cumulus_client_cli::RelayChainMode::ExternalRpc(rpc_target_urls) =>
295			return build_minimal_relay_chain_node_with_rpc(
296				relay_chain_config,
297				parachain_prometheus_registry,
298				task_manager,
299				rpc_target_urls,
300			)
301			.await
302			.map(|r| r.0),
303		cumulus_client_cli::RelayChainMode::LightClient =>
304			return build_minimal_relay_chain_node_light_client(relay_chain_config, task_manager)
305				.await
306				.map(|r| r.0),
307	};
308
309	task_manager.add_child(relay_chain_node.task_manager);
310	tracing::info!("Using inprocess node.");
311	Ok(Arc::new(RelayChainInProcessInterface::new(
312		relay_chain_node.client.clone(),
313		relay_chain_node.backend.clone(),
314		relay_chain_node.sync_service.clone(),
315		relay_chain_node.overseer_handle.ok_or(RelayChainError::GenericError(
316			"Overseer should be running in full node.".to_string(),
317		))?,
318	)))
319}

321/// Start a node with the given parachain `Configuration` and relay chain `Configuration`.
322///
323/// This is the actual implementation that is abstract over the executor and the runtime api.
324#[sc_tracing::logging::prefix_logs_with("Parachain")]
325pub async fn start_node_impl<RB, Net: NetworkBackend<Block, Hash>>(
326	parachain_config: Configuration,
327	collator_key: Option<CollatorPair>,
328	relay_chain_config: Configuration,
329	wrap_announce_block: Option<Box<dyn FnOnce(AnnounceBlockFn) -> AnnounceBlockFn>>,
330	fail_pov_recovery: bool,
331	rpc_ext_builder: RB,
332	consensus: Consensus,
333	collator_options: CollatorOptions,
334	proof_recording_during_import: bool,
335	use_slot_based_collator: bool,
336) -> sc_service::error::Result<(
337	TaskManager,
338	Arc<Client>,
339	Arc<dyn NetworkService>,
340	RpcHandlers,
341	TransactionPool,
342	Arc<Backend>,
343)>
344where
345	RB: Fn(Arc<Client>) -> Result<jsonrpsee::RpcModule<()>, sc_service::Error> + Send + 'static,
346{
347	let mut parachain_config = prepare_node_config(parachain_config);
348
349	let params = new_partial(&mut parachain_config, proof_recording_during_import)?;
350
351	let transaction_pool = params.transaction_pool.clone();
352	let mut task_manager = params.task_manager;
353
354	let client = params.client.clone();
355	let backend = params.backend.clone();
356
357	let block_import = params.other.0;
358	let slot_based_handle = params.other.1;
359	let relay_chain_interface = build_relay_chain_interface(
360		relay_chain_config,
361		parachain_config.prometheus_registry(),
362		collator_key.clone(),
363		collator_options.clone(),
364		&mut task_manager,
365	)
366	.await
367	.map_err(|e| sc_service::Error::Application(Box::new(e) as Box<_>))?;
368
369	let import_queue_service = params.import_queue.service();
370	let prometheus_registry = parachain_config.prometheus_registry().cloned();
371	let net_config = FullNetworkConfiguration::<Block, Hash, Net>::new(
372		&parachain_config.network,
373		prometheus_registry.clone(),
374	);
375
376	let best_hash = client.chain_info().best_hash;
377	let para_id = client
378		.runtime_api()
379		.parachain_id(best_hash)
380		.map_err(|e| sc_service::Error::Application(Box::new(e) as Box<_>))?;
381	tracing::info!("Parachain id: {:?}", para_id);
382
383	let (network, system_rpc_tx, tx_handler_controller, sync_service) =
384		build_network(BuildNetworkParams {
385			parachain_config: &parachain_config,
386			net_config,
387			client: client.clone(),
388			transaction_pool: transaction_pool.clone(),
389			para_id,
390			spawn_handle: task_manager.spawn_handle(),
391			relay_chain_interface: relay_chain_interface.clone(),
392			import_queue: params.import_queue,
393			metrics: Net::register_notification_metrics(
394				parachain_config.prometheus_config.as_ref().map(|config| &config.registry),
395			),
396			sybil_resistance_level: CollatorSybilResistance::Resistant, /* Either Aura that is
397			                                                             * resistant or null that
398			                                                             * is not producing any
399			                                                             * blocks at all. */
400		})
401		.await?;
402
403	let keystore = params.keystore_container.keystore();
404	let rpc_builder = {
405		let client = client.clone();
406		Box::new(move |_| rpc_ext_builder(client.clone()))
407	};
408
409	let rpc_handlers = sc_service::spawn_tasks(sc_service::SpawnTasksParams {
410		rpc_builder,
411		client: client.clone(),
412		transaction_pool: transaction_pool.clone(),
413		task_manager: &mut task_manager,
414		config: parachain_config,
415		keystore: keystore.clone(),
416		backend: backend.clone(),
417		network: network.clone(),
418		sync_service: sync_service.clone(),
419		system_rpc_tx,
420		tx_handler_controller,
421		telemetry: None,
422	})?;
423
424	let announce_block = {
425		let sync_service = sync_service.clone();
426		Arc::new(move |hash, data| sync_service.announce_block(hash, data))
427	};
428
429	let announce_block = wrap_announce_block
430		.map(|w| (w)(announce_block.clone()))
431		.unwrap_or_else(|| announce_block);
432
433	let overseer_handle = relay_chain_interface
434		.overseer_handle()
435		.map_err(|e| sc_service::Error::Application(Box::new(e)))?;
436
437	let recovery_handle: Box<dyn RecoveryHandle> = if fail_pov_recovery {
438		Box::new(FailingRecoveryHandle::new(overseer_handle.clone()))
439	} else {
440		Box::new(overseer_handle.clone())
441	};
442	let relay_chain_slot_duration = Duration::from_secs(6);
443
444	start_relay_chain_tasks(StartRelayChainTasksParams {
445		client: client.clone(),
446		announce_block: announce_block.clone(),
447		para_id,
448		relay_chain_interface: relay_chain_interface.clone(),
449		task_manager: &mut task_manager,
450		// Increase speed of recovery for testing purposes.
451		da_recovery_profile: DARecoveryProfile::Other(RecoveryDelayRange {
452			min: Duration::from_secs(1),
453			max: Duration::from_secs(5),
454		}),
455		import_queue: import_queue_service,
456		relay_chain_slot_duration,
457		recovery_handle,
458		sync_service: sync_service.clone(),
459		prometheus_registry: None,
460	})?;
461
462	if let Some(collator_key) = collator_key {
463		if let Consensus::Null = consensus {
464			#[allow(deprecated)]
465			old_consensus::start_collator(old_consensus::StartCollatorParams {
466				block_status: client.clone(),
467				announce_block,
468				runtime_api: client.clone(),
469				spawner: task_manager.spawn_handle(),
470				para_id,
471				parachain_consensus: Box::new(NullConsensus) as Box<_>,
472				key: collator_key,
473				overseer_handle,
474			})
475			.await;
476		} else {
477			let proposer_factory = sc_basic_authorship::ProposerFactory::with_proof_recording(
478				task_manager.spawn_handle(),
479				client.clone(),
480				transaction_pool.clone(),
481				prometheus_registry.as_ref(),
482				None,
483			);
484			let proposer = Proposer::new(proposer_factory);
485
486			let collator_service = CollatorService::new(
487				client.clone(),
488				Arc::new(task_manager.spawn_handle()),
489				announce_block,
490				client.clone(),
491			);
492
493			let client_for_aura = client.clone();
494
495			if use_slot_based_collator {
496				tracing::info!(target: LOG_TARGET, "Starting block authoring with slot based authoring.");
497				let params = SlotBasedParams {
498					create_inherent_data_providers: move |_, ()| async move { Ok(()) },
499					block_import,
500					para_client: client.clone(),
501					para_backend: backend.clone(),
502					relay_client: relay_chain_interface,
503					code_hash_provider: move |block_hash| {
504						client_for_aura
505							.code_at(block_hash)
506							.ok()
507							.map(|c| ValidationCode::from(c).hash())
508					},
509					keystore,
510					collator_key,
511					relay_chain_slot_duration,
512					para_id,
513					proposer,
514					collator_service,
515					authoring_duration: Duration::from_millis(2000),
516					reinitialize: false,
517					slot_offset: Duration::from_secs(1),
518					block_import_handle: slot_based_handle,
519					spawner: task_manager.spawn_handle(),
520					export_pov: None,
521					max_pov_percentage: None,
522				};
523
524				slot_based::run::<Block, AuthorityPair, _, _, _, _, _, _, _, _, _>(params);
525			} else {
526				tracing::info!(target: LOG_TARGET, "Starting block authoring with lookahead collator.");
527				let params = AuraParams {
528					create_inherent_data_providers: move |_, ()| async move { Ok(()) },
529					block_import,
530					para_client: client.clone(),
531					para_backend: backend.clone(),
532					relay_client: relay_chain_interface,
533					code_hash_provider: move |block_hash| {
534						client_for_aura
535							.code_at(block_hash)
536							.ok()
537							.map(|c| ValidationCode::from(c).hash())
538					},
539					keystore,
540					collator_key,
541					para_id,
542					overseer_handle,
543					relay_chain_slot_duration,
544					proposer,
545					collator_service,
546					authoring_duration: Duration::from_millis(2000),
547					reinitialize: false,
548					max_pov_percentage: None,
549				};
550
551				let fut = aura::run::<Block, AuthorityPair, _, _, _, _, _, _, _, _>(params);
552				task_manager.spawn_essential_handle().spawn("aura", None, fut);
553			}
554		}
555	}
556
557	Ok((task_manager, client, network, rpc_handlers, transaction_pool, backend))
558}

746	/// Build the [`TestNode`].
747	pub async fn build(self) -> TestNode {
748		let parachain_config = node_config(
749			self.storage_update_func_parachain.unwrap_or_else(|| Box::new(|| ())),
750			self.tokio_handle.clone(),
751			self.key,
752			self.parachain_nodes,
753			self.parachain_nodes_exclusive,
754			self.para_id,
755			self.collator_key.is_some(),
756			self.endowed_accounts,
757		)
758		.expect("could not generate Configuration");
759
760		let mut relay_chain_config = polkadot_test_service::node_config(
761			self.storage_update_func_relay_chain.unwrap_or_else(|| Box::new(|| ())),
762			self.tokio_handle,
763			self.key,
764			self.relay_chain_nodes,
765			false,
766		);
767
768		let collator_options = CollatorOptions {
769			relay_chain_mode: self.relay_chain_mode,
770			embedded_dht_bootnode: true,
771			dht_bootnode_discovery: true,
772		};
773
774		relay_chain_config.network.node_name =
775			format!("{} (relay chain)", relay_chain_config.network.node_name);
776
777		let (task_manager, client, network, rpc_handlers, transaction_pool, backend) =
778			match relay_chain_config.network.network_backend {
779				sc_network::config::NetworkBackendType::Libp2p =>
780					start_node_impl::<_, sc_network::NetworkWorker<_, _>>(
781						parachain_config,
782						self.collator_key,
783						relay_chain_config,
784						self.wrap_announce_block,
785						false,
786						|_| Ok(jsonrpsee::RpcModule::new(())),
787						self.consensus,
788						collator_options,
789						self.record_proof_during_import,
790						false,
791					)
792					.await
793					.expect("could not create Cumulus test service"),
794				sc_network::config::NetworkBackendType::Litep2p =>
795					start_node_impl::<_, sc_network::Litep2pNetworkBackend>(
796						parachain_config,
797						self.collator_key,
798						relay_chain_config,
799						self.wrap_announce_block,
800						false,
801						|_| Ok(jsonrpsee::RpcModule::new(())),
802						self.consensus,
803						collator_options,
804						self.record_proof_during_import,
805						false,
806					)
807					.await
808					.expect("could not create Cumulus test service"),
809			};
810		let peer_id = network.local_peer_id();
811		let multiaddr = polkadot_test_service::get_listen_address(network.clone()).await;
812		let addr = MultiaddrWithPeerId { multiaddr, peer_id };
813
814		TestNode { task_manager, client, network, addr, rpc_handlers, transaction_pool, backend }
815	}

818/// Create a Cumulus `Configuration`.
819///
820/// By default a TCP socket will be used, therefore you need to provide nodes if you want the
821/// node to be connected to other nodes.
822///
823/// If `nodes_exclusive` is `true`, the node will only connect to the given `nodes` and not to any
824/// other node.
825///
826/// The `storage_update_func` can be used to make adjustments to the runtime genesis.
827pub fn node_config(
828	storage_update_func: impl Fn(),
829	tokio_handle: tokio::runtime::Handle,
830	key: Sr25519Keyring,
831	nodes: Vec<MultiaddrWithPeerId>,
832	nodes_exclusive: bool,
833	para_id: ParaId,
834	is_collator: bool,
835	endowed_accounts: Vec<AccountId>,
836) -> Result<Configuration, ServiceError> {
837	let base_path = BasePath::new_temp_dir()?;
838	let root = base_path.path().join(format!("cumulus_test_service_{}", key));
839	let role = if is_collator { Role::Authority } else { Role::Full };
840	let key_seed = key.to_seed();
841	let mut spec = Box::new(chain_spec::get_chain_spec_with_extra_endowed(
842		Some(para_id),
843		endowed_accounts,
844		cumulus_test_runtime::WASM_BINARY.expect("WASM binary was not built, please build it!"),
845	));
846
847	let mut storage = spec.as_storage_builder().build_storage().expect("could not build storage");
848
849	BasicExternalities::execute_with_storage(&mut storage, storage_update_func);
850	spec.set_storage(storage);
851
852	let mut network_config = NetworkConfiguration::new(
853		format!("{} (parachain)", key_seed),
854		"network/test/0.1",
855		Default::default(),
856		None,
857	);
858
859	if nodes_exclusive {
860		network_config.default_peers_set.reserved_nodes = nodes;
861		network_config.default_peers_set.non_reserved_mode =
862			sc_network::config::NonReservedPeerMode::Deny;

867	network_config.allow_non_globals_in_dht = true;

870	network_config.listen_addresses.push(addr.clone());
871	network_config.transport =
872		TransportConfig::Normal { enable_mdns: false, allow_private_ip: true };
873
874	Ok(Configuration {
875		impl_name: "cumulus-test-node".to_string(),
876		impl_version: "0.1".to_string(),
877		role,
878		tokio_handle,
879		transaction_pool: Default::default(),

881		keystore: KeystoreConfig::InMemory,
882		database: DatabaseSource::RocksDb { path: root.join("db"), cache_size: 128 },
883		trie_cache_maximum_size: Some(64 * 1024 * 1024),
884		warm_up_trie_cache: None,
885		state_pruning: Some(PruningMode::ArchiveAll),

892			},
893			..ExecutorConfiguration::default()
894		},
895		rpc: RpcConfiguration {

898			cors: None,
899			methods: Default::default(),
900			max_request_size: Default::default(),
901			max_response_size: Default::default(),
902			id_provider: None,
903			max_subs_per_conn: Default::default(),
904			port: 9945,
905			message_buffer_capacity: Default::default(),
906			batch_config: RpcBatchRequestConfig::Unlimited,

913		offchain_worker: OffchainWorkerConfig { enabled: true, indexing_enabled: false },
914		force_authoring: false,

917		tracing_targets: None,

961/// Fetch account nonce for key pair

963	let best_hash = client.chain_info().best_hash;
964	client
965		.runtime_api()

970/// Construct an extrinsic that can be applied to the test runtime.
971pub fn construct_extrinsic(
972	client: &Client,
973	function: impl Into<runtime::RuntimeCall>,
974	caller: sp_core::sr25519::Pair,
975	nonce: Option<u32>,
976) -> runtime::UncheckedExtrinsic {
977	let function = function.into();
978	let current_block_hash = client.info().best_hash;
979	let current_block = client.info().best_number.saturated_into();
980	let genesis_block = client.hash(0).unwrap().unwrap();
981	let nonce = nonce.unwrap_or_else(|| fetch_nonce(client, caller.public()));
982	let period = runtime::BlockHashCount::get()
983		.checked_next_power_of_two()
984		.map(|c| c / 2)
985		.unwrap_or(2) as u64;
986	let tip = 0;
987	let tx_ext: runtime::TxExtension = (
988		frame_system::AuthorizeCall::<runtime::Runtime>::new(),
989		frame_system::CheckNonZeroSender::<runtime::Runtime>::new(),
990		frame_system::CheckSpecVersion::<runtime::Runtime>::new(),
991		frame_system::CheckGenesis::<runtime::Runtime>::new(),
992		frame_system::CheckEra::<runtime::Runtime>::from(generic::Era::mortal(
993			period,
994			current_block,
995		)),
996		frame_system::CheckNonce::<runtime::Runtime>::from(nonce),
997		frame_system::CheckWeight::<runtime::Runtime>::new(),
998		pallet_transaction_payment::ChargeTransactionPayment::<runtime::Runtime>::from(tip),
999	)
1000		.into();
1001	let raw_payload = runtime::SignedPayload::from_raw(
1002		function.clone(),
1003		tx_ext.clone(),
1004		((), (), runtime::VERSION.spec_version, genesis_block, current_block_hash, (), (), ()),
1005	);
1006	let signature = raw_payload.using_encoded(|e| caller.sign(e));
1007	runtime::UncheckedExtrinsic::new_signed(
1008		function,
1009		MultiAddress::Id(caller.public().into()),
1010		runtime::Signature::Sr25519(signature),
1011		tx_ext,
1012	)
1013}
