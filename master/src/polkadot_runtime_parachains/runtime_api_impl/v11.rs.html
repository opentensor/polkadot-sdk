54/// Implementation for the `validator_groups` function of the runtime API.
55pub fn validator_groups<T: initializer::Config>(
56) -> (Vec<Vec<ValidatorIndex>>, GroupRotationInfo<BlockNumberFor<T>>) {
57	// This formula needs to be the same as the one we use
58	// when populating group_responsible in `availability_cores`
59	let now = frame_system::Pallet::<T>::block_number() + One::one();
60
61	let groups = scheduler::ValidatorGroups::<T>::get();
62	let rotation_info = scheduler::Pallet::<T>::group_rotation_info(now);
63
64	(groups, rotation_info)
65}

67/// Implementation for the `availability_cores` function of the runtime API.
68pub fn availability_cores<T: initializer::Config>() -> Vec<CoreState<T::Hash, BlockNumberFor<T>>> {
69	let time_out_for = scheduler::Pallet::<T>::availability_timeout_predicate();
70
71	let group_responsible_for =
72		|backed_in_number, core_index| match scheduler::Pallet::<T>::group_assigned_to_core(
73			core_index,
74			backed_in_number,
75		) {
76			Some(g) => g,
77			None => {
78				log::warn!(
79					target: "runtime::polkadot-api::v2",
80					"Could not determine the group responsible for core extracted \
81					from list of cores for some prior block in same session",
82				);
83
84				GroupIndex(0)
85			},
86		};
87
88	let claim_queue = scheduler::Pallet::<T>::get_claim_queue();
89	let occupied_cores: BTreeMap<CoreIndex, inclusion::CandidatePendingAvailability<_, _>> =
90		inclusion::Pallet::<T>::get_occupied_cores().collect();
91	let n_cores = scheduler::Pallet::<T>::num_availability_cores();
92
93	(0..n_cores)
94		.map(|core_idx| {
95			let core_idx = CoreIndex(core_idx as u32);
96			if let Some(pending_availability) = occupied_cores.get(&core_idx) {
97				// Use the same block number for determining the responsible group as what the
98				// backing subsystem would use when it calls validator_groups api.
99				let backing_group_allocation_time =
100					pending_availability.relay_parent_number() + One::one();
101				CoreState::Occupied(OccupiedCore {
102					next_up_on_available: scheduler::Pallet::<T>::next_up_on_available(core_idx),
103					occupied_since: pending_availability.backed_in_number(),
104					time_out_at: time_out_for(pending_availability.backed_in_number()).live_until,
105					next_up_on_time_out: scheduler::Pallet::<T>::next_up_on_available(core_idx),
106					availability: pending_availability.availability_votes().clone(),
107					group_responsible: group_responsible_for(
108						backing_group_allocation_time,
109						pending_availability.core_occupied(),
110					),
111					candidate_hash: pending_availability.candidate_hash(),
112					candidate_descriptor: pending_availability.candidate_descriptor().clone(),
113				})
114			} else {
115				if let Some(assignment) = claim_queue.get(&core_idx).and_then(|q| q.front()) {
116					CoreState::Scheduled(polkadot_primitives::ScheduledCore {
117						para_id: assignment.para_id(),
118						collator: None,
119					})
120				} else {
121					CoreState::Free
122				}
123			}
124		})
125		.collect()
126}

128/// Returns current block number being processed and the corresponding root hash.
129fn current_relay_parent<T: frame_system::Config>(
130) -> (BlockNumberFor<T>, <T as frame_system::Config>::Hash) {
131	use codec::Decode as _;
132	let state_version = frame_system::Pallet::<T>::runtime_version().state_version();
133	let relay_parent_number = frame_system::Pallet::<T>::block_number();
134	let relay_parent_storage_root = T::Hash::decode(&mut &sp_io::storage::root(state_version)[..])
135		.expect("storage root must decode to the Hash type; qed");
136	(relay_parent_number, relay_parent_storage_root)
137}

139fn with_assumption<Config, T, F>(
140	para_id: ParaId,
141	assumption: OccupiedCoreAssumption,
142	build: F,
143) -> Option<T>
144where
145	Config: inclusion::Config,
146	F: FnOnce() -> Option<T>,
147{
148	match assumption {
149		OccupiedCoreAssumption::Included => {
150			<inclusion::Pallet<Config>>::force_enact(para_id);
151			build()
152		},
153		OccupiedCoreAssumption::TimedOut => build(),
154		OccupiedCoreAssumption::Free =>
155			if !<inclusion::Pallet<Config>>::candidates_pending_availability(para_id).is_empty() {
156				None
157			} else {
158				build()
159			},
160	}
161}

163/// Implementation for the `persisted_validation_data` function of the runtime API.
164pub fn persisted_validation_data<T: initializer::Config>(
165	para_id: ParaId,
166	assumption: OccupiedCoreAssumption,

168	let (relay_parent_number, relay_parent_storage_root) = current_relay_parent::<T>();

170		crate::util::make_persisted_validation_data::<T>(
171			para_id,
172			relay_parent_number,
173			relay_parent_storage_root,
174		)
175	})

178/// Implementation for the `assumed_validation_data` function of the runtime API.
179pub fn assumed_validation_data<T: initializer::Config>(
180	para_id: ParaId,
181	expected_persisted_validation_data_hash: Hash,
182) -> Option<(PersistedValidationData<T::Hash, BlockNumberFor<T>>, ValidationCodeHash)> {
183	let (relay_parent_number, relay_parent_storage_root) = current_relay_parent::<T>();
184	// This closure obtains the `persisted_validation_data` for the given `para_id` and matches
185	// its hash against an expected one.
186	let make_validation_data = || {
187		crate::util::make_persisted_validation_data::<T>(
188			para_id,
189			relay_parent_number,
190			relay_parent_storage_root,
191		)
192		.filter(|validation_data| validation_data.hash() == expected_persisted_validation_data_hash)
193	};
194
195	let persisted_validation_data = make_validation_data().or_else(|| {
196		// Try again with force enacting the pending candidates. This check only makes sense if
197		// there are any pending candidates.
198		(!inclusion::Pallet::<T>::candidates_pending_availability(para_id).is_empty())
199			.then_some(())
200			.and_then(|_| {
201				inclusion::Pallet::<T>::force_enact(para_id);
202				make_validation_data()
203			})
204	});
205	// If we were successful, also query current validation code hash.
206	persisted_validation_data.zip(paras::CurrentCodeHash::<T>::get(&para_id))
207}

234/// Implementation for the `AuthorityDiscoveryApi::authorities()` function of the runtime API.
235/// It is a heavy call, but currently only used for authority discovery, so it is fine.
236/// Gets next, current and some historical authority ids using `session_info` module.
237pub fn relevant_authority_ids<T: initializer::Config + pallet_authority_discovery::Config>(
238) -> Vec<AuthorityDiscoveryId> {
239	let current_session_index = session_index_for_child::<T>();
240	let earliest_stored_session = session_info::EarliestStoredSession::<T>::get();
241
242	// Due to `max_validators`, the `SessionInfo` stores only the validators who are actively
243	// selected to participate in parachain consensus. We'd like all authorities for the current
244	// and next sessions to be used in authority-discovery. The two sets likely have large overlap.
245	let mut authority_ids = pallet_authority_discovery::Pallet::<T>::current_authorities().to_vec();
246	authority_ids.extend(pallet_authority_discovery::Pallet::<T>::next_authorities().to_vec());
247
248	// Due to disputes, we'd like to remain connected to authorities of the previous few sessions.
249	// For this, we don't need anyone other than the validators actively participating in consensus.
250	for session_index in earliest_stored_session..current_session_index {
251		let info = session_info::Sessions::<T>::get(session_index);
252		if let Some(mut info) = info {
253			authority_ids.append(&mut info.discovery_keys);
254		}
255	}
256
257	authority_ids.sort();
258	authority_ids.dedup();
259
260	authority_ids
261}

282/// Implementation for the `candidate_events` function of the runtime API.
283// NOTE: this runs without block initialization, as it accesses events.
284// this means it can run in a different session than other runtime APIs at the same block.
285pub fn candidate_events<T, F>(extract_event: F) -> Vec<CandidateEvent<T::Hash>>
286where
287	T: initializer::Config,
288	F: Fn(<T as frame_system::Config>::RuntimeEvent) -> Option<inclusion::Event<T>>,
289{
290	use inclusion::Event as RawEvent;
291
292	frame_system::Pallet::<T>::read_events_no_consensus()
293		.into_iter()
294		.filter_map(|record| extract_event(record.event))
295		.filter_map(|event| {
296			Some(match event {
297				RawEvent::<T>::CandidateBacked(c, h, core, group) =>
298					CandidateEvent::CandidateBacked(c, h, core, group),
299				RawEvent::<T>::CandidateIncluded(c, h, core, group) =>
300					CandidateEvent::CandidateIncluded(c, h, core, group),
301				RawEvent::<T>::CandidateTimedOut(c, h, core) =>
302					CandidateEvent::CandidateTimedOut(c, h, core),
303				// Not needed for candidate events.
304				RawEvent::<T>::UpwardMessagesReceived { .. } => return None,
305				RawEvent::<T>::__Ignore(_, _) => unreachable!("__Ignore cannot be used"),
306			})
307		})
308		.collect()
309}

404// Helper function that returns the backing constraints given a parachain id.
405pub(crate) fn backing_constraints<T: initializer::Config>(
406	para_id: ParaId,
407) -> Option<Constraints<BlockNumberFor<T>>> {
408	let config = configuration::ActiveConfig::<T>::get();
409	// Async backing is only expected to be enabled with a tracker capacity of 1.
410	// Subsequent configuration update gets applied on new session, which always
411	// clears the buffer.
412	//
413	// Thus, minimum relay parent is ensured to have asynchronous backing enabled.
414	let now = frame_system::Pallet::<T>::block_number();
415
416	// Use the right storage depending on version to ensure #64 doesn't cause issues with this
417	// migration.
418	let min_relay_parent_number = if shared::Pallet::<T>::on_chain_storage_version() ==
419		StorageVersion::new(0)
420	{
421		shared::migration::v0::AllowedRelayParents::<T>::get().hypothetical_earliest_block_number(
422			now,
423			config.scheduler_params.lookahead.saturating_sub(1),
424		)
425	} else {
426		shared::AllowedRelayParents::<T>::get().hypothetical_earliest_block_number(
427			now,
428			config.scheduler_params.lookahead.saturating_sub(1),
429		)
430	};
431
432	let required_parent = paras::Heads::<T>::get(para_id)?;
433	let validation_code_hash = paras::CurrentCodeHash::<T>::get(para_id)?;
434
435	let upgrade_restriction = paras::UpgradeRestrictionSignal::<T>::get(para_id);
436	let future_validation_code =
437		paras::FutureCodeUpgrades::<T>::get(para_id).and_then(|block_num| {
438			// Only read the storage if there's a pending upgrade.
439			Some(block_num).zip(paras::FutureCodeHash::<T>::get(para_id))
440		});
441
442	let (ump_msg_count, ump_total_bytes) =
443		inclusion::Pallet::<T>::relay_dispatch_queue_size(para_id);
444	let ump_remaining = config.max_upward_queue_count - ump_msg_count;
445	let ump_remaining_bytes = config.max_upward_queue_size - ump_total_bytes;
446
447	let dmp_remaining_messages = dmp::Pallet::<T>::dmq_contents(para_id)
448		.into_iter()
449		.map(|msg| msg.sent_at)
450		.collect();
451
452	let valid_watermarks = hrmp::Pallet::<T>::valid_watermarks(para_id);
453	let hrmp_inbound = InboundHrmpLimitations { valid_watermarks };
454	let hrmp_channels_out = hrmp::Pallet::<T>::outbound_remaining_capacity(para_id)
455		.into_iter()
456		.map(|(para, (messages_remaining, bytes_remaining))| {
457			(para, OutboundHrmpChannelLimitations { messages_remaining, bytes_remaining })
458		})
459		.collect();
460
461	Some(Constraints {
462		min_relay_parent_number,
463		max_pov_size: config.max_pov_size,
464		max_code_size: config.max_code_size,
465		ump_remaining,
466		ump_remaining_bytes,
467		max_ump_num_per_candidate: config.max_upward_message_num_per_candidate,
468		dmp_remaining_messages,
469		hrmp_inbound,
470		hrmp_channels_out,
471		max_hrmp_num_per_candidate: config.hrmp_max_message_num_per_candidate,
472		required_parent,
473		validation_code_hash,
474		upgrade_restriction,
475		future_validation_code,
476	})
477}

479/// Implementation for `ParaBackingState` function from the runtime API
480#[deprecated(note = "`backing_state` will be removed. Use `backing_constraints` and
481	`candidates_pending_availability` instead.")]
482pub fn backing_state<T: initializer::Config>(
483	para_id: ParaId,
484) -> Option<BackingState<T::Hash, BlockNumberFor<T>>> {
485	let constraints = backing_constraints::<T>(para_id)?;
486
487	let pending_availability = {
488		crate::inclusion::PendingAvailability::<T>::get(&para_id)
489			.map(|pending_candidates| {
490				pending_candidates
491					.into_iter()
492					.map(|candidate| {
493						CandidatePendingAvailability {
494							candidate_hash: candidate.candidate_hash(),
495							descriptor: candidate.candidate_descriptor().clone(),
496							commitments: candidate.candidate_commitments().clone(),
497							relay_parent_number: candidate.relay_parent_number(),
498							max_pov_size: constraints.max_pov_size, /* assume always same in
499							                                         * session. */
500						}
501					})
502					.collect()
503			})
504			.unwrap_or_else(|| vec![])
505	};
506
507	Some(BackingState { constraints, pending_availability })
508}

536/// Returns the claimqueue from the scheduler
537pub fn claim_queue<T: scheduler::Config>() -> BTreeMap<CoreIndex, VecDeque<ParaId>> {
538	let config = configuration::ActiveConfig::<T>::get();
539	// Extra sanity, config should already never be smaller than 1:
540	let n_lookahead = config.scheduler_params.lookahead.max(1);
541	scheduler::Pallet::<T>::get_claim_queue()
542		.into_iter()
543		.map(|(core_index, entries)| {
544			(
545				core_index,
546				entries.into_iter().map(|e| e.para_id()).take(n_lookahead as usize).collect(),
547			)
548		})
549		.collect()
550}
