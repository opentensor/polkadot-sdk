95/// Time after starting a collation download from a collator we will start another one from the
96/// next collator even if the upload was not finished yet.
97///
98/// This is to protect from a single slow collator preventing collations from happening.
99///
100/// With a collation size of 5MB and bandwidth of 500Mbit/s (requirement for Kusama validators),
101/// the transfer should be possible within 0.1 seconds. 400 milliseconds should therefore be
102/// plenty, even with multiple heads and should be low enough for later collators to still be able
103/// to finish on time.
104///
105/// There is debug logging output, so we can adjust this value based on production results.
106#[cfg(not(test))]
107const MAX_UNSHARED_DOWNLOAD_TIME: Duration = Duration::from_millis(400);
108

141enum InsertAdvertisementError {
142	/// Advertisement is already known.
143	Duplicate,
144	/// Collation relay parent is out of our view.

159impl PeerData {
160	/// Update the view, clearing all advertisements that are no longer in the
161	/// current view.
162	fn update_view(
163		&mut self,
164		implicit_view: &ImplicitView,
165		active_leaves: &HashSet<Hash>,
166		new_view: View,
167	) {
168		let old_view = std::mem::replace(&mut self.view, new_view);
169		if let PeerState::Collating(ref mut peer_state) = self.state {
170			for removed in old_view.difference(&self.view) {
171				// Remove relay parent advertisements if it went out of our (implicit) view.
172				let keep = is_relay_parent_in_implicit_view(
173					removed,
174					implicit_view,
175					active_leaves,
176					peer_state.para_id,
177				);
178
179				if !keep {
180					peer_state.advertisements.remove(&removed);
181				}
182			}
183		}
184	}
185
186	/// Prune old advertisements relative to our view.
187	fn prune_old_advertisements(
188		&mut self,
189		implicit_view: &ImplicitView,
190		active_leaves: &HashSet<Hash>,
191	) {
192		if let PeerState::Collating(ref mut peer_state) = self.state {
193			peer_state.advertisements.retain(|hash, _| {
194				// Either
195				// - Relay parent is an active leaf
196				// - It belongs to allowed ancestry under some leaf
197				// Discard otherwise.
198				is_relay_parent_in_implicit_view(
199					hash,
200					implicit_view,
201					active_leaves,
202					peer_state.para_id,
203				)
204			});
205		}
206	}
207
208	/// Performs sanity check for an advertisement and notes it as advertised.
209	fn insert_advertisement(
210		&mut self,
211		on_relay_parent: Hash,
212		candidate_hash: Option<CandidateHash>,
213		implicit_view: &ImplicitView,
214		active_leaves: &HashSet<Hash>,
215		per_relay_parent: &PerRelayParent,
216	) -> std::result::Result<(CollatorId, ParaId), InsertAdvertisementError> {
217		match self.state {
218			PeerState::Connected(_) => Err(InsertAdvertisementError::UndeclaredCollator),
219			PeerState::Collating(ref mut state) => {
220				if !is_relay_parent_in_implicit_view(
221					&on_relay_parent,
222					implicit_view,
223					active_leaves,
224					state.para_id,
225				) {
226					return Err(InsertAdvertisementError::OutOfOurView)
227				}
228
229				if let Some(candidate_hash) = candidate_hash {
230					if state
231						.advertisements
232						.get(&on_relay_parent)
233						.map_or(false, |candidates| candidates.contains(&candidate_hash))
234					{
235						return Err(InsertAdvertisementError::Duplicate)
236					}
237
238					let candidates = state.advertisements.entry(on_relay_parent).or_default();
239
240					// Current assignments is equal to the length of the claim queue. No honest
241					// collator should send that many advertisements.
242					if candidates.len() > per_relay_parent.assignment.current.len() {
243						return Err(InsertAdvertisementError::PeerLimitReached)
244					}
245
246					candidates.insert(candidate_hash);
247				} else {
248					if self.version != CollationVersion::V1 {
249						gum::error!(
250							target: LOG_TARGET,
251							"Programming error, `candidate_hash` can not be `None` \
252							 for non `V1` networking.",
253						);
254					}
255
256					if state.advertisements.contains_key(&on_relay_parent) {
257						return Err(InsertAdvertisementError::Duplicate)
258					}
259
260					state
261						.advertisements
262						.insert(on_relay_parent, HashSet::from_iter(candidate_hash));
263				};
264
265				state.last_active = Instant::now();
266				Ok((state.collator_id.clone(), state.para_id))
267			},
268		}
269	}
270
271	/// Whether a peer is collating.
272	fn is_collating(&self) -> bool {
273		match self.state {
274			PeerState::Connected(_) => false,
275			PeerState::Collating(_) => true,
276		}
277	}
278
279	/// Note that a peer is now collating with the given collator and para id.
280	///
281	/// This will overwrite any previous call to `set_collating` and should only be called
282	/// if `is_collating` is false.
283	fn set_collating(&mut self, collator_id: CollatorId, para_id: ParaId) {
284		self.state = PeerState::Collating(CollatingPeerState {
285			collator_id,
286			para_id,
287			advertisements: HashMap::new(),
288			last_active: Instant::now(),
289		});
290	}
291
292	fn collator_id(&self) -> Option<&CollatorId> {
293		match self.state {
294			PeerState::Connected(_) => None,
295			PeerState::Collating(ref state) => Some(&state.collator_id),
296		}
297	}
298
299	fn collating_para(&self) -> Option<ParaId> {
300		match self.state {
301			PeerState::Connected(_) => None,
302			PeerState::Collating(ref state) => Some(state.para_id),
303		}
304	}
305
306	/// Whether the peer has advertised the given collation.
307	fn has_advertised(
308		&self,
309		relay_parent: &Hash,
310		maybe_candidate_hash: Option<CandidateHash>,
311	) -> bool {
312		let collating_state = match self.state {
313			PeerState::Connected(_) => return false,
314			PeerState::Collating(ref state) => state,
315		};
316
317		if let Some(ref candidate_hash) = maybe_candidate_hash {
318			collating_state
319				.advertisements
320				.get(relay_parent)
321				.map_or(false, |candidates| candidates.contains(candidate_hash))
322		} else {
323			collating_state.advertisements.contains_key(relay_parent)
324		}
325	}
326
327	/// Whether the peer is now inactive according to the current instant and the eviction policy.
328	fn is_inactive(&self, policy: &crate::CollatorEvictionPolicy) -> bool {
329		match self.state {
330			PeerState::Connected(connected_at) => connected_at.elapsed() >= policy.undeclared,
331			PeerState::Collating(ref state) =>
332				state.last_active.elapsed() >= policy.inactive_collator,
333		}
334	}
335}

351/// All state relevant for the validator side of the protocol lives here.
352#[derive(Default)]
353struct State {
354	/// Leaves that do support asynchronous backing along with
355	/// implicit ancestry. Leaves from the implicit view are present in
356	/// `active_leaves`, the opposite doesn't hold true.
357	///
358	/// Relay-chain blocks which don't support prospective parachains are
359	/// never included in the fragment chains of active leaves which do. In
360	/// particular, this means that if a given relay parent belongs to implicit
361	/// ancestry of some active leaf, then it does support prospective parachains.
362	implicit_view: ImplicitView,
363
364	/// All active leaves observed by us. This works as a replacement for
365	/// [`polkadot_node_network_protocol::View`] and can be dropped once the transition
366	/// to asynchronous backing is done.
367	active_leaves: HashSet<Hash>,
368
369	/// State tracked per relay parent.
370	per_relay_parent: HashMap<Hash, PerRelayParent>,
371
372	/// Track all active collators and their data.
373	peer_data: HashMap<PeerId, PeerData>,
374
375	/// Parachains we're currently assigned to. With async backing enabled
376	/// this includes assignments from the implicit view.
377	current_assignments: HashMap<ParaId, usize>,
378
379	/// The collations we have requested from collators.
380	collation_requests: FuturesUnordered<CollationFetchRequest>,
381
382	/// Cancellation handles for the collation fetch requests.
383	collation_requests_cancel_handles: HashMap<PendingCollation, CancellationToken>,
384
385	/// Metrics.
386	metrics: Metrics,
387
388	/// When a timer in this `FuturesUnordered` triggers, we should dequeue the next request
389	/// attempt in the corresponding `collations_per_relay_parent`.
390	///
391	/// A triggering timer means that the fetching took too long for our taste and we should give
392	/// another collator the chance to be faster (dequeue next fetch request as well).
393	collation_fetch_timeouts:
394		FuturesUnordered<BoxFuture<'static, (CollatorId, Option<CandidateHash>, Hash)>>,
395
396	/// Collations that we have successfully requested from peers and waiting
397	/// on validation.
398	fetched_candidates: HashMap<FetchedCollation, CollationEvent>,
399
400	/// Collations which we haven't been able to second due to their parent not being known by
401	/// prospective-parachains. Mapped from the paraid and parent_head_hash to the fetched
402	/// collation data. Only needed for async backing. For elastic scaling, the fetched collation
403	/// must contain the full parent head data.
404	blocked_from_seconding: HashMap<BlockedCollationId, Vec<PendingCollationFetch>>,
405
406	/// Aggregated reputation change
407	reputation: ReputationAggregator,
408}
409
410impl State {
411	// Returns the number of seconded and pending collations for a specific `ParaId`. Pending
412	// collations are:
413	// 1. Collations being fetched from a collator.
414	// 2. Collations waiting for validation from backing subsystem.
415	// 3. Collations blocked from seconding due to parent not being known by backing subsystem.
416	fn seconded_and_pending_for_para(&self, relay_parent: &Hash, para_id: &ParaId) -> usize {
417		let seconded = self
418			.per_relay_parent
419			.get(relay_parent)
420			.map_or(0, |per_relay_parent| per_relay_parent.collations.seconded_for_para(para_id));
421
422		let pending_fetch = self.per_relay_parent.get(relay_parent).map_or(0, |rp_state| {
423			match rp_state.collations.status {
424				CollationStatus::Fetching(pending_para_id) if pending_para_id == *para_id => 1,
425				_ => 0,
426			}
427		});
428
429		let waiting_for_validation = self
430			.fetched_candidates
431			.keys()
432			.filter(|fc| fc.relay_parent == *relay_parent && fc.para_id == *para_id)
433			.count();
434
435		let blocked_from_seconding =
436			self.blocked_from_seconding.values().fold(0, |acc, blocked_collations| {
437				acc + blocked_collations
438					.iter()
439					.filter(|pc| {
440						pc.candidate_receipt.descriptor.para_id() == *para_id &&
441							pc.candidate_receipt.descriptor.relay_parent() == *relay_parent
442					})
443					.count()
444			});
445
446		gum::trace!(
447			target: LOG_TARGET,
448			?relay_parent,
449			?para_id,
450			seconded,
451			pending_fetch,
452			waiting_for_validation,
453			blocked_from_seconding,
454			"Seconded and pending collations for para",
455		);
456
457		seconded + pending_fetch + waiting_for_validation + blocked_from_seconding
458	}
459}

461fn is_relay_parent_in_implicit_view(
462	relay_parent: &Hash,
463	implicit_view: &ImplicitView,
464	active_leaves: &HashSet<Hash>,
465	para_id: ParaId,
466) -> bool {
467	active_leaves.iter().any(|hash| {
468		implicit_view
469			.known_allowed_relay_parents_under(hash, Some(para_id))
470			.unwrap_or_default()
471			.contains(relay_parent)
472	})
473}

475async fn construct_per_relay_parent<Sender>(
476	sender: &mut Sender,
477	current_assignments: &mut HashMap<ParaId, usize>,
478	keystore: &KeystorePtr,
479	relay_parent: Hash,
480	v2_receipts: bool,
481	session_index: SessionIndex,
482) -> Result<Option<PerRelayParent>>
483where
484	Sender: CollatorProtocolSenderTrait,
485{
486	let validators = polkadot_node_subsystem_util::request_validators(relay_parent, sender)
487		.await
488		.await
489		.map_err(Error::CancelledActiveValidators)??;
490
491	let (groups, rotation_info) =
492		polkadot_node_subsystem_util::request_validator_groups(relay_parent, sender)
493			.await
494			.await
495			.map_err(Error::CancelledValidatorGroups)??;
496
497	let core_now = if let Some(group) =
498		polkadot_node_subsystem_util::signing_key_and_index(&validators, keystore).and_then(
499			|(_, index)| polkadot_node_subsystem_util::find_validator_group(&groups, index),
500		) {
501		rotation_info.core_for_group(group, groups.len())
502	} else {
503		gum::trace!(target: LOG_TARGET, ?relay_parent, "Not a validator");
504		return Ok(None)
505	};
506
507	let mut claim_queue = request_claim_queue(relay_parent, sender)
508		.await
509		.await
510		.map_err(Error::CancelledClaimQueue)??;
511
512	let assigned_paras = claim_queue.remove(&core_now).unwrap_or_else(|| VecDeque::new());
513
514	for para_id in assigned_paras.iter() {
515		let entry = current_assignments.entry(*para_id).or_default();
516		*entry += 1;
517		if *entry == 1 {
518			gum::debug!(
519				target: LOG_TARGET,
520				?relay_parent,
521				para_id = ?para_id,
522				"Assigned to a parachain",
523			);
524		}
525	}
526
527	let assignment = GroupAssignments { current: assigned_paras.into_iter().collect() };
528	let collations = Collations::new(&assignment.current);
529
530	Ok(Some(PerRelayParent {
531		assignment,
532		collations,
533		v2_receipts,
534		session_index,
535		current_core: core_now,
536	}))
537}

539fn remove_outgoing(
540	current_assignments: &mut HashMap<ParaId, usize>,
541	per_relay_parent: PerRelayParent,
542) {
543	let GroupAssignments { current, .. } = per_relay_parent.assignment;
544
545	for cur in current {
546		if let Entry::Occupied(mut occupied) = current_assignments.entry(cur) {
547			*occupied.get_mut() -= 1;
548			if *occupied.get() == 0 {
549				occupied.remove_entry();
550				gum::debug!(
551					target: LOG_TARGET,
552					para_id = ?cur,
553					"Unassigned from a parachain",
554				);
555			}
556		}
557	}
558}

577/// Another subsystem has requested to fetch collations on a particular leaf for some para.
578async fn fetch_collation(
579	sender: &mut impl overseer::CollatorProtocolSenderTrait,
580	state: &mut State,
581	pc: PendingCollation,
582	id: CollatorId,
583) -> std::result::Result<(), FetchError> {
584	let PendingCollation { relay_parent, peer_id, prospective_candidate, .. } = pc;
585	let candidate_hash = prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
586
587	let peer_data = state.peer_data.get(&peer_id).ok_or(FetchError::UnknownPeer)?;
588
589	if peer_data.has_advertised(&relay_parent, candidate_hash) {
590		request_collation(sender, state, pc, id.clone(), peer_data.version).await?;
591		let timeout = |collator_id, candidate_hash, relay_parent| async move {
592			Delay::new(MAX_UNSHARED_DOWNLOAD_TIME).await;
593			(collator_id, candidate_hash, relay_parent)
594		};
595		state
596			.collation_fetch_timeouts
597			.push(timeout(id.clone(), candidate_hash, relay_parent).boxed());
598
599		Ok(())
600	} else {
601		Err(FetchError::NotAdvertised)
602	}
603}

629/// Notify a collator that its collation got seconded.
630async fn notify_collation_seconded(
631	sender: &mut impl overseer::CollatorProtocolSenderTrait,
632	peer_id: PeerId,
633	version: CollationVersion,
634	relay_parent: Hash,
635	statement: SignedFullStatement,
636) {
637	let statement = statement.into();
638	let wire_message = match version {
639		CollationVersion::V1 =>
640			CollationProtocols::V1(protocol_v1::CollationProtocol::CollatorProtocol(
641				protocol_v1::CollatorProtocolMessage::CollationSeconded(relay_parent, statement),
642			)),
643		CollationVersion::V2 =>
644			CollationProtocols::V2(protocol_v2::CollationProtocol::CollatorProtocol(
645				protocol_v2::CollatorProtocolMessage::CollationSeconded(relay_parent, statement),
646			)),
647	};
648	sender
649		.send_message(NetworkBridgeTxMessage::SendCollationMessage(vec![peer_id], wire_message))
650		.await;
651}

653/// A peer's view has changed. A number of things should be done:
654///  - Ongoing collation requests have to be canceled.
655///  - Advertisements by this peer that are no longer relevant have to be removed.
656fn handle_peer_view_change(state: &mut State, peer_id: PeerId, view: View) {
657	let peer_data = match state.peer_data.get_mut(&peer_id) {
658		Some(peer_data) => peer_data,
659		None => return,
660	};
661
662	peer_data.update_view(&state.implicit_view, &state.active_leaves, view);
663	state.collation_requests_cancel_handles.retain(|pc, handle| {
664		let keep = pc.peer_id != peer_id || peer_data.has_advertised(&pc.relay_parent, None);
665		if !keep {
666			handle.cancel();
667		}
668		keep
669	});
670}

672/// Request a collation from the network.
673/// This function will
674///  - Check for duplicate requests.
675///  - Check if the requested collation is in our view.
676/// And as such invocations of this function may rely on that.
677async fn request_collation(
678	sender: &mut impl overseer::CollatorProtocolSenderTrait,
679	state: &mut State,
680	pending_collation: PendingCollation,
681	collator_id: CollatorId,
682	peer_protocol_version: CollationVersion,
683) -> std::result::Result<(), FetchError> {
684	if state.collation_requests_cancel_handles.contains_key(&pending_collation) {
685		return Err(FetchError::AlreadyRequested)
686	}
687
688	let PendingCollation { relay_parent, para_id, peer_id, prospective_candidate, .. } =
689		pending_collation;
690	let per_relay_parent = state
691		.per_relay_parent
692		.get_mut(&relay_parent)
693		.ok_or(FetchError::RelayParentOutOfView)?;
694
695	let (requests, response_recv) = match (peer_protocol_version, prospective_candidate) {
696		(CollationVersion::V1, _) => {
697			let (req, response_recv) = OutgoingRequest::new(
698				Recipient::Peer(peer_id),
699				request_v1::CollationFetchingRequest { relay_parent, para_id },
700			);
701			let requests = Requests::CollationFetchingV1(req);
702			(requests, response_recv.boxed())
703		},
704		(CollationVersion::V2, Some(ProspectiveCandidate { candidate_hash, .. })) => {
705			let (req, response_recv) = OutgoingRequest::new(
706				Recipient::Peer(peer_id),
707				request_v2::CollationFetchingRequest { relay_parent, para_id, candidate_hash },
708			);
709			let requests = Requests::CollationFetchingV2(req);
710			(requests, response_recv.boxed())
711		},
712		_ => return Err(FetchError::ProtocolMismatch),
713	};
714
715	let cancellation_token = CancellationToken::new();
716	let collation_request = CollationFetchRequest {
717		pending_collation,
718		collator_id: collator_id.clone(),
719		collator_protocol_version: peer_protocol_version,
720		from_collator: response_recv,
721		cancellation_token: cancellation_token.clone(),
722		_lifetime_timer: state.metrics.time_collation_request_duration(),
723	};
724
725	state.collation_requests.push(collation_request);
726	state
727		.collation_requests_cancel_handles
728		.insert(pending_collation, cancellation_token);
729
730	gum::debug!(
731		target: LOG_TARGET,
732		peer_id = %peer_id,
733		%para_id,
734		?relay_parent,
735		"Requesting collation",
736	);
737
738	let maybe_candidate_hash =
739		prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
740	per_relay_parent.collations.status = CollationStatus::Fetching(para_id);
741	per_relay_parent
742		.collations
743		.fetching_from
744		.replace((collator_id, maybe_candidate_hash));
745
746	sender
747		.send_message(NetworkBridgeTxMessage::SendRequests(
748			vec![requests],
749			IfDisconnected::ImmediateError,
750		))
751		.await;
752	Ok(())
753}

755/// Networking message has been received.
756#[overseer::contextbounds(CollatorProtocol, prefix = overseer)]
757async fn process_incoming_peer_message<Context>(
758	ctx: &mut Context,
759	state: &mut State,
760	origin: PeerId,
761	msg: CollationProtocols<
762		protocol_v1::CollatorProtocolMessage,
763		protocol_v2::CollatorProtocolMessage,
764	>,
765) {
766	use protocol_v1::CollatorProtocolMessage as V1;
767	use protocol_v2::CollatorProtocolMessage as V2;
768	use sp_runtime::traits::AppVerify;
769
770	match msg {
771		CollationProtocols::V1(V1::Declare(collator_id, para_id, signature)) |
772		CollationProtocols::V2(V2::Declare(collator_id, para_id, signature)) => {
773			if collator_peer_id(&state.peer_data, &collator_id).is_some() {
774				modify_reputation(
775					&mut state.reputation,
776					ctx.sender(),
777					origin,
778					COST_UNEXPECTED_MESSAGE,
779				)
780				.await;
781				return
782			}
783
784			let peer_data = match state.peer_data.get_mut(&origin) {
785				Some(p) => p,
786				None => {
787					gum::debug!(
788						target: LOG_TARGET,
789						peer_id = ?origin,
790						?para_id,
791						"Unknown peer",
792					);
793					modify_reputation(
794						&mut state.reputation,
795						ctx.sender(),
796						origin,
797						COST_UNEXPECTED_MESSAGE,
798					)
799					.await;
800					return
801				},
802			};
803
804			if peer_data.is_collating() {
805				gum::debug!(
806					target: LOG_TARGET,
807					peer_id = ?origin,
808					?para_id,
809					"Peer is already in the collating state",
810				);
811				modify_reputation(
812					&mut state.reputation,
813					ctx.sender(),
814					origin,
815					COST_UNEXPECTED_MESSAGE,
816				)
817				.await;
818				return
819			}
820
821			if !signature.verify(&*protocol_v1::declare_signature_payload(&origin), &collator_id) {
822				gum::debug!(
823					target: LOG_TARGET,
824					peer_id = ?origin,
825					?para_id,
826					"Signature verification failure",
827				);
828				modify_reputation(
829					&mut state.reputation,
830					ctx.sender(),
831					origin,
832					COST_INVALID_SIGNATURE,
833				)
834				.await;
835				return
836			}
837
838			if state.current_assignments.contains_key(&para_id) {
839				gum::debug!(
840					target: LOG_TARGET,
841					peer_id = ?origin,
842					?collator_id,
843					?para_id,
844					"Declared as collator for current para",
845				);
846
847				peer_data.set_collating(collator_id, para_id);
848			} else {
849				gum::debug!(
850					target: LOG_TARGET,
851					peer_id = ?origin,
852					?collator_id,
853					?para_id,
854					"Declared as collator for unneeded para. Current assignments: {:?}",
855					&state.current_assignments
856				);
857
858				modify_reputation(
859					&mut state.reputation,
860					ctx.sender(),
861					origin,
862					COST_UNNEEDED_COLLATOR,
863				)
864				.await;
865				gum::trace!(target: LOG_TARGET, "Disconnecting unneeded collator");
866				disconnect_peer(ctx.sender(), origin).await;
867			}
868		},
869		CollationProtocols::V1(V1::AdvertiseCollation(relay_parent)) =>
870			if let Err(err) =
871				handle_advertisement(ctx.sender(), state, relay_parent, origin, None).await
872			{
873				gum::debug!(
874					target: LOG_TARGET,
875					peer_id = ?origin,
876					?relay_parent,
877					error = ?err,
878					"Rejected v1 advertisement",
879				);
880
881				if let Some(rep) = err.reputation_changes() {
882					modify_reputation(&mut state.reputation, ctx.sender(), origin, rep).await;
883				}
884			},
885		CollationProtocols::V2(V2::AdvertiseCollation {
886			relay_parent,
887			candidate_hash,
888			parent_head_data_hash,
889		}) => {
890			if let Err(err) = handle_advertisement(
891				ctx.sender(),
892				state,
893				relay_parent,
894				origin,
895				Some((candidate_hash, parent_head_data_hash)),
896			)
897			.await
898			{
899				gum::debug!(
900					target: LOG_TARGET,
901					peer_id = ?origin,
902					?relay_parent,
903					?candidate_hash,
904					error = ?err,
905					"Rejected v2 advertisement",
906				);
907
908				if let Some(rep) = err.reputation_changes() {
909					modify_reputation(&mut state.reputation, ctx.sender(), origin, rep).await;
910				}
911			}
912		},
913		CollationProtocols::V1(V1::CollationSeconded(..)) |
914		CollationProtocols::V2(V2::CollationSeconded(..)) => {
915			gum::warn!(
916				target: LOG_TARGET,
917				peer_id = ?origin,
918				"Unexpected `CollationSeconded` message, decreasing reputation",
919			);
920
921			modify_reputation(&mut state.reputation, ctx.sender(), origin, COST_UNEXPECTED_MESSAGE)
922				.await;
923		},
924	}
925}

927#[derive(Debug)]
928enum AdvertisementError {
929	/// Relay parent is unknown.
930	RelayParentUnknown,
931	/// Peer is not present in the subsystem state.
932	UnknownPeer,
933	/// Peer has not declared its para id.
934	UndeclaredCollator,
935	/// We're assigned to a different para at the given relay parent.
936	InvalidAssignment,
937	/// Para reached a limit of seconded candidates for this relay parent.
938	SecondedLimitReached,
939	/// Collator trying to advertise a collation using V1 protocol for an async backing relay
940	/// parent.
941	ProtocolMisuse,
942	/// Advertisement is invalid.
943	#[allow(dead_code)]
944	Invalid(InsertAdvertisementError),
945	/// Seconding not allowed by backing subsystem
946	BlockedByBacking,
947}
948
949impl AdvertisementError {
950	fn reputation_changes(&self) -> Option<Rep> {
951		use AdvertisementError::*;
952		match self {
953			InvalidAssignment => Some(COST_WRONG_PARA),
954			ProtocolMisuse => Some(COST_PROTOCOL_MISUSE),
955			RelayParentUnknown | UndeclaredCollator | Invalid(_) => Some(COST_UNEXPECTED_MESSAGE),
956			UnknownPeer | SecondedLimitReached | BlockedByBacking => None,
957		}
958	}
959}

962async fn can_second<Sender>(
963	sender: &mut Sender,
964	candidate_para_id: ParaId,
965	candidate_relay_parent: Hash,
966	candidate_hash: CandidateHash,

996#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]

998	ctx: &mut Context,
999	state: &mut State,
1000	para_id: ParaId,
1001	head_data: HeadData,
1002	head_data_hash: Hash,
1003) {

1005		.blocked_from_seconding
1006		.remove(&BlockedCollationId { para_id, parent_head_data_hash: head_data_hash })
1007	{
1008		if !unblocked_collations.is_empty() {
1009			gum::debug!(
1010				target: LOG_TARGET,
1011				"Candidate outputting head data with hash {} unblocked {} collations for seconding.",
1012				head_data_hash,
1013				unblocked_collations.len()
1014			);
1015		}
1016
1017		for mut unblocked_collation in unblocked_collations {
1018			unblocked_collation.maybe_parent_head_data = Some(head_data.clone());
1019			let peer_id = unblocked_collation.collation_event.pending_collation.peer_id;
1020			let relay_parent = unblocked_collation.candidate_receipt.descriptor.relay_parent();

1030				);

1034					modify_reputation(
1035						&mut state.reputation,
1036						ctx.sender(),
1037						peer_id,

1039					)

1041				}
1042			}
1043		}
1044	}

1047fn ensure_seconding_limit_is_respected(
1048	relay_parent: &Hash,
1049	para_id: ParaId,
1050	state: &State,
1051) -> std::result::Result<(), AdvertisementError> {
1052	let paths = state.implicit_view.paths_via_relay_parent(relay_parent);
1053
1054	gum::trace!(
1055		target: LOG_TARGET,
1056		?relay_parent,
1057		?para_id,
1058		?paths,
1059		"Checking seconding limit",
1060	);
1061
1062	let mut has_claim_at_some_path = false;
1063	for path in paths {
1064		let mut cq_state = ClaimQueueState::new();
1065		for ancestor in &path {
1066			let seconded_and_pending = state.seconded_and_pending_for_para(&ancestor, &para_id);
1067			cq_state.add_leaf(
1068				&ancestor,
1069				&state
1070					.per_relay_parent
1071					.get(ancestor)
1072					.ok_or(AdvertisementError::RelayParentUnknown)?
1073					.assignment
1074					.current,
1075			);
1076			for _ in 0..seconded_and_pending {
1077				cq_state.claim_at(ancestor, &para_id);
1078			}
1079		}
1080
1081		if cq_state.can_claim_at(relay_parent, &para_id) {
1082			gum::trace!(
1083				target: LOG_TARGET,
1084				?relay_parent,
1085				?para_id,
1086				?path,
1087				"Seconding limit respected at path",
1088			);
1089			has_claim_at_some_path = true;
1090			break
1091		}
1092	}
1093
1094	// If there is a place in the claim queue for the candidate at at least one path we will accept
1095	// it.
1096	if has_claim_at_some_path {
1097		Ok(())
1098	} else {
1099		Err(AdvertisementError::SecondedLimitReached)
1100	}
1101}

1103async fn handle_advertisement<Sender>(
1104	sender: &mut Sender,
1105	state: &mut State,
1106	relay_parent: Hash,
1107	peer_id: PeerId,
1108	prospective_candidate: Option<(CandidateHash, Hash)>,
1109) -> std::result::Result<(), AdvertisementError>
1110where
1111	Sender: CollatorProtocolSenderTrait,
1112{
1113	let peer_data = state.peer_data.get_mut(&peer_id).ok_or(AdvertisementError::UnknownPeer)?;
1114
1115	if peer_data.version == CollationVersion::V1 && !state.active_leaves.contains(&relay_parent) {
1116		return Err(AdvertisementError::ProtocolMisuse)
1117	}
1118
1119	let per_relay_parent = state
1120		.per_relay_parent
1121		.get(&relay_parent)
1122		.ok_or(AdvertisementError::RelayParentUnknown)?;
1123
1124	let assignment = &per_relay_parent.assignment;
1125
1126	let collator_para_id =
1127		peer_data.collating_para().ok_or(AdvertisementError::UndeclaredCollator)?;
1128
1129	// Check if this is assigned to us.
1130	if !assignment.current.contains(&collator_para_id) {
1131		return Err(AdvertisementError::InvalidAssignment)
1132	}
1133
1134	// Always insert advertisements that pass all the checks for spam protection.
1135	let candidate_hash = prospective_candidate.map(|(hash, ..)| hash);
1136	let (collator_id, para_id) = peer_data
1137		.insert_advertisement(
1138			relay_parent,
1139			candidate_hash,
1140			&state.implicit_view,
1141			&state.active_leaves,
1142			&per_relay_parent,
1143		)
1144		.map_err(AdvertisementError::Invalid)?;
1145
1146	ensure_seconding_limit_is_respected(&relay_parent, para_id, state)?;
1147
1148	if let Some((candidate_hash, parent_head_data_hash)) = prospective_candidate {
1149		// Check if backing subsystem allows to second this candidate.
1150		//
1151		// This is also only important when async backing or elastic scaling is enabled.
1152		let can_second = can_second(
1153			sender,
1154			collator_para_id,
1155			relay_parent,
1156			candidate_hash,
1157			parent_head_data_hash,
1158		)
1159		.await;
1160
1161		if !can_second {
1162			return Err(AdvertisementError::BlockedByBacking)
1163		}
1164	}
1165
1166	let result = enqueue_collation(
1167		sender,
1168		state,
1169		relay_parent,
1170		para_id,
1171		peer_id,
1172		collator_id,
1173		prospective_candidate,
1174	)
1175	.await;
1176
1177	if let Err(fetch_error) = result {
1178		gum::debug!(
1179			target: LOG_TARGET,
1180			relay_parent = ?relay_parent,
1181			para_id = ?para_id,
1182			peer_id = ?peer_id,
1183			error = %fetch_error,
1184			"Failed to request advertised collation",
1185		);
1186	}
1187
1188	Ok(())
1189}

1191/// Enqueue collation for fetching. The advertisement is expected to be validated and the seconding
1192/// limit checked.
1193async fn enqueue_collation<Sender>(
1194	sender: &mut Sender,
1195	state: &mut State,
1196	relay_parent: Hash,
1197	para_id: ParaId,
1198	peer_id: PeerId,
1199	collator_id: CollatorId,
1200	prospective_candidate: Option<(CandidateHash, Hash)>,
1201) -> std::result::Result<(), FetchError>
1202where
1203	Sender: CollatorProtocolSenderTrait,
1204{
1205	gum::debug!(
1206		target: LOG_TARGET,
1207		peer_id = ?peer_id,
1208		%para_id,
1209		?relay_parent,
1210		"Received advertise collation",
1211	);
1212	let per_relay_parent = match state.per_relay_parent.get_mut(&relay_parent) {
1213		Some(rp_state) => rp_state,
1214		None => {
1215			// Race happened, not an error.
1216			gum::trace!(
1217				target: LOG_TARGET,
1218				peer_id = ?peer_id,
1219				%para_id,
1220				?relay_parent,
1221				?prospective_candidate,
1222				"Candidate relay parent went out of view for valid advertisement",
1223			);
1224			return Ok(())
1225		},
1226	};
1227	let prospective_candidate =
1228		prospective_candidate.map(|(candidate_hash, parent_head_data_hash)| ProspectiveCandidate {
1229			candidate_hash,
1230			parent_head_data_hash,
1231		});
1232
1233	let collations = &mut per_relay_parent.collations;
1234	let pending_collation =
1235		PendingCollation::new(relay_parent, para_id, &peer_id, prospective_candidate);
1236
1237	match collations.status {
1238		CollationStatus::Fetching(_) | CollationStatus::WaitingOnValidation => {
1239			gum::trace!(
1240				target: LOG_TARGET,
1241				peer_id = ?peer_id,
1242				%para_id,
1243				?relay_parent,
1244				"Added collation to the pending list"
1245			);
1246			collations.add_to_waiting_queue((pending_collation, collator_id));
1247		},
1248		CollationStatus::Waiting => {
1249			// We were waiting for a collation to be advertised to us (we were idle) so we can fetch
1250			// the new collation immediately
1251			fetch_collation(sender, state, pending_collation, collator_id).await?;
1252		},
1253	}
1254
1255	Ok(())
1256}

1258/// Our view has changed.
1259async fn handle_our_view_change<Sender>(
1260	sender: &mut Sender,
1261	state: &mut State,
1262	keystore: &KeystorePtr,
1263	view: OurView,
1264) -> Result<()>
1265where
1266	Sender: CollatorProtocolSenderTrait,
1267{
1268	let current_leaves = state.active_leaves.clone();
1269
1270	let removed = current_leaves.iter().filter(|h| !view.contains(h));
1271	let added = view.iter().filter(|h| !current_leaves.contains(h));
1272
1273	for leaf in added {
1274		let session_index = request_session_index_for_child(*leaf, sender)
1275			.await
1276			.await
1277			.map_err(Error::CancelledSessionIndex)??;
1278
1279		let v2_receipts = request_node_features(*leaf, session_index, sender)
1280			.await
1281			.await
1282			.map_err(Error::CancelledNodeFeatures)??
1283			.get(node_features::FeatureIndex::CandidateReceiptV2 as usize)
1284			.map(|b| *b)
1285			.unwrap_or(false);
1286
1287		let Some(per_relay_parent) = construct_per_relay_parent(
1288			sender,
1289			&mut state.current_assignments,
1290			keystore,
1291			*leaf,
1292			v2_receipts,
1293			session_index,
1294		)
1295		.await?
1296		else {
1297			continue
1298		};
1299
1300		state.active_leaves.insert(*leaf);
1301		state.per_relay_parent.insert(*leaf, per_relay_parent);
1302
1303		state
1304			.implicit_view
1305			.activate_leaf(sender, *leaf)
1306			.await
1307			.map_err(Error::ImplicitViewFetchError)?;
1308
1309		// Order is always descending.
1310		let allowed_ancestry = state
1311			.implicit_view
1312			.known_allowed_relay_parents_under(leaf, None)
1313			.unwrap_or_default();
1314		for block_hash in allowed_ancestry {
1315			if let Entry::Vacant(entry) = state.per_relay_parent.entry(*block_hash) {
1316				// Safe to use the same v2 receipts config for the allowed relay parents as well
1317				// as the same session index since they must be in the same session.
1318				if let Some(per_relay_parent) = construct_per_relay_parent(
1319					sender,
1320					&mut state.current_assignments,
1321					keystore,
1322					*block_hash,
1323					v2_receipts,
1324					session_index,
1325				)
1326				.await?
1327				{
1328					entry.insert(per_relay_parent);
1329				}
1330			}
1331		}
1332	}
1333
1334	for removed in removed {
1335		gum::trace!(
1336			target: LOG_TARGET,
1337			?view,
1338			?removed,
1339			"handle_our_view_change - removed",
1340		);
1341
1342		state.active_leaves.remove(removed);
1343		// If the leaf is deactivated it still may stay in the view as a part
1344		// of implicit ancestry. Only update the state after the hash is actually
1345		// pruned from the block info storage.
1346		let pruned = state.implicit_view.deactivate_leaf(*removed);
1347
1348		for removed in pruned {
1349			if let Some(per_relay_parent) = state.per_relay_parent.remove(&removed) {
1350				remove_outgoing(&mut state.current_assignments, per_relay_parent);
1351			}
1352
1353			state.collation_requests_cancel_handles.retain(|pc, handle| {
1354				let keep = pc.relay_parent != removed;
1355				if !keep {
1356					handle.cancel();
1357				}
1358				keep
1359			});
1360			state.fetched_candidates.retain(|k, _| k.relay_parent != removed);
1361		}
1362	}
1363
1364	// Remove blocked seconding requests that left the view.
1365	state.blocked_from_seconding.retain(|_, collations| {
1366		collations.retain(|collation| {
1367			state
1368				.per_relay_parent
1369				.contains_key(&collation.candidate_receipt.descriptor.relay_parent())
1370		});
1371
1372		!collations.is_empty()
1373	});
1374
1375	for (peer_id, peer_data) in state.peer_data.iter_mut() {
1376		peer_data.prune_old_advertisements(&state.implicit_view, &state.active_leaves);
1377
1378		// Disconnect peers who are not relevant to our current or next para.
1379		//
1380		// If the peer hasn't declared yet, they will be disconnected if they do not
1381		// declare.
1382		if let Some(para_id) = peer_data.collating_para() {
1383			if !state.current_assignments.contains_key(&para_id) {
1384				gum::trace!(
1385					target: LOG_TARGET,
1386					?peer_id,
1387					?para_id,
1388					"Disconnecting peer on view change (not current parachain id)"
1389				);
1390				disconnect_peer(sender, *peer_id).await;
1391			}
1392		}
1393	}
1394
1395	Ok(())
1396}

1398/// Bridge event switch.
1399#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1400async fn handle_network_msg<Context>(
1401	ctx: &mut Context,
1402	state: &mut State,
1403	keystore: &KeystorePtr,
1404	bridge_message: NetworkBridgeEvent<net_protocol::CollatorProtocolMessage>,
1405) -> Result<()> {
1406	use NetworkBridgeEvent::*;
1407
1408	match bridge_message {
1409		PeerConnected(peer_id, observed_role, protocol_version, _) => {
1410			let version = match protocol_version.try_into() {
1411				Ok(version) => version,
1412				Err(err) => {
1413					// Network bridge is expected to handle this.
1414					gum::error!(
1415						target: LOG_TARGET,
1416						?peer_id,
1417						?observed_role,
1418						?err,
1419						"Unsupported protocol version"
1420					);
1421					return Ok(())
1422				},
1423			};
1424			state.peer_data.entry(peer_id).or_insert_with(|| PeerData {
1425				view: View::default(),
1426				state: PeerState::Connected(Instant::now()),
1427				version,
1428			});
1429			state.metrics.note_collator_peer_count(state.peer_data.len());
1430		},
1431		PeerDisconnected(peer_id) => {
1432			state.peer_data.remove(&peer_id);
1433			state.metrics.note_collator_peer_count(state.peer_data.len());
1434		},
1435		NewGossipTopology { .. } => {
1436			// impossible!
1437		},
1438		PeerViewChange(peer_id, view) => {
1439			handle_peer_view_change(state, peer_id, view);
1440		},
1441		OurViewChange(view) => {
1442			handle_our_view_change(ctx.sender(), state, keystore, view).await?;
1443		},
1444		PeerMessage(remote, msg) => {
1445			process_incoming_peer_message(ctx, state, remote, msg).await;
1446		},
1447		UpdatedAuthorityIds { .. } => {
1448			// The validator side doesn't deal with `AuthorityDiscoveryId`s.
1449		},
1450	}
1451
1452	Ok(())
1453}

1455/// The main message receiver switch.
1456#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1457async fn process_msg<Context>(
1458	ctx: &mut Context,
1459	keystore: &KeystorePtr,
1460	msg: CollatorProtocolMessage,
1461	state: &mut State,
1462) {
1463	use CollatorProtocolMessage::*;
1464
1465	let _timer = state.metrics.time_process_msg();
1466
1467	match msg {
1468		CollateOn(id) => {
1469			gum::warn!(
1470				target: LOG_TARGET,
1471				para_id = %id,
1472				"CollateOn message is not expected on the validator side of the protocol",
1473			);
1474		},
1475		DistributeCollation { .. } => {
1476			gum::warn!(
1477				target: LOG_TARGET,
1478				"DistributeCollation message is not expected on the validator side of the protocol",
1479			);
1480		},
1481		NetworkBridgeUpdate(event) => {
1482			if let Err(e) = handle_network_msg(ctx, state, keystore, event).await {
1483				gum::warn!(
1484					target: LOG_TARGET,
1485					err = ?e,
1486					"Failed to handle incoming network message",
1487				);
1488			}
1489		},
1490		Seconded(parent, stmt) => {
1491			let receipt = match stmt.payload() {
1492				Statement::Seconded(receipt) => receipt,
1493				Statement::Valid(_) => {
1494					gum::warn!(
1495						target: LOG_TARGET,
1496						?stmt,
1497						relay_parent = %parent,
1498						"Seconded message received with a `Valid` statement",
1499					);
1500					return
1501				},
1502			};
1503			let output_head_data = receipt.commitments.head_data.clone();
1504			let output_head_data_hash = receipt.descriptor.para_head();
1505			let fetched_collation = FetchedCollation::from(&receipt.to_plain());
1506			if let Some(CollationEvent { collator_id, pending_collation, .. }) =
1507				state.fetched_candidates.remove(&fetched_collation)
1508			{
1509				let PendingCollation {
1510					relay_parent, peer_id, prospective_candidate, para_id, ..
1511				} = pending_collation;
1512				note_good_collation(
1513					&mut state.reputation,
1514					ctx.sender(),
1515					&state.peer_data,
1516					collator_id.clone(),
1517				)
1518				.await;
1519				if let Some(peer_data) = state.peer_data.get(&peer_id) {
1520					notify_collation_seconded(
1521						ctx.sender(),
1522						peer_id,
1523						peer_data.version,
1524						relay_parent,
1525						stmt,
1526					)
1527					.await;
1528				}
1529
1530				if let Some(rp_state) = state.per_relay_parent.get_mut(&parent) {
1531					rp_state.collations.note_seconded(para_id);
1532				}
1533
1534				// See if we've unblocked other collations for seconding.
1535				second_unblocked_collations(
1536					ctx,
1537					state,
1538					fetched_collation.para_id,
1539					output_head_data,
1540					output_head_data_hash,
1541				)
1542				.await;
1543
1544				// If async backing is enabled, make an attempt to fetch next collation.
1545				let maybe_candidate_hash =
1546					prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1547				dequeue_next_collation_and_fetch(
1548					ctx,
1549					state,
1550					parent,
1551					(collator_id, maybe_candidate_hash),
1552				)
1553				.await;
1554			} else {
1555				gum::debug!(
1556					target: LOG_TARGET,
1557					relay_parent = ?parent,
1558					"Collation has been seconded, but the relay parent is deactivated",
1559				);
1560			}
1561		},
1562		Invalid(parent, candidate_receipt) => {
1563			// Remove collations which were blocked from seconding and had this candidate as parent.
1564			state.blocked_from_seconding.remove(&BlockedCollationId {
1565				para_id: candidate_receipt.descriptor.para_id(),
1566				parent_head_data_hash: candidate_receipt.descriptor.para_head(),
1567			});
1568
1569			let fetched_collation = FetchedCollation::from(&candidate_receipt);
1570			let candidate_hash = fetched_collation.candidate_hash;
1571			let id = match state.fetched_candidates.entry(fetched_collation) {
1572				Entry::Occupied(entry)
1573					if entry.get().pending_collation.commitments_hash ==
1574						Some(candidate_receipt.commitments_hash) =>
1575					entry.remove().collator_id,
1576				Entry::Occupied(_) => {
1577					gum::error!(
1578						target: LOG_TARGET,
1579						relay_parent = ?parent,
1580						candidate = ?candidate_receipt.hash(),
1581						"Reported invalid candidate for unknown `pending_candidate`!",
1582					);
1583					return
1584				},
1585				Entry::Vacant(_) => return,
1586			};
1587
1588			report_collator(&mut state.reputation, ctx.sender(), &state.peer_data, id.clone())
1589				.await;
1590
1591			dequeue_next_collation_and_fetch(ctx, state, parent, (id, Some(candidate_hash))).await;
1592		},
1593	}
1594}

1615#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1616async fn run_inner<Context>(
1617	mut ctx: Context,
1618	keystore: KeystorePtr,
1619	eviction_policy: crate::CollatorEvictionPolicy,
1620	metrics: Metrics,
1621	reputation: ReputationAggregator,
1622	reputation_interval: Duration,
1623) -> std::result::Result<(), std::convert::Infallible> {
1624	let new_reputation_delay = || futures_timer::Delay::new(reputation_interval).fuse();
1625	let mut reputation_delay = new_reputation_delay();
1626
1627	let mut state = State { metrics, reputation, ..Default::default() };
1628
1629	let next_inactivity_stream = tick_stream(ACTIVITY_POLL);
1630	futures::pin_mut!(next_inactivity_stream);
1631
1632	let mut network_error_freq = gum::Freq::new();
1633	let mut canceled_freq = gum::Freq::new();
1634
1635	loop {
1636		select! {
1637			_ = reputation_delay => {
1638				state.reputation.send(ctx.sender()).await;
1639				reputation_delay = new_reputation_delay();
1640			},
1641			res = ctx.recv().fuse() => {
1642				match res {
1643					Ok(FromOrchestra::Communication { msg }) => {
1644						gum::trace!(target: LOG_TARGET, msg = ?msg, "received a message");
1645						process_msg(
1646							&mut ctx,
1647							&keystore,
1648							msg,
1649							&mut state,
1650						).await;
1651					}
1652					Ok(FromOrchestra::Signal(OverseerSignal::Conclude)) | Err(_) => break,
1653					Ok(FromOrchestra::Signal(_)) => continue,
1654				}
1655			},
1656			_ = next_inactivity_stream.next() => {
1657				disconnect_inactive_peers(ctx.sender(), &eviction_policy, &state.peer_data).await;
1658			},
1659			resp = state.collation_requests.select_next_some() => {
1660				let relay_parent = resp.0.pending_collation.relay_parent;
1661				let res = match handle_collation_fetch_response(
1662					&mut state,
1663					resp,
1664					&mut network_error_freq,
1665					&mut canceled_freq,
1666				).await {
1667					Err(Some((peer_id, rep))) => {
1668						modify_reputation(&mut state.reputation, ctx.sender(), peer_id, rep).await;
1669						// Reset the status for the relay parent
1670						state.per_relay_parent.get_mut(&relay_parent).map(|rp| {
1671							rp.collations.status.back_to_waiting();
1672						});
1673						continue
1674					},
1675					Err(None) => {
1676						// Reset the status for the relay parent
1677						state.per_relay_parent.get_mut(&relay_parent).map(|rp| {
1678							rp.collations.status.back_to_waiting();
1679						});
1680						continue
1681					},
1682					Ok(res) => res
1683				};
1684
1685				let CollationEvent {collator_id, pending_collation, .. } = res.collation_event.clone();
1686
1687				match kick_off_seconding(&mut ctx, &mut state, res).await {
1688					Err(err) => {
1689						gum::warn!(
1690							target: LOG_TARGET,
1691							relay_parent = ?pending_collation.relay_parent,
1692							para_id = ?pending_collation.para_id,
1693							peer_id = ?pending_collation.peer_id,
1694							error = %err,
1695							"Seconding aborted due to an error",
1696						);
1697
1698						if err.is_malicious() {
1699							// Report malicious peer.
1700							modify_reputation(&mut state.reputation, ctx.sender(), pending_collation.peer_id, COST_REPORT_BAD).await;
1701						}
1702						let maybe_candidate_hash =
1703						pending_collation.prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1704						dequeue_next_collation_and_fetch(
1705							&mut ctx,
1706							&mut state,
1707							pending_collation.relay_parent,
1708							(collator_id, maybe_candidate_hash),
1709						)
1710						.await;
1711					},
1712					Ok(false) => {
1713						// No hard error occurred, but we can try fetching another collation.
1714						let maybe_candidate_hash =
1715						pending_collation.prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1716						dequeue_next_collation_and_fetch(
1717							&mut ctx,
1718							&mut state,
1719							pending_collation.relay_parent,
1720							(collator_id, maybe_candidate_hash),
1721						)
1722						.await;
1723					}
1724					Ok(true) => {}
1725				}
1726			},
1727			res = state.collation_fetch_timeouts.select_next_some() => {
1728				let (collator_id, maybe_candidate_hash, relay_parent) = res;
1729				gum::debug!(
1730					target: LOG_TARGET,
1731					?relay_parent,
1732					?collator_id,
1733					"Timeout hit - already seconded?"
1734				);
1735				dequeue_next_collation_and_fetch(
1736					&mut ctx,
1737					&mut state,
1738					relay_parent,
1739					(collator_id, maybe_candidate_hash),
1740				)
1741				.await;
1742			}
1743		}
1744	}
1745
1746	Ok(())
1747}

1753	state: &mut State,

1762			?id,

1771				error = %err,
1772				"Failed to request a collation, dequeueing next one",
1773			);
1774		} else {
1775			break
1776		}
1777	}

1829/// Handle a fetched collation result.
1830/// Returns whether or not seconding has begun.
1831#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1832async fn kick_off_seconding<Context>(
1833	ctx: &mut Context,
1834	state: &mut State,
1835	PendingCollationFetch { mut collation_event, candidate_receipt, pov, maybe_parent_head_data }: PendingCollationFetch,
1836) -> std::result::Result<bool, SecondingError> {
1837	let pending_collation = collation_event.pending_collation;
1838	let relay_parent = pending_collation.relay_parent;
1839
1840	let per_relay_parent = match state.per_relay_parent.get_mut(&relay_parent) {
1841		Some(state) => state,
1842		None => {
1843			// Relay parent went out of view, not an error.
1844			gum::trace!(
1845				target: LOG_TARGET,
1846				relay_parent = ?relay_parent,
1847				"Fetched collation for a parent out of view",
1848			);
1849			return Ok(false)
1850		},
1851	};
1852
1853	// Sanity check of the candidate receipt version.
1854	descriptor_version_sanity_check(candidate_receipt.descriptor(), per_relay_parent)?;
1855
1856	let collations = &mut per_relay_parent.collations;
1857
1858	let fetched_collation = FetchedCollation::from(&candidate_receipt);
1859	if let Entry::Vacant(entry) = state.fetched_candidates.entry(fetched_collation) {
1860		collation_event.pending_collation.commitments_hash =
1861			Some(candidate_receipt.commitments_hash);
1862
1863		let (maybe_pvd, maybe_parent_head, maybe_parent_head_hash) = match (
1864			collation_event.collator_protocol_version,
1865			collation_event.pending_collation.prospective_candidate,
1866		) {
1867			(CollationVersion::V2, Some(ProspectiveCandidate { parent_head_data_hash, .. })) => {
1868				let pvd = request_prospective_validation_data(
1869					ctx.sender(),
1870					relay_parent,
1871					parent_head_data_hash,
1872					pending_collation.para_id,
1873					maybe_parent_head_data.clone(),
1874				)
1875				.await?;
1876
1877				(pvd, maybe_parent_head_data, Some(parent_head_data_hash))
1878			},
1879			(CollationVersion::V1, _) => {
1880				let pvd = request_persisted_validation_data(
1881					ctx.sender(),
1882					candidate_receipt.descriptor().relay_parent(),
1883					candidate_receipt.descriptor().para_id(),
1884				)
1885				.await?;
1886				(
1887					Some(pvd.ok_or(SecondingError::PersistedValidationDataNotFound)?),
1888					maybe_parent_head_data,
1889					None,
1890				)
1891			},
1892			_ => {
1893				// `handle_advertisement` checks for protocol mismatch.
1894				return Ok(false)
1895			},
1896		};
1897
1898		let pvd = match (maybe_pvd, maybe_parent_head.clone(), maybe_parent_head_hash) {
1899			(Some(pvd), _, _) => pvd,
1900			(None, None, Some(parent_head_data_hash)) => {
1901				// In this case, the collator did not supply the head data and neither could
1902				// prospective-parachains. We add this to the blocked_from_seconding collection
1903				// until we second its parent.
1904				let blocked_collation = PendingCollationFetch {
1905					collation_event,
1906					candidate_receipt,
1907					pov,
1908					maybe_parent_head_data: None,
1909				};
1910				gum::debug!(
1911					target: LOG_TARGET,
1912					candidate_hash = ?blocked_collation.candidate_receipt.hash(),
1913					relay_parent = ?blocked_collation.candidate_receipt.descriptor.relay_parent(),
1914					"Collation having parent head data hash {} is blocked from seconding. Waiting on its parent to be validated.",
1915					parent_head_data_hash
1916				);
1917				state
1918					.blocked_from_seconding
1919					.entry(BlockedCollationId {
1920						para_id: blocked_collation.candidate_receipt.descriptor.para_id(),
1921						parent_head_data_hash,
1922					})
1923					.or_insert_with(Vec::new)
1924					.push(blocked_collation);
1925
1926				return Ok(false)
1927			},
1928			(None, _, _) => {
1929				// Even though we already have the parent head data, the pvd fetching failed. We
1930				// don't need to wait for seconding another collation outputting this head data.
1931				return Err(SecondingError::PersistedValidationDataNotFound)
1932			},
1933		};
1934
1935		fetched_collation_sanity_check(
1936			&collation_event.pending_collation,
1937			&candidate_receipt,
1938			&pvd,
1939			maybe_parent_head.and_then(|head| maybe_parent_head_hash.map(|hash| (head, hash))),
1940		)?;
1941
1942		ctx.send_message(CandidateBackingMessage::Second(
1943			relay_parent,
1944			candidate_receipt,
1945			pvd,
1946			pov,
1947		))
1948		.await;
1949		// There's always a single collation being fetched at any moment of time.
1950		// In case of a failure, we reset the status back to waiting.
1951		collations.status = CollationStatus::WaitingOnValidation;
1952
1953		entry.insert(collation_event);
1954		Ok(true)
1955	} else {
1956		Err(SecondingError::Duplicate)
1957	}
1958}

1976/// Handle a collation fetch response.
1977async fn handle_collation_fetch_response(
1978	state: &mut State,
1979	response: <CollationFetchRequest as Future>::Output,
1980	network_error_freq: &mut gum::Freq,
1981	canceled_freq: &mut gum::Freq,
1982) -> std::result::Result<PendingCollationFetch, Option<(PeerId, Rep)>> {
1983	let (CollationEvent { collator_id, collator_protocol_version, pending_collation }, response) =
1984		response;
1985	// Remove the cancellation handle, as the future already completed.
1986	state.collation_requests_cancel_handles.remove(&pending_collation);
1987
1988	let response = match response {
1989		Err(CollationFetchError::Cancelled) => {
1990			gum::debug!(
1991				target: LOG_TARGET,
1992				hash = ?pending_collation.relay_parent,
1993				para_id = ?pending_collation.para_id,
1994				peer_id = ?pending_collation.peer_id,
1995				"Request was cancelled from the validator side"
1996			);
1997			return Err(None)
1998		},
1999		Err(CollationFetchError::Request(req_error)) => Err(req_error),
2000		Ok(resp) => Ok(resp),
2001	};
2002
2003	let _timer = state.metrics.time_handle_collation_request_result();
2004
2005	let mut metrics_result = Err(());
2006
2007	let result = match response {
2008		Err(RequestError::InvalidResponse(err)) => {
2009			gum::warn!(
2010				target: LOG_TARGET,
2011				hash = ?pending_collation.relay_parent,
2012				para_id = ?pending_collation.para_id,
2013				peer_id = ?pending_collation.peer_id,
2014				err = ?err,
2015				"Collator provided response that could not be decoded"
2016			);
2017			Err(Some((pending_collation.peer_id, COST_CORRUPTED_MESSAGE)))
2018		},
2019		Err(err) if err.is_timed_out() => {
2020			gum::debug!(
2021				target: LOG_TARGET,
2022				hash = ?pending_collation.relay_parent,
2023				para_id = ?pending_collation.para_id,
2024				peer_id = ?pending_collation.peer_id,
2025				"Request timed out"
2026			);
2027			// For now we don't want to change reputation on timeout, to mitigate issues like
2028			// this: https://github.com/paritytech/polkadot/issues/4617
2029			Err(None)
2030		},
2031		Err(RequestError::NetworkError(err)) => {
2032			gum::warn_if_frequent!(
2033				freq: network_error_freq,
2034				max_rate: gum::Times::PerHour(100),
2035				target: LOG_TARGET,
2036				hash = ?pending_collation.relay_parent,
2037				para_id = ?pending_collation.para_id,
2038				peer_id = ?pending_collation.peer_id,
2039				err = ?err,
2040				"Fetching collation failed due to network error"
2041			);
2042			// A minor decrease in reputation for any network failure seems
2043			// sensible. In theory this could be exploited, by DoSing this node,
2044			// which would result in reduced reputation for proper nodes, but the
2045			// same can happen for penalties on timeouts, which we also have.
2046			Err(Some((pending_collation.peer_id, COST_NETWORK_ERROR)))
2047		},
2048		Err(RequestError::Canceled(err)) => {
2049			gum::warn_if_frequent!(
2050				freq: canceled_freq,
2051				max_rate: gum::Times::PerHour(100),
2052				target: LOG_TARGET,
2053				hash = ?pending_collation.relay_parent,
2054				para_id = ?pending_collation.para_id,
2055				peer_id = ?pending_collation.peer_id,
2056				err = ?err,
2057				"Canceled should be handled by `is_timed_out` above - this is a bug!"
2058			);
2059			Err(None)
2060		},
2061		Ok(
2062			request_v1::CollationFetchingResponse::Collation(receipt, _) |
2063			request_v2::CollationFetchingResponse::Collation(receipt, _) |
2064			request_v1::CollationFetchingResponse::CollationWithParentHeadData { receipt, .. } |
2065			request_v2::CollationFetchingResponse::CollationWithParentHeadData { receipt, .. },
2066		) if receipt.descriptor().para_id() != pending_collation.para_id => {
2067			gum::debug!(
2068				target: LOG_TARGET,
2069				expected_para_id = ?pending_collation.para_id,
2070				got_para_id = ?receipt.descriptor().para_id(),
2071				peer_id = ?pending_collation.peer_id,
2072				"Got wrong para ID for requested collation."
2073			);
2074
2075			Err(Some((pending_collation.peer_id, COST_WRONG_PARA)))
2076		},
2077		Ok(request_v1::CollationFetchingResponse::Collation(candidate_receipt, pov)) => {
2078			gum::debug!(
2079				target: LOG_TARGET,
2080				para_id = %pending_collation.para_id,
2081				hash = ?pending_collation.relay_parent,
2082				candidate_hash = ?candidate_receipt.hash(),
2083				"Received collation",
2084			);
2085
2086			metrics_result = Ok(());
2087			Ok(PendingCollationFetch {
2088				collation_event: CollationEvent {
2089					collator_id,
2090					pending_collation,
2091					collator_protocol_version,
2092				},
2093				candidate_receipt,
2094				pov,
2095				maybe_parent_head_data: None,
2096			})
2097		},
2098		Ok(request_v2::CollationFetchingResponse::CollationWithParentHeadData {
2099			receipt,
2100			pov,
2101			parent_head_data,
2102		}) => {
2103			gum::debug!(
2104				target: LOG_TARGET,
2105				para_id = %pending_collation.para_id,
2106				hash = ?pending_collation.relay_parent,
2107				candidate_hash = ?receipt.hash(),
2108				"Received collation (v3)",
2109			);
2110
2111			metrics_result = Ok(());
2112			Ok(PendingCollationFetch {
2113				collation_event: CollationEvent {
2114					collator_id,
2115					pending_collation,
2116					collator_protocol_version,
2117				},
2118				candidate_receipt: receipt,
2119				pov,
2120				maybe_parent_head_data: Some(parent_head_data),
2121			})
2122		},
2123	};
2124	state.metrics.on_request(metrics_result);
2125	result
2126}

2128// Returns the claim queue without fetched or pending advertisement. The resulting `Vec` keeps the
2129// order in the claim queue so the earlier an element is located in the `Vec` the higher its
2130// priority is.
2131fn unfulfilled_claim_queue_entries(relay_parent: &Hash, state: &State) -> Result<Vec<ParaId>> {
2132	let relay_parent_state = state
2133		.per_relay_parent
2134		.get(relay_parent)
2135		.ok_or(Error::RelayParentStateNotFound)?;
2136	let scheduled_paras = relay_parent_state.assignment.current.iter().collect::<HashSet<_>>();
2137	let paths = state.implicit_view.paths_via_relay_parent(relay_parent);
2138
2139	let mut claim_queue_states = Vec::new();
2140	for path in paths {
2141		let mut cq_state = ClaimQueueState::new();
2142		for ancestor in &path {
2143			cq_state.add_leaf(
2144				&ancestor,
2145				&state
2146					.per_relay_parent
2147					.get(&ancestor)
2148					.ok_or(Error::RelayParentStateNotFound)?
2149					.assignment
2150					.current,
2151			);
2152
2153			for para_id in &scheduled_paras {
2154				let seconded_and_pending = state.seconded_and_pending_for_para(&ancestor, &para_id);
2155				for _ in 0..seconded_and_pending {
2156					cq_state.claim_at(&ancestor, &para_id);
2157				}
2158			}
2159		}
2160		claim_queue_states.push(cq_state);
2161	}
2162
2163	// From the claim queue state for each leaf we have to return a combined single one. Go for a
2164	// simple solution and return the longest one. In theory we always prefer the earliest entries
2165	// in the claim queue so there is a good chance that the longest path is the one with
2166	// unsatisfied entries in the beginning. This is not guaranteed as we might have fetched 2nd or
2167	// 3rd spot from the claim queue but it should be good enough.
2168	let unfulfilled_entries = claim_queue_states
2169		.iter_mut()
2170		.map(|cq| cq.unclaimed_at(relay_parent))
2171		.max_by(|a, b| a.len().cmp(&b.len()))
2172		.unwrap_or_default();
2173
2174	Ok(unfulfilled_entries)
2175}

2177/// Returns the next collation to fetch from the `waiting_queue` and reset the status back to
2178/// `Waiting`.
2179fn get_next_collation_to_fetch(
2180	finished_one: &(CollatorId, Option<CandidateHash>),
2181	relay_parent: Hash,
2182	state: &mut State,
2183) -> Option<(PendingCollation, CollatorId)> {
2184	let unfulfilled_entries = match unfulfilled_claim_queue_entries(&relay_parent, &state) {
2185		Ok(entries) => entries,
2186		Err(err) => {
2187			gum::error!(
2188				target: LOG_TARGET,
2189				?relay_parent,
2190				?err,
2191				"Failed to get unfulfilled claim queue entries"
2192			);
2193			return None
2194		},
2195	};
2196	let rp_state = match state.per_relay_parent.get_mut(&relay_parent) {
2197		Some(rp_state) => rp_state,
2198		None => {
2199			gum::error!(
2200				target: LOG_TARGET,
2201				?relay_parent,
2202				"Failed to get relay parent state"
2203			);
2204			return None
2205		},
2206	};
2207
2208	// If finished one does not match waiting_collation, then we already dequeued another fetch
2209	// to replace it.
2210	if let Some((collator_id, maybe_candidate_hash)) = rp_state.collations.fetching_from.as_ref() {
2211		// If a candidate hash was saved previously, `finished_one` must include this too.
2212		if collator_id != &finished_one.0 &&
2213			maybe_candidate_hash.map_or(true, |hash| Some(&hash) != finished_one.1.as_ref())
2214		{
2215			gum::trace!(
2216				target: LOG_TARGET,
2217				waiting_collation = ?rp_state.collations.fetching_from,
2218				?finished_one,
2219				"Not proceeding to the next collation - has already been done."
2220			);
2221			return None
2222		}
2223	}
2224	rp_state.collations.status.back_to_waiting();
2225	rp_state.collations.pick_a_collation_to_fetch(unfulfilled_entries)
2226}
