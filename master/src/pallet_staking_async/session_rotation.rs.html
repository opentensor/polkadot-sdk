18//! Manages all era rotation logic based on session increments.
19//!
20//! # Lifecycle:
21//!
22//! When a session ends in RC, a session report is sent to AH with the ending session index. Given
23//! there are 6 sessions per Era, and we configure the PlanningEraOffset to be 1, the following
24//! happens.
25//!
26//! ## Idle Sessions
27//! In the happy path, first 3 sessions are idle. Nothing much happens in these sessions.
28//!
29//!
30//! ## Planning New Era Session
31//! In the happy path, `planning new era` session is initiated when 3rd session ends and the 4th
32//! starts in the active era.
33//!
34//! **Triggers**
35//! 1. `SessionProgress == SessionsPerEra - PlanningEraOffset`
36//! 2. Forcing is set to `ForceNew` or `ForceAlways`
37//!
38//! **Actions**
39//! 1. Triggers the election process,
40//! 2. Updates the CurrentEra.
41//!
42//! **SkipIf**
43//! CurrentEra = ActiveEra + 1 // this implies planning session has already been triggered.
44//!
45//! **FollowUp**
46//! When the election process is over, we send the new validator set, with the CurrentEra index
47//! as the id of the validator set.
48//!
49//!
50//! ## Era Rotation Session
51//! In the happy path, this happens when the 5th session ends and the 6th starts in the active era.
52//!
53//! **Triggers**
54//! When we receive an activation timestamp from RC.
55//!
56//! **Assertions**
57//! 1. CurrentEra must be ActiveEra + 1.
58//! 2. Id of the activation timestamp same as CurrentEra.
59//!
60//! **Actions**
61//! - Finalize the currently active era.
62//! - Increment ActiveEra by 1.
63//! - Cleanup the old era information.
64//!
65//! **Exceptional Scenarios**
66//! - Delay in exporting validator set: Triggered in a session later than 7th.
67//! - Forcing Era: May triggered in a session earlier than 7th.
68//!
69//! ## Example Flow of a happy path
70//!
71//! * end 0, start 1, plan 2
72//! * end 1, start 2, plan 3
73//! * end 2, start 3, plan 4
74//! * end 3, start 4, plan 5 // `Plan new era` session. Current Era++. Trigger Election.
75//! * **** Somewhere here: Election set is sent to RC, keyed with Current Era
76//! * end 4, start 5, plan 6 // RC::session receives and queues this set.
77//! * end 5, start 6, plan 7 // Session report contains activation timestamp with Current Era.

105	/// Prune all associated information with the given era.
106	///
107	/// Implementation note: ATM this is deleting all the information in one go, yet it can very
108	/// well be done lazily.
109	pub(crate) fn prune_era(era: EraIndex) {
110		crate::log!(debug, "Pruning era {:?}", era);
111		let mut cursor = <ErasValidatorPrefs<T>>::clear_prefix(era, u32::MAX, None);
112		debug_assert!(cursor.maybe_cursor.is_none());
113		cursor = <ClaimedRewards<T>>::clear_prefix(era, u32::MAX, None);
114		debug_assert!(cursor.maybe_cursor.is_none());
115		cursor = <ErasStakersPaged<T>>::clear_prefix((era,), u32::MAX, None);
116		debug_assert!(cursor.maybe_cursor.is_none());
117		cursor = <ErasStakersOverview<T>>::clear_prefix(era, u32::MAX, None);
118		debug_assert!(cursor.maybe_cursor.is_none());
119
120		<ErasValidatorReward<T>>::remove(era);
121		<ErasRewardPoints<T>>::remove(era);
122		<ErasTotalStake<T>>::remove(era);
123
124		// register the weight of the pruning.
125		Pallet::<T>::register_weight(T::WeightInfo::prune_era(ValidatorCount::<T>::get()));
126	}

151	/// Get exposure for a validator at a given era and page.
152	///
153	/// This builds a paged exposure from `PagedExposureMetadata` and `ExposurePage` of the
154	/// validator. For older non-paged exposure, it returns the clipped exposure directly.
155	pub(crate) fn get_paged_exposure(
156		era: EraIndex,
157		validator: &T::AccountId,
158		page: Page,
159	) -> Option<PagedExposure<T::AccountId, BalanceOf<T>>> {
160		let overview = <ErasStakersOverview<T>>::get(&era, validator)?;
161
162		// validator stake is added only in page zero
163		let validator_stake = if page == 0 { overview.own } else { Zero::zero() };
164
165		// since overview is present, paged exposure will always be present except when a
166		// validator has only own stake and no nominator stake.
167		let exposure_page = <ErasStakersPaged<T>>::get((era, validator, page)).unwrap_or_default();
168
169		// build the exposure
170		Some(PagedExposure {
171			exposure_metadata: PagedExposureMetadata { own: validator_stake, ..overview },
172			exposure_page: exposure_page.into(),
173		})
174	}

176	/// Get full exposure of the validator at a given era.
177	pub(crate) fn get_full_exposure(
178		era: EraIndex,
179		validator: &T::AccountId,
180	) -> Exposure<T::AccountId, BalanceOf<T>> {
181		let Some(overview) = <ErasStakersOverview<T>>::get(&era, validator) else {
182			return Exposure::default();
183		};
184
185		let mut others = Vec::with_capacity(overview.nominator_count as usize);
186		for page in 0..overview.page_count {
187			let nominators = <ErasStakersPaged<T>>::get((era, validator, page));
188			others.append(&mut nominators.map(|n| n.others.clone()).defensive_unwrap_or_default());
189		}
190
191		Exposure { total: overview.total, own: overview.own, others }
192	}

194	/// Returns the number of pages of exposure a validator has for the given era.
195	///
196	/// For eras where paged exposure does not exist, this returns 1 to keep backward compatibility.
197	pub(crate) fn exposure_page_count(era: EraIndex, validator: &T::AccountId) -> Page {
198		<ErasStakersOverview<T>>::get(&era, validator)
199			.map(|overview| {
200				if overview.page_count == 0 && overview.own > Zero::zero() {
201					// Even though there are no nominator pages, there is still validator's own
202					// stake exposed which needs to be paid out in a page.
203					1
204				} else {
205					overview.page_count
206				}
207			})
208			// Always returns 1 page for older non-paged exposure.
209			// FIXME: Can be cleaned up with issue #13034.
210			.unwrap_or(1)
211	}

213	/// Returns the next page that can be claimed or `None` if nothing to claim.
214	pub(crate) fn get_next_claimable_page(era: EraIndex, validator: &T::AccountId) -> Option<Page> {
215		// Find next claimable page of paged exposure.
216		let page_count = Self::exposure_page_count(era, validator);
217		let all_claimable_pages: Vec<Page> = (0..page_count).collect();
218		let claimed_pages = ClaimedRewards::<T>::get(era, validator);
219
220		all_claimable_pages.into_iter().find(|p| !claimed_pages.contains(p))
221	}

223	/// Creates an entry to track validator reward has been claimed for a given era and page.
224	/// Noop if already claimed.
225	pub(crate) fn set_rewards_as_claimed(era: EraIndex, validator: &T::AccountId, page: Page) {
226		let mut claimed_pages = ClaimedRewards::<T>::get(era, validator).into_inner();
227
228		// this should never be called if the reward has already been claimed
229		if claimed_pages.contains(&page) {
230			defensive!("Trying to set an already claimed reward");
231			// nevertheless don't do anything since the page already exist in claimed rewards.
232			return
233		}
234
235		// add page to claimed entries
236		claimed_pages.push(page);
237		ClaimedRewards::<T>::insert(
238			era,
239			validator,
240			WeakBoundedVec::<_, _>::force_from(claimed_pages, Some("set_rewards_as_claimed")),
241		);
242	}

244	/// Store exposure for elected validators at start of an era.
245	///
246	/// If the exposure does not exist yet for the tuple (era, validator), it sets it. Otherwise,
247	/// it updates the existing record by ensuring *intermediate* exposure pages are filled up with
248	/// `T::MaxExposurePageSize` number of backers per page and the remaining exposures are added
249	/// to new exposure pages.
250	pub fn upsert_exposure(
251		era: EraIndex,
252		validator: &T::AccountId,
253		mut exposure: Exposure<T::AccountId, BalanceOf<T>>,
254	) {
255		let page_size = T::MaxExposurePageSize::get().defensive_max(1);
256
257		if let Some(stored_overview) = ErasStakersOverview::<T>::get(era, &validator) {
258			let last_page_idx = stored_overview.page_count.saturating_sub(1);
259
260			let mut last_page =
261				ErasStakersPaged::<T>::get((era, validator, last_page_idx)).unwrap_or_default();
262			let last_page_empty_slots =
263				T::MaxExposurePageSize::get().saturating_sub(last_page.others.len() as u32);
264
265			// splits the exposure so that `exposures_append` will fit within the last exposure
266			// page, up to the max exposure page size. The remaining individual exposures in
267			// `exposure` will be added to new pages.
268			let exposures_append = exposure.split_others(last_page_empty_slots);
269
270			ErasStakersOverview::<T>::mutate(era, &validator, |stored| {
271				// new metadata is updated based on 3 different set of exposures: the
272				// current one, the exposure split to be "fitted" into the current last page and
273				// the exposure set that will be appended from the new page onwards.
274				let new_metadata =
275					stored.defensive_unwrap_or_default().update_with::<T::MaxExposurePageSize>(
276						[&exposures_append, &exposure]
277							.iter()
278							.fold(Default::default(), |total, expo| {
279								total.saturating_add(expo.total.saturating_sub(expo.own))
280							}),
281						[&exposures_append, &exposure]
282							.iter()
283							.fold(Default::default(), |count, expo| {
284								count.saturating_add(expo.others.len() as u32)
285							}),
286					);
287				*stored = new_metadata.into();
288			});
289
290			// fill up last page with exposures.
291			last_page.page_total = last_page
292				.page_total
293				.saturating_add(exposures_append.total)
294				.saturating_sub(exposures_append.own);
295			last_page.others.extend(exposures_append.others);
296			ErasStakersPaged::<T>::insert((era, &validator, last_page_idx), last_page);
297
298			// now handle the remaining exposures and append the exposure pages. The metadata update
299			// has been already handled above.
300			let (_, exposure_pages) = exposure.into_pages(page_size);
301
302			exposure_pages.into_iter().enumerate().for_each(|(idx, paged_exposure)| {
303				let append_at =
304					(last_page_idx.saturating_add(1).saturating_add(idx as u32)) as Page;
305				<ErasStakersPaged<T>>::insert((era, &validator, append_at), paged_exposure);
306			});
307		} else {
308			// expected page count is the number of nominators divided by the page size, rounded up.
309			let expected_page_count = exposure
310				.others
311				.len()
312				.defensive_saturating_add((page_size as usize).defensive_saturating_sub(1))
313				.saturating_div(page_size as usize);
314
315			// no exposures yet for this (era, validator) tuple, calculate paged exposure pages and
316			// metadata from a blank slate.
317			let (exposure_metadata, exposure_pages) = exposure.into_pages(page_size);
318			defensive_assert!(exposure_pages.len() == expected_page_count, "unexpected page count");
319
320			// insert metadata.
321			ErasStakersOverview::<T>::insert(era, &validator, exposure_metadata);
322
323			// insert validator's overview.
324			exposure_pages.into_iter().enumerate().for_each(|(idx, paged_exposure)| {
325				let append_at = idx as Page;
326				<ErasStakersPaged<T>>::insert((era, &validator, append_at), paged_exposure);
327			});
328		};
329	}

351	/// Add reward points to validators using their stash account ID.
352	pub(crate) fn reward_active_era(
353		validators_points: impl IntoIterator<Item = (T::AccountId, u32)>,
354	) {
355		if let Some(active_era) = ActiveEra::<T>::get() {
356			<ErasRewardPoints<T>>::mutate(active_era.index, |era_rewards| {
357				for (validator, points) in validators_points.into_iter() {
358					match era_rewards.individual.get_mut(&validator) {
359						Some(individual) => individual.saturating_accrue(points),
360						None => {
361							// not much we can do -- validators should always be less than
362							// `MaxValidatorCount`.
363							let _ =
364								era_rewards.individual.try_insert(validator, points).defensive();
365						},
366					}
367					era_rewards.total += points;
368				}
369			});
370		}
371	}

381	/// Ensure the given era is present, i.e. has not been pruned yet.
382	pub(crate) fn era_present(era: EraIndex) -> Result<(), sp_runtime::TryRuntimeError> {
383		// these two are only set if we have some validators in an era.
384		let e0 = ErasValidatorPrefs::<T>::iter_prefix_values(era).count() != 0;
385		// note: we don't check `ErasStakersPaged` as a validator can have no backers.
386		let e1 = ErasStakersOverview::<T>::iter_prefix_values(era).count() != 0;
387		ensure!(e0 == e1, "ErasValidatorPrefs and ErasStakersOverview should be consistent");
388
389		// these two must always be set
390		let e2 = ErasTotalStake::<T>::contains_key(era);
391
392		let active_era = Rotator::<T>::active_era();
393		let e4 = if era.saturating_sub(1) > 0 &&
394			era.saturating_sub(1) > active_era.saturating_sub(T::HistoryDepth::get() + 1)
395		{
396			// `ErasValidatorReward` is set at active era n for era n-1, and is not set for era 0 in
397			// our tests. Moreover, it cannot be checked for presence in the oldest present era
398			// (`active_era.saturating_sub(1)`)
399			ErasValidatorReward::<T>::contains_key(era.saturating_sub(1))
400		} else {
401			// ignore
402			e2
403		};
404
405		ensure!(e2 == e4, "era info presence not consistent");
406
407		if e2 {
408			Ok(())
409		} else {
410			Err("era presence mismatch".into())
411		}
412	}

414	/// Ensure the given era has indeed been already pruned.
415	pub(crate) fn era_absent(era: EraIndex) -> Result<(), sp_runtime::TryRuntimeError> {
416		// check double+ maps
417		let e0 = ErasValidatorPrefs::<T>::iter_prefix_values(era).count() != 0;
418		let e1 = ErasStakersPaged::<T>::iter_prefix_values((era,)).count() != 0;
419		let e2 = ErasStakersOverview::<T>::iter_prefix_values(era).count() != 0;
420
421		// check maps
422		// `ErasValidatorReward` is set at active era n for era n-1
423		let e3 = ErasValidatorReward::<T>::contains_key(era);
424		let e4 = ErasTotalStake::<T>::contains_key(era);
425
426		// these two are only populated conditionally, so we only check them for lack of existence
427		let e6 = ClaimedRewards::<T>::iter_prefix_values(era).count() != 0;
428		let e7 = ErasRewardPoints::<T>::contains_key(era);
429
430		assert!(
431			vec![e0, e1, e2, e3, e4, e6, e7].windows(2).all(|w| w[0] == w[1]),
432			"era info absence not consistent for era {}: {}, {}, {}, {}, {}, {}, {}",
433			era,
434			e0,
435			e1,
436			e2,
437			e3,
438			e4,
439			e6,
440			e7
441		);
442
443		if !e0 {
444			Ok(())
445		} else {
446			Err("era absence mismatch".into())
447		}
448	}

450	pub(crate) fn do_try_state() -> Result<(), sp_runtime::TryRuntimeError> {
451		// pruning window works.
452		let active_era = Rotator::<T>::active_era();
453		// we max with 1 as in active era 0 we don't do an election and therefore we don't have some
454		// of the maps populated.
455		let oldest_present_era = active_era.saturating_sub(T::HistoryDepth::get()).max(1);
456		let maybe_first_pruned_era =
457			active_era.saturating_sub(T::HistoryDepth::get()).checked_sub(One::one());
458
459		for e in oldest_present_era..=active_era {
460			Self::era_present(e)?
461		}
462		if let Some(first_pruned_era) = maybe_first_pruned_era {
463			Self::era_absent(first_pruned_era)?;
464		}
465		Ok(())
466	}

469/// Manages session rotation logic.
470///
471/// This controls the following storage items in FULL, meaning that they should not be accessed
472/// directly from anywhere else in this pallet:
473///
474/// * `CurrentEra`: The current planning era
475/// * `ActiveEra`: The current active era
476/// * `BondedEras`: the list of ACTIVE eras and their session index
477pub struct Rotator<T: Config>(core::marker::PhantomData<T>);
478
479impl<T: Config> Rotator<T> {
480	#[cfg(feature = "runtime-benchmarks")]
481	pub(crate) fn legacy_insta_plan_era() -> Vec<T::AccountId> {
482		// Plan the era,
483		Self::plan_new_era();
484		// signal that we are about to call into elect asap.
485		<<T as Config>::ElectionProvider as ElectionProvider>::asap();
486		// immediately call into the election provider to fetch and process the results. We assume
487		// we are using an instant, onchain election here.
488		let msp = <T::ElectionProvider as ElectionProvider>::msp();
489		let lsp = 0;
490		for p in (lsp..=msp).rev() {
491			EraElectionPlanner::<T>::do_elect_paged(p);
492		}
493
494		crate::ElectableStashes::<T>::take().into_iter().collect()
495	}

497	#[cfg(any(feature = "try-runtime", test))]
498	pub(crate) fn do_try_state() -> Result<(), sp_runtime::TryRuntimeError> {
499		// planned era can always be at most one more than active era
500		let planned = Self::planned_era();
501		let active = Self::active_era();
502		ensure!(
503			planned == active || planned == active + 1,
504			"planned era is always equal or one more than active"
505		);
506
507		// bonded eras must always be the range [active - bonding_duration .. active_era]
508		let bonded = BondedEras::<T>::get();
509		ensure!(
510			bonded.into_iter().map(|(era, _sess)| era).collect::<Vec<_>>() ==
511				(active.saturating_sub(T::BondingDuration::get())..=active).collect::<Vec<_>>(),
512			"BondedEras range incorrect"
513		);
514
515		Ok(())
516	}

542	/// Next era that is planned to be started.
543	///
544	/// Returns None if no era is planned.
545	pub fn is_planning() -> Option<EraIndex> {
546		let (active, planned) = (Self::active_era(), Self::planned_era());
547		if planned.defensive_saturating_sub(active) > 1 {
548			defensive!("planned era must always be equal or one more than active");
549		}
550
551		(planned > active).then_some(planned)
552	}
553
554	/// End the session and start the next one.
555	pub(crate) fn end_session(end_index: SessionIndex, activation_timestamp: Option<(u64, u32)>) {
556		let Some(active_era) = ActiveEra::<T>::get() else {
557			defensive!("Active era must always be available.");
558			return;
559		};
560		let current_planned_era = Self::is_planning();
561		let starting = end_index + 1;
562		// the session after the starting session.
563		let planning = starting + 1;
564
565		log!(
566			info,
567			"Session: end {:?}, start {:?} (ts: {:?}), planning {:?}",
568			end_index,
569			starting,
570			activation_timestamp,
571			planning
572		);
573		log!(info, "Era: active {:?}, planned {:?}", active_era.index, current_planned_era);
574
575		match activation_timestamp {
576			Some((time, id)) if Some(id) == current_planned_era => {
577				// We rotate the era if we have the activation timestamp.
578				Self::start_era(active_era, starting, time);
579			},
580			Some((_time, id)) => {
581				// RC has done something wrong -- we received the wrong ID. Don't start a new era.
582				crate::log!(
583					warn,
584					"received wrong ID with activation timestamp. Got {}, expected {:?}",
585					id,
586					current_planned_era
587				);
588				Pallet::<T>::deposit_event(Event::Unexpected(
589					UnexpectedKind::UnknownValidatorActivation,
590				));
591			},
592			None => (),
593		}
594
595		// check if we should plan new era.
596		let should_plan_era = match ForceEra::<T>::get() {
597			// see if it's good time to plan a new era.
598			Forcing::NotForcing => Self::is_plan_era_deadline(starting),
599			// Force plan new era only once.
600			Forcing::ForceNew => {
601				ForceEra::<T>::put(Forcing::NotForcing);
602				true
603			},
604			// always plan the new era.
605			Forcing::ForceAlways => true,
606			// never force.
607			Forcing::ForceNone => false,
608		};
609
610		// Note: we call `planning_era` again, as a new era might have started since we checked
611		// it last.
612		let has_pending_era = Self::is_planning().is_some();
613		match (should_plan_era, has_pending_era) {
614			(false, _) => {
615				// nothing to consider
616			},
617			(true, false) => {
618				// happy path
619				Self::plan_new_era();
620			},
621			(true, true) => {
622				// we are waiting for to start the previously planned era, we cannot plan a new era
623				// now.
624				crate::log!(
625					debug,
626					"time to plan a new era {:?}, but waiting for the activation of the previous.",
627					current_planned_era
628				);
629			},
630		}
631
632		Pallet::<T>::deposit_event(Event::SessionRotated {
633			starting_session: starting,
634			active_era: Self::active_era(),
635			planned_era: Self::planned_era(),
636		});
637	}

639	pub(crate) fn start_era(
640		ending_era: ActiveEraInfo,
641		starting_session: SessionIndex,
642		new_era_start_timestamp: u64,
643	) {
644		// verify that a new era was planned
645		debug_assert!(CurrentEra::<T>::get().unwrap_or(0) == ending_era.index + 1);
646
647		let starting_era = ending_era.index + 1;
648
649		// finalize the ending era.
650		Self::end_era(&ending_era, new_era_start_timestamp);
651
652		// start the next era.
653		Self::start_era_inc_active_era(new_era_start_timestamp);
654		Self::start_era_update_bonded_eras(starting_era, starting_session);
655
656		// discard old era information that is no longer needed.
657		Self::cleanup_old_era(starting_era);
658	}
659
660	fn start_era_inc_active_era(start_timestamp: u64) {
661		ActiveEra::<T>::mutate(|active_era| {
662			let new_index = active_era.as_ref().map(|info| info.index + 1).unwrap_or(0);
663			log!(
664				debug,
665				"starting active era {:?} with RC-provided timestamp {:?}",
666				new_index,
667				start_timestamp
668			);
669			*active_era = Some(ActiveEraInfo { index: new_index, start: Some(start_timestamp) });
670		});
671	}

682		BondedEras::<T>::get()
683			.into_iter()
684			.rev()

688	fn start_era_update_bonded_eras(starting_era: EraIndex, start_session: SessionIndex) {
689		let bonding_duration = T::BondingDuration::get();
690
691		BondedEras::<T>::mutate(|bonded| {
692			if bonded.is_full() {
693				// remove oldest
694				let (era_removed, _) = bonded.remove(0);
695				debug_assert!(
696					era_removed <= (starting_era.saturating_sub(bonding_duration)),
697					"should not delete an era that is not older than bonding duration"
698				);
699				slashing::clear_era_metadata::<T>(era_removed);
700			}
701
702			// must work -- we were not full, or just removed the oldest era.
703			let _ = bonded.try_push((starting_era, start_session)).defensive();
704		});
705	}

707	fn end_era(ending_era: &ActiveEraInfo, new_era_start: u64) {
708		let previous_era_start = ending_era.start.defensive_unwrap_or(new_era_start);
709		let uncapped_era_duration = new_era_start.saturating_sub(previous_era_start);
710
711		// maybe cap the era duration to the maximum allowed by the runtime.
712		let cap = T::MaxEraDuration::get();
713		let era_duration = if cap == 0 {
714			// if the cap is zero (not set), we don't cap the era duration.
715			uncapped_era_duration
716		} else if uncapped_era_duration > cap {
717			Pallet::<T>::deposit_event(Event::Unexpected(UnexpectedKind::EraDurationBoundExceeded));
718
719			// if the cap is set, and era duration exceeds the cap, we cap the era duration to the
720			// maximum allowed.
721			log!(
722				warn,
723				"capping era duration for era {:?} from {:?} to max allowed {:?}",
724				ending_era.index,
725				uncapped_era_duration,
726				cap
727			);
728			cap
729		} else {
730			uncapped_era_duration
731		};
732
733		Self::end_era_compute_payout(ending_era, era_duration);
734	}
735
736	fn end_era_compute_payout(ending_era: &ActiveEraInfo, era_duration: u64) {
737		let staked = ErasTotalStake::<T>::get(ending_era.index);
738		let issuance = asset::total_issuance::<T>();
739
740		log!(
741			debug,
742			"computing inflation for era {:?} with duration {:?}",
743			ending_era.index,
744			era_duration
745		);
746		let (validator_payout, remainder) =
747			T::EraPayout::era_payout(staked, issuance, era_duration);
748
749		let total_payout = validator_payout.saturating_add(remainder);
750		let max_staked_rewards = MaxStakedRewards::<T>::get().unwrap_or(Percent::from_percent(100));
751
752		// apply cap to validators payout and add difference to remainder.
753		let validator_payout = validator_payout.min(max_staked_rewards * total_payout);
754		let remainder = total_payout.saturating_sub(validator_payout);
755
756		Pallet::<T>::deposit_event(Event::<T>::EraPaid {
757			era_index: ending_era.index,
758			validator_payout,
759			remainder,
760		});
761
762		// Set ending era reward.
763		Eras::<T>::set_validators_reward(ending_era.index, validator_payout);
764		T::RewardRemainder::on_unbalanced(asset::issue::<T>(remainder));
765	}

767	/// Plans a new era by kicking off the election process.
768	///
769	/// The newly planned era is targeted to activate in the next session.
770	fn plan_new_era() {
771		let _ = CurrentEra::<T>::try_mutate(|x| {
772			log!(info, "Planning new era: {:?}, sending election start signal", x.unwrap_or(0));
773			let could_start_election = EraElectionPlanner::<T>::plan_new_election();
774			*x = Some(x.unwrap_or(0) + 1);
775			could_start_election
776		});
777	}

779	/// Returns whether we are at the session where we should plan the new era.
780	fn is_plan_era_deadline(start_session: SessionIndex) -> bool {
781		let planning_era_offset = T::PlanningEraOffset::get().min(T::SessionsPerEra::get());
782		// session at which we should plan the new era.
783		let target_plan_era_session = T::SessionsPerEra::get().saturating_sub(planning_era_offset);
784		let era_start_session = Self::active_era_start_session_index();
785
786		// progress of the active era in sessions.
787		let session_progress = start_session.defensive_saturating_sub(era_start_session);
788
789		log!(
790			debug,
791			"Session progress within era: {:?}, target_plan_era_session: {:?}",
792			session_progress,
793			target_plan_era_session
794		);
795		session_progress >= target_plan_era_session
796	}

798	fn cleanup_old_era(starting_era: EraIndex) {
799		EraElectionPlanner::<T>::cleanup();
800
801		// discard the ancient era info.
802		if let Some(old_era) = starting_era.checked_sub(T::HistoryDepth::get() + 1) {
803			log!(debug, "Removing era information for {:?}", old_era);
804			Eras::<T>::prune_era(old_era);
805		}
806	}

809/// Manager type which collects the election results from [`Config::ElectionProvider`] and
810/// finalizes the planning of a new era.
811///
812/// This type managed 3 storage items:
813///
814/// * [`crate::VoterSnapshotStatus`]
815/// * [`crate::NextElectionPage`]
816/// * [`crate::ElectableStashes`]
817///
818/// A new election is fetched over multiple pages, and finalized upon fetching the last page.
819///
820/// * The intermediate state of fetching the election result is kept in [`NextElectionPage`]. If
821///   `Some(_)` something is ongoing, otherwise not.
822/// * We fully trust [`Config::ElectionProvider`] to give us a full set of validators, with enough
823///   backing after all calls to `maybe_fetch_election_results` are done. Note that older versions
824///   of this pallet had a `MinimumValidatorCount` to double-check this, but we don't check it
825///   anymore.
826/// * `maybe_fetch_election_results` returns no weight. Its weight should be taken account in the
827///   e2e benchmarking of the [`Config::ElectionProvider`].
828///
829/// TODOs:
830///
831/// * Add a try-state check based on the 3 storage items
832/// * Move snapshot creation functions here as well.
833pub(crate) struct EraElectionPlanner<T: Config>(PhantomData<T>);
834impl<T: Config> EraElectionPlanner<T> {
835	/// Cleanup all associated storage items.
836	pub(crate) fn cleanup() {
837		VoterSnapshotStatus::<T>::kill();
838		NextElectionPage::<T>::kill();
839		ElectableStashes::<T>::kill();
840		Pallet::<T>::register_weight(T::DbWeight::get().writes(3));
841	}

853	}

855	/// Hook to be used in the pallet's on-initialize.
856	pub(crate) fn maybe_fetch_election_results() {
857		if let Ok(true) = T::ElectionProvider::status() {
858			crate::log!(
859				debug,
860				"Election provider is ready, our status is {:?}",
861				NextElectionPage::<T>::get()
862			);
863
864			debug_assert!(
865				CurrentEra::<T>::get().unwrap_or(0) ==
866					ActiveEra::<T>::get().map_or(0, |a| a.index) + 1,
867				"Next era must be already planned."
868			);
869
870			let current_page = NextElectionPage::<T>::get()
871				.unwrap_or(Self::election_pages().defensive_saturating_sub(1));
872			let maybe_next_page = current_page.checked_sub(1);
873			crate::log!(debug, "fetching page {:?}, next {:?}", current_page, maybe_next_page);
874
875			Self::do_elect_paged(current_page);
876			NextElectionPage::<T>::set(maybe_next_page);
877
878			// if current page was `Some`, and next is `None`, we have finished an election and
879			// we can report it now.
880			if maybe_next_page.is_none() {
881				use pallet_staking_async_rc_client::RcClientInterface;
882				let id = CurrentEra::<T>::get().defensive_unwrap_or(0);
883				let prune_up_to = Self::get_prune_up_to();
884				let rc_validators = ElectableStashes::<T>::take().into_iter().collect::<Vec<_>>();
885
886				crate::log!(
887					info,
888					"Sending new validator set of size {:?} to RC. ID: {:?}, prune_up_to: {:?}",
889					rc_validators.len(),
890					id,
891					prune_up_to
892				);
893
894				T::RcClientInterface::validator_set(rc_validators, id, prune_up_to);
895			}
896		}
897	}

899	/// Get the right value of the first session that needs to be pruned on the RC's historical
900	/// session pallet.
901	fn get_prune_up_to() -> Option<SessionIndex> {
902		let bonded_eras = BondedEras::<T>::get();
903
904		// get the first session of the oldest era in the bonded eras.
905		if bonded_eras.is_full() {
906			bonded_eras.first().map(|(_, first_session)| first_session.saturating_sub(1))
907		} else {
908			None
909		}

912	/// Paginated elect.
913	///
914	/// Fetches the election page with index `page` from the election provider.
915	///
916	/// The results from the elect call should be stored in the `ElectableStashes` storage. In
917	/// addition, it stores stakers' information for next planned era based on the paged
918	/// solution data returned.
919	///
920	/// If any new election winner does not fit in the electable stashes storage, it truncates
921	/// the result of the election. We ensure that only the winners that are part of the
922	/// electable stashes have exposures collected for the next era.
923	pub(crate) fn do_elect_paged(page: PageIndex) {
924		let election_result = T::ElectionProvider::elect(page);
925		match election_result {
926			Ok(supports) => {
927				let inner_processing_results = Self::do_elect_paged_inner(supports);
928				if let Err(not_included) = inner_processing_results {
929					defensive!(
930						"electable stashes exceeded limit, unexpected but election proceeds.\
931                		{} stashes from election result discarded",
932						not_included
933					);
934				};
935
936				Pallet::<T>::deposit_event(Event::PagedElectionProceeded {
937					page,
938					result: inner_processing_results.map(|x| x as u32).map_err(|x| x as u32),
939				});
940			},
941			Err(e) => {
942				log!(warn, "election provider page failed due to {:?} (page: {})", e, page);
943				Pallet::<T>::deposit_event(Event::PagedElectionProceeded { page, result: Err(0) });
944			},
945		}
946	}
947
948	/// Inner implementation of [`Self::do_elect_paged`].
949	///
950	/// Returns an error if adding election winners to the electable stashes storage fails due
951	/// to exceeded bounds. In case of error, it returns the index of the first stash that
952	/// failed to be included.
953	pub(crate) fn do_elect_paged_inner(
954		mut supports: BoundedSupportsOf<T::ElectionProvider>,
955	) -> Result<usize, usize> {
956		let planning_era = Rotator::<T>::planned_era();
957
958		match Self::add_electables(supports.iter().map(|(s, _)| s.clone())) {
959			Ok(added) => {
960				let exposures = Self::collect_exposures(supports);
961				let _ = Self::store_stakers_info(exposures, planning_era);
962				Ok(added)
963			},
964			Err(not_included_idx) => {
965				let not_included = supports.len().saturating_sub(not_included_idx);
966
967				log!(
968					warn,
969					"not all winners fit within the electable stashes, excluding {:?} accounts from solution.",
970					not_included,
971				);
972
973				// filter out supports of stashes that do not fit within the electable stashes
974				// storage bounds to prevent collecting their exposures.
975				supports.truncate(not_included_idx);
976				let exposures = Self::collect_exposures(supports);
977				let _ = Self::store_stakers_info(exposures, planning_era);
978
979				Err(not_included)
980			},
981		}
982	}

984	/// Process the output of a paged election.
985	///
986	/// Store staking information for the new planned era of a single election page.
987	pub(crate) fn store_stakers_info(
988		exposures: BoundedExposuresOf<T>,
989		new_planned_era: EraIndex,
990	) -> BoundedVec<T::AccountId, MaxWinnersPerPageOf<T::ElectionProvider>> {
991		// populate elected stash, stakers, exposures, and the snapshot of validator prefs.
992		let mut total_stake_page: BalanceOf<T> = Zero::zero();
993		let mut elected_stashes_page = Vec::with_capacity(exposures.len());
994		let mut total_backers = 0u32;
995
996		exposures.into_iter().for_each(|(stash, exposure)| {
997			log!(
998				trace,
999				"stored exposure for stash {:?} and {:?} backers",
1000				stash,
1001				exposure.others.len()
1002			);
1003			// build elected stash.
1004			elected_stashes_page.push(stash.clone());
1005			// accumulate total stake.
1006			total_stake_page = total_stake_page.saturating_add(exposure.total);
1007			// set or update staker exposure for this era.
1008			total_backers += exposure.others.len() as u32;
1009			Eras::<T>::upsert_exposure(new_planned_era, &stash, exposure);
1010		});
1011
1012		let elected_stashes: BoundedVec<_, MaxWinnersPerPageOf<T::ElectionProvider>> =
1013			elected_stashes_page
1014				.try_into()
1015				.expect("both types are bounded by MaxWinnersPerPageOf; qed");
1016
1017		// adds to total stake in this era.
1018		Eras::<T>::add_total_stake(new_planned_era, total_stake_page);
1019
1020		// collect or update the pref of all winners.
1021		for stash in &elected_stashes {
1022			let pref = Validators::<T>::get(stash);
1023			Eras::<T>::set_validator_prefs(new_planned_era, stash, pref);
1024		}
1025
1026		log!(
1027			debug,
1028			"stored a page of stakers with {:?} validators and {:?} total backers for era {:?}",
1029			elected_stashes.len(),
1030			total_backers,
1031			new_planned_era,
1032		);
1033
1034		elected_stashes
1035	}

1037	/// Consume a set of [`BoundedSupports`] from [`sp_npos_elections`] and collect them into a
1038	/// [`Exposure`].
1039	///
1040	/// Returns vec of all the exposures of a validator in `paged_supports`, bounded by the
1041	/// number of max winners per page returned by the election provider.
1042	fn collect_exposures(
1043		supports: BoundedSupportsOf<T::ElectionProvider>,
1044	) -> BoundedExposuresOf<T> {
1045		let total_issuance = asset::total_issuance::<T>();
1046		let to_currency = |e: frame_election_provider_support::ExtendedBalance| {
1047			T::CurrencyToVote::to_currency(e, total_issuance)
1048		};
1049
1050		supports
1051			.into_iter()
1052			.map(|(validator, support)| {
1053				// Build `struct exposure` from `support`.
1054				let mut others = Vec::with_capacity(support.voters.len());
1055				let mut own: BalanceOf<T> = Zero::zero();
1056				let mut total: BalanceOf<T> = Zero::zero();
1057				support
1058					.voters
1059					.into_iter()
1060					.map(|(nominator, weight)| (nominator, to_currency(weight)))
1061					.for_each(|(nominator, stake)| {
1062						if nominator == validator {
1063							defensive_assert!(own == Zero::zero(), "own stake should be unique");
1064							own = own.saturating_add(stake);
1065						} else {
1066							others.push(IndividualExposure { who: nominator, value: stake });
1067						}
1068						total = total.saturating_add(stake);
1069					});
1070
1071				let exposure = Exposure { own, others, total };
1072				(validator, exposure)
1073			})
1074			.try_collect()
1075			.expect("we only map through support vector which cannot change the size; qed")
1076	}

1078	/// Adds a new set of stashes to the electable stashes.
1079	///
1080	/// Returns:
1081	///
1082	/// `Ok(newly_added)` if all stashes were added successfully.
1083	/// `Err(first_un_included)` if some stashes cannot be added due to bounds.
1084	pub(crate) fn add_electables(
1085		new_stashes: impl Iterator<Item = T::AccountId>,
1086	) -> Result<usize, usize> {
1087		ElectableStashes::<T>::mutate(|electable| {
1088			let pre_size = electable.len();
1089
1090			for (idx, stash) in new_stashes.enumerate() {
1091				if electable.try_insert(stash).is_err() {
1092					return Err(idx);
1093				}
1094			}
1095
1096			Ok(electable.len() - pre_size)
1097		})
1098	}
