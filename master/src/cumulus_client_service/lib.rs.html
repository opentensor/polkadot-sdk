153		import_queue,
154		collator_key,

157		sync_service,

159	}: StartCollatorParams<'a, Block, BS, Client, RCInterface, Spawner>,

193		sync_service,

195	})?;

202		overseer_handle,
203		spawner,

205		key: collator_key,

207	})
208	.await;

210	Ok(())

213/// Start necessary consensus tasks related to the relay chain.
214///
215/// Parachain nodes need to track the state of the relay chain and use the
216/// relay chain's data availability service to fetch blocks if they don't
217/// arrive via the normal p2p layer (i.e. when authors withhold their blocks deliberately).
218///
219/// This function spawns work for those side tasks.
220///
221/// It also spawns a parachain informant task that will log the relay chain state and some metrics.
222pub fn start_relay_chain_tasks<Block, Client, Backend, RCInterface>(
223	StartRelayChainTasksParams {
224		client,
225		announce_block,
226		para_id,
227		task_manager,
228		da_recovery_profile,
229		relay_chain_interface,
230		import_queue,
231		relay_chain_slot_duration,
232		recovery_handle,
233		sync_service,
234		prometheus_registry,
235	}: StartRelayChainTasksParams<Block, Client, RCInterface>,
236) -> sc_service::error::Result<()>
237where
238	Block: BlockT,
239	Client: Finalizer<Block, Backend>
240		+ UsageProvider<Block>
241		+ HeaderBackend<Block>
242		+ Send
243		+ Sync
244		+ BlockBackend<Block>
245		+ BlockchainEvents<Block>
246		+ 'static,
247	for<'a> &'a Client: BlockImport<Block>,
248	Backend: BackendT<Block> + 'static,
249	RCInterface: RelayChainInterface + Clone + 'static,
250{
251	let (recovery_chan_tx, recovery_chan_rx) = mpsc::channel(RECOVERY_CHAN_SIZE);
252
253	let consensus = cumulus_client_consensus_common::run_parachain_consensus(
254		para_id,
255		client.clone(),
256		relay_chain_interface.clone(),
257		announce_block.clone(),
258		Some(recovery_chan_tx),
259	);
260
261	task_manager
262		.spawn_essential_handle()
263		.spawn_blocking("cumulus-consensus", None, consensus);
264
265	let da_recovery_profile = match da_recovery_profile {
266		DARecoveryProfile::Collator => {
267			// We want that collators wait at maximum the relay chain slot duration before starting
268			// to recover blocks. Additionally, we wait at least half the slot time to give the
269			// relay chain the chance to increase availability.
270			RecoveryDelayRange {
271				min: relay_chain_slot_duration / 2,
272				max: relay_chain_slot_duration,
273			}
274		},
275		DARecoveryProfile::FullNode => {
276			// Full nodes should at least wait 2.5 minutes (assuming 6 seconds slot duration) and
277			// in maximum 5 minutes before starting to recover blocks. Collators should already
278			// start the recovery way before full nodes try to recover a certain block and then
279			// share the block with the network using "the normal way". Full nodes are just the
280			// "last resort" for block recovery.
281			RecoveryDelayRange {
282				min: relay_chain_slot_duration * 25,
283				max: relay_chain_slot_duration * 50,
284			}
285		},
286		DARecoveryProfile::Other(profile) => profile,
287	};
288
289	let pov_recovery = PoVRecovery::new(
290		recovery_handle,
291		da_recovery_profile,
292		client.clone(),
293		import_queue,
294		relay_chain_interface.clone(),
295		para_id,
296		recovery_chan_rx,
297		sync_service.clone(),
298	);
299
300	task_manager
301		.spawn_essential_handle()
302		.spawn("cumulus-pov-recovery", None, pov_recovery.run());
303
304	let parachain_informant = parachain_informant::<Block, _>(
305		relay_chain_interface.clone(),
306		client.clone(),
307		prometheus_registry.map(ParachainInformantMetrics::new).transpose()?,
308	);
309	task_manager
310		.spawn_handle()
311		.spawn("parachain-informant", None, parachain_informant);
312
313	Ok(())
314}

372/// Prepare the parachain's node configuration
373///
374/// This function will:
375/// * Disable the default announcement of Substrate for the parachain in favor of the one of
376///   Cumulus.
377/// * Set peers needed to start warp sync to 1.
378pub fn prepare_node_config(mut parachain_config: Configuration) -> Configuration {
379	parachain_config.announce_block = false;
380	// Parachains only need 1 peer to start warp sync, because the target block is fetched from the

383
384	parachain_config

387/// Build a relay chain interface.
388/// Will return a minimal relay chain node with RPC
389/// client or an inprocess node, based on the [`CollatorOptions`] passed in.

391	relay_chain_config: Configuration,

394	task_manager: &mut TaskManager,

414				parachain_config.prometheus_registry(),
415				task_manager,
416				rpc_target_urls,

419		cumulus_client_cli::RelayChainMode::LightClient =>
420			build_minimal_relay_chain_node_light_client(relay_chain_config, task_manager).await,
421	}

424/// The expected level of collator sybil-resistance on the network. This is used to
425/// configure the type of metadata passed alongside block announcements on the network.
426pub enum CollatorSybilResistance {
427	/// There is a collator-selection protocol which provides sybil-resistance,
428	/// such as Aura. Sybil-resistant collator-selection protocols are able to

434	/// validators to avoid handling unbounded numbers of blocks.
435	Unresistant,
436}

467/// Build the network service, the network status sinks and an RPC sender.
468pub async fn build_network<'a, Block, Client, RCInterface, IQ, Network>(
469	BuildNetworkParams {
470		parachain_config,
471		net_config,
472		client,
473		transaction_pool,
474		para_id,
475		spawn_handle,
476		relay_chain_interface,
477		import_queue,
478		sybil_resistance_level,
479		metrics,
480	}: BuildNetworkParams<'a, Block, Client, Network, RCInterface, IQ>,
481) -> sc_service::error::Result<(
482	Arc<dyn NetworkService>,
483	TracingUnboundedSender<sc_rpc::system::Request<Block>>,
484	TransactionsHandlerController<Block::Hash>,
485	Arc<SyncingService<Block>>,
486)>
487where
488	Block: BlockT,
489	Client: UsageProvider<Block>
490		+ HeaderBackend<Block>
491		+ sp_consensus::block_validation::Chain<Block>
492		+ Send
493		+ Sync
494		+ BlockBackend<Block>
495		+ BlockchainEvents<Block>
496		+ ProvideRuntimeApi<Block>
497		+ HeaderMetadata<Block, Error = sp_blockchain::Error>
498		+ BlockIdTo<Block, Error = sp_blockchain::Error>
499		+ ProofProvider<Block>
500		+ 'static,
501	Client::Api: CollectCollationInfo<Block>
502		+ sp_transaction_pool::runtime_api::TaggedTransactionQueue<Block>,
503	for<'b> &'b Client: BlockImport<Block>,
504	RCInterface: RelayChainInterface + Clone + 'static,
505	IQ: ImportQueue<Block> + 'static,
506	Network: NetworkBackend<Block, <Block as BlockT>::Hash>,
507{
508	let warp_sync_config = match parachain_config.network.sync_mode {
509		SyncMode::Warp => {
510			log::debug!(target: LOG_TARGET_SYNC, "waiting for announce block...");
511
512			let target_block =
513				wait_for_finalized_para_head::<Block, _>(para_id, relay_chain_interface.clone())
514					.await
515					.inspect_err(|e| {
516						log::error!(
517							target: LOG_TARGET_SYNC,
518							"Unable to determine parachain target block {:?}",
519							e
520						);
521					})?;
522			Some(WarpSyncConfig::WithTarget(target_block))
523		},
524		_ => None,
525	};
526
527	let block_announce_validator = match sybil_resistance_level {
528		CollatorSybilResistance::Resistant => {
529			let block_announce_validator = AssumeSybilResistance::allow_seconded_messages();
530			Box::new(block_announce_validator) as Box<_>
531		},
532		CollatorSybilResistance::Unresistant => {
533			let block_announce_validator =
534				RequireSecondedInBlockAnnounce::new(relay_chain_interface, para_id);
535			Box::new(block_announce_validator) as Box<_>
536		},
537	};
538
539	sc_service::build_network(sc_service::BuildNetworkParams {
540		config: parachain_config,
541		net_config,
542		client,
543		transaction_pool,
544		spawn_handle,
545		import_queue,
546		block_announce_validator_builder: Some(Box::new(move |_| block_announce_validator)),
547		warp_sync_config,
548		block_relay: None,
549		metrics,
550	})
551}

553/// Waits for the relay chain to have finished syncing and then gets the parachain header that
554/// corresponds to the last finalized relay chain block.
555async fn wait_for_finalized_para_head<B, RCInterface>(
556	para_id: ParaId,
557	relay_chain_interface: RCInterface,
558) -> sc_service::error::Result<<B as BlockT>::Header>
559where
560	B: BlockT + 'static,
561	RCInterface: RelayChainInterface + Send + 'static,
562{
563	let mut imported_blocks = relay_chain_interface
564		.import_notification_stream()
565		.await
566		.map_err(|error| {
567			sc_service::Error::Other(format!(
568				"Relay chain import notification stream error when waiting for parachain head: \
569				{error}"
570			))
571		})?
572		.fuse();
573	while imported_blocks.next().await.is_some() {
574		let is_syncing = relay_chain_interface
575			.is_major_syncing()
576			.await
577			.map_err(|e| format!("Unable to determine sync status: {e}"))?;
578
579		if !is_syncing {
580			let relay_chain_best_hash = relay_chain_interface
581				.finalized_block_hash()
582				.await
583				.map_err(|e| Box::new(e) as Box<_>)?;
584
585			let validation_data = relay_chain_interface
586				.persisted_validation_data(
587					relay_chain_best_hash,
588					para_id,
589					OccupiedCoreAssumption::TimedOut,
590				)
591				.await
592				.map_err(|e| format!("{e:?}"))?
593				.ok_or("Could not find parachain head in relay chain")?;
594
595			let finalized_header = B::Header::decode(&mut &validation_data.parent_head.0[..])
596				.map_err(|e| format!("Failed to decode parachain head: {e}"))?;
597
598			log::info!(
599				"๐ Received target parachain header #{} ({}) from the relay chain.",
600				finalized_header.number(),
601				finalized_header.hash()
602			);
603			return Ok(finalized_header)
604		}
605	}
606
607	Err("Stopping following imported blocks. Could not determine parachain target block".into())
608}

610/// Task for logging candidate events and some related metrics.
611async fn parachain_informant<Block: BlockT, Client>(
612	relay_chain_interface: impl RelayChainInterface + Clone,
613	client: Arc<Client>,
614	metrics: Option<ParachainInformantMetrics>,
615) where
616	Client: HeaderBackend<Block> + Send + Sync + 'static,
617{
618	let mut import_notifications = match relay_chain_interface.import_notification_stream().await {
619		Ok(import_notifications) => import_notifications,
620		Err(e) => {
621			log::error!("Failed to get import notification stream: {e:?}. Parachain informant will not run!");
622			return
623		},
624	};
625	let mut last_backed_block_time: Option<Instant> = None;
626	while let Some(n) = import_notifications.next().await {
627		let candidate_events = match relay_chain_interface.candidate_events(n.hash()).await {
628			Ok(candidate_events) => candidate_events,
629			Err(e) => {
630				log::warn!("Failed to get candidate events for block {}: {e:?}", n.hash());
631				continue
632			},
633		};
634		let mut backed_candidates = Vec::new();
635		let mut included_candidates = Vec::new();
636		let mut timed_out_candidates = Vec::new();
637		for event in candidate_events {
638			match event {
639				CandidateEvent::CandidateBacked(_, head, _, _) => {
640					let backed_block = match Block::Header::decode(&mut &head.0[..]) {
641						Ok(header) => header,
642						Err(e) => {
643							log::warn!(
644								"Failed to decode parachain header from backed block: {e:?}"
645							);
646							continue
647						},
648					};
649					let backed_block_time = Instant::now();
650					if let Some(last_backed_block_time) = &last_backed_block_time {
651						let duration = backed_block_time.duration_since(*last_backed_block_time);
652						if let Some(metrics) = &metrics {
653							metrics.parachain_block_backed_duration.observe(duration.as_secs_f64());
654						}
655					}
656					last_backed_block_time = Some(backed_block_time);
657					backed_candidates.push(backed_block);
658				},
659				CandidateEvent::CandidateIncluded(_, head, _, _) => {
660					let included_block = match Block::Header::decode(&mut &head.0[..]) {
661						Ok(header) => header,
662						Err(e) => {
663							log::warn!(
664								"Failed to decode parachain header from included block: {e:?}"
665							);
666							continue
667						},
668					};
669					let unincluded_segment_size =
670						client.info().best_number.saturating_sub(*included_block.number());
671					let unincluded_segment_size: u32 = unincluded_segment_size.saturated_into();
672					if let Some(metrics) = &metrics {
673						metrics.unincluded_segment_size.observe(unincluded_segment_size.into());
674					}
675					included_candidates.push(included_block);
676				},
677				CandidateEvent::CandidateTimedOut(_, head, _) => {
678					let timed_out_block = match Block::Header::decode(&mut &head.0[..]) {
679						Ok(header) => header,
680						Err(e) => {
681							log::warn!(
682								"Failed to decode parachain header from timed out block: {e:?}"
683							);
684							continue
685						},
686					};
687					timed_out_candidates.push(timed_out_block);
688				},
689			}
690		}
691		let mut log_parts = Vec::new();
692		if !backed_candidates.is_empty() {
693			let backed_candidates = backed_candidates
694				.into_iter()
695				.map(|c| format!("#{} ({})", c.number(), c.hash()))
696				.collect::<Vec<_>>()
697				.join(", ");
698			log_parts.push(format!("backed: {}", backed_candidates));
699		};
700		if !included_candidates.is_empty() {
701			let included_candidates = included_candidates
702				.into_iter()
703				.map(|c| format!("#{} ({})", c.number(), c.hash()))
704				.collect::<Vec<_>>()
705				.join(", ");
706			log_parts.push(format!("included: {}", included_candidates));
707		};
708		if !timed_out_candidates.is_empty() {
709			let timed_out_candidates = timed_out_candidates
710				.into_iter()
711				.map(|c| format!("#{} ({})", c.number(), c.hash()))
712				.collect::<Vec<_>>()
713				.join(", ");
714			log_parts.push(format!("timed out: {}", timed_out_candidates));
715		};
716		if !log_parts.is_empty() {
717			log::info!(
718				"Update at relay chain block #{} ({}) - {}",
719				n.number(),
720				n.hash(),
721				log_parts.join(", ")
722			);
723		}
724	}
725}
726
727struct ParachainInformantMetrics {
728	/// Time between parachain blocks getting backed by the relaychain.
729	parachain_block_backed_duration: Histogram,
730	/// Number of blocks between best block and last included block.
731	unincluded_segment_size: Histogram,
732}
733
734impl ParachainInformantMetrics {
735	fn new(prometheus_registry: &Registry) -> prometheus::Result<Self> {
736		let parachain_block_authorship_duration = Histogram::with_opts(HistogramOpts::new(
737			"parachain_block_backed_duration",
738			"Time between parachain blocks getting backed by the relaychain",
739		))?;
740		prometheus_registry.register(Box::new(parachain_block_authorship_duration.clone()))?;
741
742		let unincluded_segment_size = Histogram::with_opts(
743			HistogramOpts::new(
744				"parachain_unincluded_segment_size",
745				"Number of blocks between best block and last included block",
746			)
747			.buckets((0..=24).into_iter().map(|i| i as f64).collect()),
748		)?;
749		prometheus_registry.register(Box::new(unincluded_segment_size.clone()))?;
750
751		Ok(Self {
752			parachain_block_backed_duration: parachain_block_authorship_duration,
753			unincluded_segment_size,
754		})
755	}
756}
