46// This is an arbitrary value which is likely guaranteed to exceed any reasonable
47// limit, as it would correspond to 30 non-included blocks.
48//
49// Since we only search for parent blocks which have already been imported,
50// we can guarantee that all imported blocks respect the unincluded segment
51// rules specified by the parachain's runtime and thus will never be too deep. This is just an extra
52// sanity check.
53const PARENT_SEARCH_DEPTH: usize = 30;

55/// Check the `local_validation_code_hash` against the validation code hash in the relay chain
56/// state.
57///
58/// If the code hashes do not match, it prints a warning.
59async fn check_validation_code_or_log(
60	local_validation_code_hash: &ValidationCodeHash,
61	para_id: ParaId,
62	relay_client: &impl RelayChainInterface,
63	relay_parent: RelayHash,
64) {
65	let state_validation_code_hash = match relay_client
66		.validation_code_hash(relay_parent, para_id, OccupiedCoreAssumption::Included)
67		.await
68	{
69		Ok(hash) => hash,
70		Err(error) => {
71			tracing::debug!(
72				target: super::LOG_TARGET,
73				%error,
74				?relay_parent,
75				%para_id,
76				"Failed to fetch validation code hash",
77			);
78			return
79		},
80	};
81
82	match state_validation_code_hash {
83		Some(state) =>
84			if state != *local_validation_code_hash {
85				tracing::warn!(
86					target: super::LOG_TARGET,
87					%para_id,
88					?relay_parent,
89					?local_validation_code_hash,
90					relay_validation_code_hash = ?state,
91					"Parachain code doesn't match validation code stored in the relay chain state.",
92				);
93			},
94		None => {
95			tracing::warn!(
96				target: super::LOG_TARGET,
97				%para_id,
98				?relay_parent,
99				"Could not find validation code for parachain in the relay chain state.",
100			);
101		},
102	}
103}

105/// Fetch scheduling lookahead at given relay parent.
106async fn scheduling_lookahead(
107	relay_parent: RelayHash,
108	relay_client: &impl RelayChainInterface,
109) -> Option<u32> {
110	let runtime_api_version = relay_client
111		.version(relay_parent)
112		.await
113		.map_err(|e| {
114			tracing::error!(
115				target: super::LOG_TARGET,
116				error = ?e,
117				"Failed to fetch relay chain runtime version.",
118			)
119		})
120		.ok()?;
121
122	let parachain_host_runtime_api_version = runtime_api_version
123		.api_version(
124			&<dyn polkadot_primitives::runtime_api::ParachainHost<polkadot_primitives::Block>>::ID,
125		)
126		.unwrap_or_default();
127
128	if parachain_host_runtime_api_version <
129		RuntimeApiRequest::SCHEDULING_LOOKAHEAD_RUNTIME_REQUIREMENT
130	{
131		return None
132	}
133
134	match relay_client.scheduling_lookahead(relay_parent).await {
135		Ok(scheduling_lookahead) => Some(scheduling_lookahead),
136		Err(err) => {
137			tracing::error!(
138				target: crate::LOG_TARGET,
139				?err,
140				?relay_parent,
141				"Failed to fetch scheduling lookahead from relay chain",
142			);
143			None
144		},
145	}
146}

148// Return all the cores assigned to the para at the provided relay parent, using the claim queue
149// offset.
150// Will return an empty vec if the provided offset is higher than the claim queue length (which
151// corresponds to the scheduling_lookahead on the relay chain).
152async fn cores_scheduled_for_para(
153	relay_parent: RelayHash,
154	para_id: ParaId,
155	relay_client: &impl RelayChainInterface,
156	claim_queue_offset: ClaimQueueOffset,
157) -> Vec<CoreIndex> {
158	// Get `ClaimQueue` from runtime
159	let claim_queue: ClaimQueueSnapshot = match relay_client.claim_queue(relay_parent).await {
160		Ok(claim_queue) => claim_queue.into(),
161		Err(error) => {
162			tracing::error!(
163				target: crate::LOG_TARGET,
164				?error,
165				?relay_parent,
166				"Failed to query claim queue runtime API",
167			);
168			return Vec::new()
169		},
170	};
171
172	claim_queue
173		.iter_claims_at_depth(claim_queue_offset.0 as usize)
174		.filter_map(|(core_index, core_para_id)| (core_para_id == para_id).then_some(core_index))
175		.collect()
176}

178// Checks if we own the slot at the given block and whether there
179// is space in the unincluded segment.
180async fn can_build_upon<Block: BlockT, Client, P>(
181	para_slot: Slot,
182	relay_slot: Slot,
183	timestamp: Timestamp,
184	parent_hash: Block::Hash,
185	included_block: Block::Hash,
186	client: &Client,
187	keystore: &KeystorePtr,
188) -> Option<SlotClaim<P::Public>>
189where
190	Client: ProvideRuntimeApi<Block>,
191	Client::Api: AuraApi<Block, P::Public> + AuraUnincludedSegmentApi<Block> + ApiExt<Block>,
192	P: Pair,
193	P::Public: Codec,
194	P::Signature: Codec,
195{
196	let runtime_api = client.runtime_api();
197	let authorities = runtime_api.authorities(parent_hash).ok()?;
198	let author_pub = aura_internal::claim_slot::<P>(para_slot, &authorities, keystore).await?;
199
200	// This function is typically called when we want to build block N. At that point, the
201	// unincluded segment in the runtime is unaware of the hash of block N-1. If the unincluded
202	// segment in the runtime is full, but block N-1 is the included block, the unincluded segment
203	// should have length 0 and we can build. Since the hash is not available to the runtime
204	// however, we need this extra check here.
205	if parent_hash == included_block {
206		return Some(SlotClaim::unchecked::<P>(author_pub, para_slot, timestamp));
207	}
208
209	let api_version = runtime_api
210		.api_version::<dyn AuraUnincludedSegmentApi<Block>>(parent_hash)
211		.ok()
212		.flatten()?;
213
214	let slot = if api_version > 1 { relay_slot } else { para_slot };
215
216	runtime_api
217		.can_build_upon(parent_hash, included_block, slot)
218		.ok()?
219		.then(|| SlotClaim::unchecked::<P>(author_pub, para_slot, timestamp))
220}

222/// Use [`cumulus_client_consensus_common::find_potential_parents`] to find parachain blocks that
223/// we can build on. Once a list of potential parents is retrieved, return the last one of the
224/// longest chain.
225async fn find_parent<Block>(
226	relay_parent: RelayHash,
227	para_id: ParaId,
228	para_backend: &impl sc_client_api::Backend<Block>,
229	relay_client: &impl RelayChainInterface,
230) -> Option<(<Block as BlockT>::Header, consensus_common::PotentialParent<Block>)>
231where
232	Block: BlockT,
233{
234	let parent_search_params = ParentSearchParams {
235		relay_parent,
236		para_id,
237		ancestry_lookback: scheduling_lookahead(relay_parent, relay_client)
238			.await
239			.unwrap_or(DEFAULT_SCHEDULING_LOOKAHEAD)
240			.saturating_sub(1) as usize,
241		max_depth: PARENT_SEARCH_DEPTH,
242		ignore_alternative_branches: true,
243	};
244
245	let potential_parents = cumulus_client_consensus_common::find_potential_parents::<Block>(
246		parent_search_params,
247		para_backend,
248		relay_client,
249	)
250	.await;
251
252	let potential_parents = match potential_parents {
253		Err(e) => {
254			tracing::error!(
255				target: crate::LOG_TARGET,
256				?relay_parent,
257				err = ?e,
258				"Could not fetch potential parents to build upon"
259			);
260
261			return None
262		},
263		Ok(x) => x,
264	};
265
266	let included_block = potential_parents.iter().find(|x| x.depth == 0)?.header.clone();
267	potential_parents
268		.into_iter()
269		.max_by_key(|a| a.depth)
270		.map(|parent| (included_block, parent))
271}

294		importer: &I,
295		block: Block,
296		origin: BlockOrigin,
297		import_as_best: bool,

305	}

341	/// This tests a special scenario where the unincluded segment in the runtime
342	/// is full. We are calling `can_build_upon`, passing the last built block as the
343	/// included one. In the runtime we will not find the hash of the included block in the
344	/// unincluded segment. The `can_build_upon` runtime API would therefore return `false`, but
345	/// we are ensuring on the node side that we are are always able to build on the included block.
346	#[tokio::test]
347	async fn test_can_build_upon() {
348		let (client, keystore) = set_up_components();
349
350		let genesis_hash = client.chain_info().genesis_hash;
351		let mut last_hash = genesis_hash;
352
353		// Fill up the unincluded segment tracker in the runtime.
354		while can_build_upon::<_, _, sp_consensus_aura::sr25519::AuthorityPair>(
355			Slot::from(u64::MAX),
356			Slot::from(u64::MAX),
357			Timestamp::default(),
358			last_hash,
359			genesis_hash,
360			&*client,
361			&keystore,
362		)
363		.await
364		.is_some()
365		{
366			let block = build_and_import_block(&client, genesis_hash).await;
367			last_hash = block.header().hash();
368		}
369
370		// Blocks were built with the genesis hash set as included block.
371		// We call `can_build_upon` with the last built block as the included block.
372		let result = can_build_upon::<_, _, sp_consensus_aura::sr25519::AuthorityPair>(
373			Slot::from(u64::MAX),
374			Slot::from(u64::MAX),
375			Timestamp::default(),
376			last_hash,
377			last_hash,
378			&*client,
379			&keystore,
380		)
381		.await;
382		assert!(result.is_some());
383	}

386/// Holds a relay parent and its descendants.
387pub struct RelayParentData {
388	/// The relay parent block header
389	relay_parent: RelayHeader,
390	/// Ordered collection of descendant block headers, from oldest to newest
391	descendants: Vec<RelayHeader>,
392}
393
394impl RelayParentData {
395	/// Creates a new instance with the given relay parent and no descendants.
396	pub fn new(relay_parent: RelayHeader) -> Self {
397		Self { relay_parent, descendants: Default::default() }
398	}
399
400	/// Creates a new instance with the given relay parent and descendants.
401	pub fn new_with_descendants(relay_parent: RelayHeader, descendants: Vec<RelayHeader>) -> Self {
402		Self { relay_parent, descendants }
403	}
404
405	/// Returns a reference to the relay parent header.
406	pub fn relay_parent(&self) -> &RelayHeader {
407		&self.relay_parent
408	}
409
410	/// Returns the number of descendants.
411	#[cfg(test)]
412	pub fn descendants_len(&self) -> usize {
413		self.descendants.len()
414	}
415
416	/// Consumes the structure and returns a vector containing the relay parent followed by its
417	/// descendants in chronological order. The resulting list should be provided to the parachain
418	/// inherent data.
419	pub fn into_inherent_descendant_list(self) -> Vec<RelayHeader> {
420		let Self { relay_parent, mut descendants } = self;
421
422		if descendants.is_empty() {
423			return Default::default()
424		}
425
426		let mut result = vec![relay_parent];
427		result.append(&mut descendants);
428		result
429	}
