53/// A trait that allows to decouple the metering from the charging of balance.
54///
55/// This mostly exists for testing so that the charging can be mocked.
56pub trait Ext<T: Config> {
57	/// This is called to inform the implementer that some balance should be charged due to
58	/// some interaction of the `origin` with a `contract`.
59	///
60	/// The balance transfer can either flow from `origin` to `contract` or the other way
61	/// around depending on whether `amount` constitutes a `Charge` or a `Refund`.
62	/// It will fail in case the `origin` has not enough balance to cover all storage deposits.
63	fn charge(
64		origin: &T::AccountId,
65		contract: &T::AccountId,
66		amount: &DepositOf<T>,
67		state: &ContractState<T>,
68	) -> Result<(), DispatchError>;
69}

93/// A type that allows the metering of consumed or freed storage of a single contract call stack.
94#[derive(DefaultNoBound, RuntimeDebugNoBound)]
95pub struct RawMeter<T: Config, E, S: State + Default + Debug> {
96	/// The limit of how much balance this meter is allowed to consume.
97	limit: BalanceOf<T>,
98	/// The amount of balance that was used in this meter and all of its already absorbed children.
99	total_deposit: DepositOf<T>,
100	/// The amount of storage changes that were recorded in this meter alone.
101	own_contribution: Contribution<T>,
102	/// List of charges that should be applied at the end of a contract stack execution.
103	///
104	/// We only have one charge per contract hence the size of this vector is
105	/// limited by the maximum call depth.
106	charges: Vec<Charge<T>>,
107	/// Type parameter only used in impls.
108	_phantom: PhantomData<(E, S)>,
109}
110
111/// This type is used to describe a storage change when charging from the meter.
112#[derive(Default, RuntimeDebugNoBound)]
113pub struct Diff {
114	/// How many bytes were added to storage.
115	pub bytes_added: u32,
116	/// How many bytes were removed from storage.
117	pub bytes_removed: u32,
118	/// How many storage items were added to storage.
119	pub items_added: u32,
120	/// How many storage items were removed from storage.
121	pub items_removed: u32,
122}
123
124impl Diff {
125	/// Calculate how much of a charge or refund results from applying the diff and store it
126	/// in the passed `info` if any.
127	///
128	/// # Note
129	///
130	/// In case `None` is passed for `info` only charges are calculated. This is because refunds
131	/// are calculated pro rata of the existing storage within a contract and hence need extract
132	/// this information from the passed `info`.
133	pub fn update_contract<T: Config>(&self, info: Option<&mut ContractInfo<T>>) -> DepositOf<T> {
134		let per_byte = T::DepositPerByte::get();
135		let per_item = T::DepositPerItem::get();
136		let bytes_added = self.bytes_added.saturating_sub(self.bytes_removed);
137		let items_added = self.items_added.saturating_sub(self.items_removed);
138		let mut bytes_deposit = Deposit::Charge(per_byte.saturating_mul((bytes_added).into()));
139		let mut items_deposit = Deposit::Charge(per_item.saturating_mul((items_added).into()));
140
141		// Without any contract info we can only calculate diffs which add storage
142		let info = if let Some(info) = info {
143			info
144		} else {
145			debug_assert_eq!(self.bytes_removed, 0);
146			debug_assert_eq!(self.items_removed, 0);
147			return bytes_deposit.saturating_add(&items_deposit)
148		};
149
150		// Refunds are calculated pro rata based on the accumulated storage within the contract
151		let bytes_removed = self.bytes_removed.saturating_sub(self.bytes_added);
152		let items_removed = self.items_removed.saturating_sub(self.items_added);
153		let ratio = FixedU128::checked_from_rational(bytes_removed, info.storage_bytes)
154			.unwrap_or_default()
155			.min(FixedU128::from_u32(1));
156		bytes_deposit = bytes_deposit
157			.saturating_add(&Deposit::Refund(ratio.saturating_mul_int(info.storage_byte_deposit)));
158		let ratio = FixedU128::checked_from_rational(items_removed, info.storage_items)
159			.unwrap_or_default()
160			.min(FixedU128::from_u32(1));
161		items_deposit = items_deposit
162			.saturating_add(&Deposit::Refund(ratio.saturating_mul_int(info.storage_item_deposit)));
163
164		// We need to update the contract info structure with the new deposits
165		info.storage_bytes =
166			info.storage_bytes.saturating_add(bytes_added).saturating_sub(bytes_removed);
167		info.storage_items =
168			info.storage_items.saturating_add(items_added).saturating_sub(items_removed);
169		match &bytes_deposit {
170			Deposit::Charge(amount) =>
171				info.storage_byte_deposit = info.storage_byte_deposit.saturating_add(*amount),
172			Deposit::Refund(amount) =>
173				info.storage_byte_deposit = info.storage_byte_deposit.saturating_sub(*amount),
174		}
175		match &items_deposit {
176			Deposit::Charge(amount) =>
177				info.storage_item_deposit = info.storage_item_deposit.saturating_add(*amount),
178			Deposit::Refund(amount) =>
179				info.storage_item_deposit = info.storage_item_deposit.saturating_sub(*amount),
180		}
181
182		bytes_deposit.saturating_add(&items_deposit)
183	}
184}
185
186impl Diff {
187	fn saturating_add(&self, rhs: &Self) -> Self {
188		Self {
189			bytes_added: self.bytes_added.saturating_add(rhs.bytes_added),
190			bytes_removed: self.bytes_removed.saturating_add(rhs.bytes_removed),
191			items_added: self.items_added.saturating_add(rhs.items_added),
192			items_removed: self.items_removed.saturating_add(rhs.items_removed),
193		}
194	}
195}

197/// The state of a contract.
198///
199/// In case of termination the beneficiary is indicated.
200#[derive(RuntimeDebugNoBound, Clone, PartialEq, Eq)]
201pub enum ContractState<T: Config> {
202	Alive,
203	Terminated { beneficiary: AccountIdOf<T> },
204}
205
206/// Records information to charge or refund a plain account.
207///
208/// All the charges are deferred to the end of a whole call stack. Reason is that by doing
209/// this we can do all the refunds before doing any charge. This way a plain account can use
210/// more deposit than it has balance as along as it is covered by a refund. This
211/// essentially makes the order of storage changes irrelevant with regard to the deposit system.
212/// The only exception is when a special (tougher) deposit limit is specified for a cross-contract
213/// call. In that case the limit is enforced once the call is returned, rolling it back if
214/// exhausted.
215#[derive(RuntimeDebugNoBound, Clone)]
216struct Charge<T: Config> {
217	contract: T::AccountId,
218	amount: DepositOf<T>,
219	state: ContractState<T>,
220}
221
222/// Records the storage changes of a storage meter.
223#[derive(RuntimeDebugNoBound)]
224enum Contribution<T: Config> {
225	/// The contract the meter belongs to is alive and accumulates changes using a [`Diff`].
226	Alive(Diff),
227	/// The meter was checked against its limit using [`RawMeter::enforce_limit`] at the end of
228	/// its execution. In this process the [`Diff`] was converted into a [`Deposit`].
229	Checked(DepositOf<T>),
230	/// The contract was terminated. In this process the [`Diff`] was converted into a [`Deposit`]
231	/// in order to calculate the refund. Upon termination the `reducible_balance` in the
232	/// contract's account is transferred to the [`beneficiary`].
233	Terminated { deposit: DepositOf<T>, beneficiary: AccountIdOf<T> },
234}
235
236impl<T: Config> Contribution<T> {
237	/// See [`Diff::update_contract`].
238	fn update_contract(&self, info: Option<&mut ContractInfo<T>>) -> DepositOf<T> {
239		match self {
240			Self::Alive(diff) => diff.update_contract::<T>(info),
241			Self::Terminated { deposit, beneficiary: _ } | Self::Checked(deposit) =>
242				deposit.clone(),
243		}
244	}
245}
246
247impl<T: Config> Default for Contribution<T> {
248	fn default() -> Self {
249		Self::Alive(Default::default())
250	}
251}

253/// Functions that apply to all states.
254impl<T, E, S> RawMeter<T, E, S>
255where
256	T: Config,
257	E: Ext<T>,
258	S: State + Default + Debug,
259{
260	/// Create a new child that has its `limit`.
261	///
262	/// This is called whenever a new subcall is initiated in order to track the storage
263	/// usage for this sub call separately. This is necessary because we want to exchange balance
264	/// with the current contract we are interacting with.
265	pub fn nested(&self, limit: BalanceOf<T>) -> RawMeter<T, E, Nested> {
266		debug_assert!(matches!(self.contract_state(), ContractState::Alive));
267
268		RawMeter { limit: self.available().min(limit), ..Default::default() }
269	}
270
271	/// Absorb a child that was spawned to handle a sub call.
272	///
273	/// This should be called whenever a sub call comes to its end and it is **not** reverted.
274	/// This does the actual balance transfer from/to `origin` and `contract` based on the
275	/// overall storage consumption of the call. It also updates the supplied contract info.
276	///
277	/// In case a contract reverted the child meter should just be dropped in order to revert
278	/// any changes it recorded.
279	///
280	/// # Parameters
281	///
282	/// - `absorbed`: The child storage meter that should be absorbed.
283	/// - `origin`: The origin that spawned the original root meter.
284	/// - `contract`: The contract's account that this sub call belongs to.
285	/// - `info`: The info of the contract in question. `None` if the contract was terminated.
286	pub fn absorb(
287		&mut self,
288		absorbed: RawMeter<T, E, Nested>,
289		contract: &T::AccountId,
290		info: Option<&mut ContractInfo<T>>,
291	) {
292		let own_deposit = absorbed.own_contribution.update_contract(info);
293		self.total_deposit = self
294			.total_deposit
295			.saturating_add(&absorbed.total_deposit)
296			.saturating_add(&own_deposit);
297		self.charges.extend_from_slice(&absorbed.charges);
298		if !own_deposit.is_zero() {
299			self.charges.push(Charge {
300				contract: contract.clone(),
301				amount: own_deposit,
302				state: absorbed.contract_state(),
303			});
304		}
305	}
306
307	/// Record a charge that has taken place externally.
308	///
309	/// This will not perform a charge. It just records it to reflect it in the
310	/// total amount of storage required for a transaction.
311	pub fn record_charge(&mut self, amount: &DepositOf<T>) {
312		self.total_deposit = self.total_deposit.saturating_add(&amount);
313	}
314
315	/// The amount of balance that is still available from the original `limit`.
316	fn available(&self) -> BalanceOf<T> {
317		self.total_deposit.available(&self.limit)
318	}
319
320	/// Returns the state of the currently executed contract.
321	fn contract_state(&self) -> ContractState<T> {
322		match &self.own_contribution {
323			Contribution::Terminated { deposit: _, beneficiary } =>
324				ContractState::Terminated { beneficiary: beneficiary.clone() },
325			_ => ContractState::Alive,
326		}
327	}
328}

330/// Functions that only apply to the root state.
331impl<T, E> RawMeter<T, E, Root>
332where
333	T: Config,
334	E: Ext<T>,
335{
336	/// Create new storage limiting storage deposits to the passed `limit`.
337	///
338	/// If the limit larger then what the origin can afford we will just fail
339	/// when collecting the deposits in `try_into_deposit`.
340	pub fn new(limit: BalanceOf<T>) -> Self {
341		Self { limit, ..Default::default() }
342	}
343
344	/// Create new storage meter without checking the limit.
345	pub fn new_unchecked(limit: BalanceOf<T>) -> Self {
346		return Self { limit, ..Default::default() }
347	}
348
349	/// The total amount of deposit that should change hands as result of the execution
350	/// that this meter was passed into. This will also perform all the charges accumulated
351	/// in the whole contract stack.
352	///
353	/// This drops the root meter in order to make sure it is only called when the whole
354	/// execution did finish.
355	pub fn try_into_deposit(
356		self,
357		origin: &Origin<T>,
358		skip_transfer: bool,
359	) -> Result<DepositOf<T>, DispatchError> {
360		if !skip_transfer {
361			// Only refund or charge deposit if the origin is not root.
362			let origin = match origin {
363				Origin::Root => return Ok(Deposit::Charge(Zero::zero())),
364				Origin::Signed(o) => o,
365			};
366			let try_charge = || {
367				for charge in self.charges.iter().filter(|c| matches!(c.amount, Deposit::Refund(_)))
368				{
369					E::charge(origin, &charge.contract, &charge.amount, &charge.state)?;
370				}
371				for charge in self.charges.iter().filter(|c| matches!(c.amount, Deposit::Charge(_)))
372				{
373					E::charge(origin, &charge.contract, &charge.amount, &charge.state)?;
374				}
375				Ok(())
376			};
377			try_charge().map_err(|_: DispatchError| <Error<T>>::StorageDepositNotEnoughFunds)?;
378		}
379
380		Ok(self.total_deposit)
381	}
382}

384/// Functions that only apply to the nested state.
385impl<T: Config, E: Ext<T>> RawMeter<T, E, Nested> {
386	/// Charges `diff` from the meter.
387	pub fn charge(&mut self, diff: &Diff) {
388		match &mut self.own_contribution {
389			Contribution::Alive(own) => *own = own.saturating_add(diff),
390			_ => panic!("Charge is never called after termination; qed"),
391		};
392	}
393
394	/// Adds a charge without recording it in the contract info.
395	///
396	/// Use this method instead of [`Self::charge`] when the charge is not the result of a storage
397	/// change within the contract's child trie. This is the case when when the `code_hash` is
398	/// updated. [`Self::charge`] cannot be used here because we keep track of the deposit charge
399	/// separately from the storage charge.
400	///
401	/// If this functions is used the amount of the charge has to be stored by the caller somewhere
402	/// alese in order to be able to refund it.
403	pub fn charge_deposit(&mut self, contract: T::AccountId, amount: DepositOf<T>) {
404		self.record_charge(&amount);
405		self.charges.push(Charge { contract, amount, state: ContractState::Alive });
406	}
407
408	/// Call to tell the meter that the currently executing contract was terminated.
409	///
410	/// This will manipulate the meter so that all storage deposit accumulated in
411	/// `contract_info` will be refunded to the `origin` of the meter. And the free
412	/// (`reducible_balance`) will be sent to the `beneficiary`.
413	pub fn terminate(&mut self, info: &ContractInfo<T>, beneficiary: T::AccountId) {
414		debug_assert!(matches!(self.contract_state(), ContractState::Alive));
415		self.own_contribution = Contribution::Terminated {
416			deposit: Deposit::Refund(info.total_deposit()),
417			beneficiary,
418		};
419	}
420
421	/// [`Self::charge`] does not enforce the storage limit since we want to do this check as late
422	/// as possible to allow later refunds to offset earlier charges.
423	pub fn enforce_limit(
424		&mut self,
425		info: Option<&mut ContractInfo<T>>,
426	) -> Result<(), DispatchError> {
427		let deposit = self.own_contribution.update_contract(info);
428		let total_deposit = self.total_deposit.saturating_add(&deposit);
429		// We don't want to override a `Terminated` with a `Checked`.
430		if matches!(self.contract_state(), ContractState::Alive) {
431			self.own_contribution = Contribution::Checked(deposit);
432		}
433		if let Deposit::Charge(amount) = total_deposit {
434			if amount > self.limit {
435				log::debug!( target: LOG_TARGET, "Storage deposit limit exhausted: {:?} > {:?}", amount, self.limit);
436				return Err(<Error<T>>::StorageDepositLimitExhausted.into())
437			}
438		}
439		Ok(())
440	}
441}

443impl<T: Config> Ext<T> for ReservingExt {
444	fn charge(
445		origin: &T::AccountId,
446		contract: &T::AccountId,
447		amount: &DepositOf<T>,
448		state: &ContractState<T>,
449	) -> Result<(), DispatchError> {
450		match amount {
451			Deposit::Charge(amount) | Deposit::Refund(amount) if amount.is_zero() => return Ok(()),
452			Deposit::Charge(amount) => {
453				T::Currency::transfer_and_hold(
454					&HoldReason::StorageDepositReserve.into(),
455					origin,
456					contract,
457					*amount,
458					Precision::Exact,
459					Preservation::Preserve,
460					Fortitude::Polite,
461				)?;
462			},
463			Deposit::Refund(amount) => {
464				let transferred = T::Currency::transfer_on_hold(
465					&HoldReason::StorageDepositReserve.into(),
466					contract,
467					origin,
468					*amount,
469					Precision::BestEffort,
470					Restriction::Free,
471					Fortitude::Polite,
472				)?;
473
474				if transferred < *amount {
475					// This should never happen, if it does it means that there is a bug in the
476					// runtime logic. In the rare case this happens we try to refund as much as we
477					// can, thus the `Precision::BestEffort`.
478					log::error!(
479						target: LOG_TARGET,
480						"Failed to repatriate full storage deposit {:?} from contract {:?} to origin {:?}. Transferred {:?}.",
481						amount, contract, origin, transferred,
482					);
483				}
484			},
485		}
486		if let ContractState::<T>::Terminated { beneficiary } = state {
487			System::<T>::dec_consumers(&contract);
488			// Whatever is left in the contract is sent to the termination beneficiary.
489			T::Currency::transfer(
490				&contract,
491				&beneficiary,
492				T::Currency::reducible_balance(&contract, Preservation::Expendable, Polite),
493				Preservation::Expendable,
494			)?;
495		}
496		Ok(())
497	}
498}
