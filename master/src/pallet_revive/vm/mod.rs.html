50/// Validated Vm module ready for execution.
51/// This data structure is immutable once created and stored.
52#[derive(Encode, Decode, scale_info::TypeInfo)]
53#[codec(mel_bound())]
54#[scale_info(skip_type_params(T))]
55pub struct ContractBlob<T: Config> {
56	code: CodeVec,
57	// This isn't needed for contract execution and is not stored alongside it.
58	#[codec(skip)]
59	code_info: CodeInfo<T>,
60	// This is for not calculating the hash every time we need it.
61	#[codec(skip)]
62	code_hash: H256,
63}
64
65/// Contract code related data, such as:
66///
67/// - owner of the contract, i.e. account uploaded its code,
68/// - storage deposit amount,
69/// - reference count,
70///
71/// It is stored in a separate storage entry to avoid loading the code when not necessary.
72#[derive(Clone, Encode, Decode, scale_info::TypeInfo, MaxEncodedLen)]
73#[codec(mel_bound())]
74#[scale_info(skip_type_params(T))]
75pub struct CodeInfo<T: Config> {
76	/// The account that has uploaded the contract code and hence is allowed to remove it.
77	owner: AccountIdOf<T>,
78	/// The amount of balance that was deposited by the owner in order to store it on-chain.
79	#[codec(compact)]
80	deposit: BalanceOf<T>,
81	/// The number of instantiated contracts that use this as their code.
82	#[codec(compact)]
83	refcount: u64,
84	/// Length of the code in bytes.
85	code_len: u32,
86	/// The behaviour version that this contract operates under.
87	///
88	/// Whenever any observeable change (with the exception of weights) are made we need
89	/// to make sure that already deployed contracts will not be affected. We do this by
90	/// exposing the old behaviour depending on the set behaviour version of the contract.
91	///
92	/// As of right now this is a reserved field that is always set to 0.
93	behaviour_version: u32,
94}
95
96impl ExportedFunction {
97	/// The vm export name for the function.
98	fn identifier(&self) -> &str {
99		match self {
100			Self::Constructor => "deploy",
101			Self::Call => "call",
102		}
103	}
104}

106/// Cost of code loading from storage.
107#[cfg_attr(test, derive(Debug, PartialEq, Eq))]
108#[derive(Clone, Copy)]
109struct CodeLoadToken(u32);
110
111impl<T: Config> Token<T> for CodeLoadToken {
112	fn weight(&self) -> Weight {
113		// the proof size impact is accounted for in the `call_with_code_per_byte`
114		// strictly speaking we are double charging for the first BASIC_BLOCK_SIZE
115		// instructions here. Let's consider this as a safety margin.
116		T::WeightInfo::call_with_code_per_byte(self.0)
117			.saturating_sub(T::WeightInfo::call_with_code_per_byte(0))
118			.saturating_add(
119				T::WeightInfo::basic_block_compilation(1)
120					.saturating_sub(T::WeightInfo::basic_block_compilation(0))
121					.set_proof_size(0),
122			)
123	}
124}

131impl<T: Config> ContractBlob<T>
132where
133	BalanceOf<T>: Into<U256> + TryFrom<U256>,
134{
135	/// We only check for size and nothing else when the code is uploaded.
136	pub fn from_code(code: Vec<u8>, owner: AccountIdOf<T>) -> Result<Self, DispatchError> {
137		// We do validation only when new code is deployed. This allows us to increase
138		// the limits later without affecting already deployed code.
139		let available_syscalls = runtime::list_syscalls(T::UnsafeUnstableInterface::get());
140		let code = limits::code::enforce::<T>(code, available_syscalls)?;
141
142		let code_len = code.len() as u32;
143		let bytes_added = code_len.saturating_add(<CodeInfo<T>>::max_encoded_len() as u32);
144		let deposit = Diff { bytes_added, items_added: 2, ..Default::default() }
145			.update_contract::<T>(None)
146			.charge_or_zero();
147		let code_info = CodeInfo {
148			owner,
149			deposit,
150			refcount: 0,
151			code_len,
152			behaviour_version: Default::default(),
153		};
154		let code_hash = H256(sp_io::hashing::keccak_256(&code));
155		Ok(ContractBlob { code, code_info, code_hash })
156	}
157
158	/// Remove the code from storage and refund the deposit to its owner.
159	///
160	/// Applies all necessary checks before removing the code.
161	pub fn remove(origin: &T::AccountId, code_hash: H256) -> DispatchResult {
162		<CodeInfoOf<T>>::try_mutate_exists(&code_hash, |existing| {
163			if let Some(code_info) = existing {
164				ensure!(code_info.refcount == 0, <Error<T>>::CodeInUse);
165				ensure!(&code_info.owner == origin, BadOrigin);
166				let _ = T::Currency::release(
167					&HoldReason::CodeUploadDepositReserve.into(),
168					&code_info.owner,
169					code_info.deposit,
170					BestEffort,
171				);
172
173				*existing = None;
174				<PristineCode<T>>::remove(&code_hash);
175				Ok(())
176			} else {
177				Err(<Error<T>>::CodeNotFound.into())
178			}
179		})
180	}
181
182	/// Puts the module blob into storage, and returns the deposit collected for the storage.
183	pub fn store_code(&mut self, skip_transfer: bool) -> Result<BalanceOf<T>, Error<T>> {
184		let code_hash = *self.code_hash();
185		<CodeInfoOf<T>>::mutate(code_hash, |stored_code_info| {
186			match stored_code_info {
187				// Contract code is already stored in storage. Nothing to be done here.
188				Some(_) => Ok(Default::default()),
189				// Upload a new contract code.
190				// We need to store the code and its code_info, and collect the deposit.
191				// This `None` case happens only with freshly uploaded modules. This means that
192				// the `owner` is always the origin of the current transaction.
193				None => {
194					let deposit = self.code_info.deposit;
195
196					if !skip_transfer {
197						T::Currency::hold(
198						&HoldReason::CodeUploadDepositReserve.into(),
199						&self.code_info.owner,
200						deposit,
201					) .map_err(|err| {
202							log::debug!(target: LOG_TARGET, "failed to hold store code deposit {deposit:?} for owner: {:?}: {err:?}", self.code_info.owner);
203							<Error<T>>::StorageDepositNotEnoughFunds
204					})?;
205					}
206
207					self.code_info.refcount = 0;
208					<PristineCode<T>>::insert(code_hash, &self.code);
209					*stored_code_info = Some(self.code_info.clone());
210					Ok(deposit)
211				},
212			}
213		})
214	}
215}

217impl<T: Config> CodeInfo<T> {
218	#[cfg(test)]
219	pub fn new(owner: T::AccountId) -> Self {
220		CodeInfo {
221			owner,
222			deposit: Default::default(),
223			refcount: 0,
224			code_len: 0,
225			behaviour_version: Default::default(),
226		}
227	}
228
229	/// Returns reference count of the module.
230	#[cfg(test)]
231	pub fn refcount(&self) -> u64 {
232		self.refcount
233	}
234
235	/// Returns the deposit of the module.
236	pub fn deposit(&self) -> BalanceOf<T> {
237		self.deposit
238	}
239
240	/// Returns the code length.
241	pub fn code_len(&self) -> u64 {
242		self.code_len.into()
243	}
244
245	/// Returns the number of times the specified contract exists on the call stack. Delegated calls
246	/// Increment the reference count of a stored code by one.
247	///
248	/// # Errors
249	///
250	/// [`Error::CodeNotFound`] is returned if no stored code found having the specified
251	/// `code_hash`.
252	pub fn increment_refcount(code_hash: H256) -> DispatchResult {
253		<CodeInfoOf<T>>::mutate(code_hash, |existing| -> Result<(), DispatchError> {
254			if let Some(info) = existing {
255				info.refcount = info
256					.refcount
257					.checked_add(1)
258					.ok_or_else(|| <Error<T>>::RefcountOverOrUnderflow)?;
259				Ok(())
260			} else {
261				Err(Error::<T>::CodeNotFound.into())
262			}
263		})
264	}
265
266	/// Decrement the reference count of a stored code by one.
267	///
268	/// # Note
269	///
270	/// A contract whose reference count dropped to zero isn't automatically removed. A
271	/// `remove_code` transaction must be submitted by the original uploader to do so.
272	pub fn decrement_refcount(code_hash: H256) -> DispatchResult {
273		<CodeInfoOf<T>>::mutate(code_hash, |existing| {
274			if let Some(info) = existing {
275				info.refcount = info
276					.refcount
277					.checked_sub(1)
278					.ok_or_else(|| <Error<T>>::RefcountOverOrUnderflow)?;
279				Ok(())
280			} else {
281				Err(Error::<T>::CodeNotFound.into())
282			}
283		})
284	}
285}

287pub struct PreparedCall<'a, E: Ext> {
288	module: polkavm::Module,
289	instance: polkavm::RawInstance,
290	runtime: Runtime<'a, E, polkavm::RawInstance>,
291}
292
293impl<'a, E: Ext> PreparedCall<'a, E>
294where
295	BalanceOf<E::T>: Into<U256>,
296	BalanceOf<E::T>: TryFrom<U256>,
297{
298	pub fn call(mut self) -> ExecResult {
299		let exec_result = loop {
300			let interrupt = self.instance.run();
301			if let Some(exec_result) =
302				self.runtime.handle_interrupt(interrupt, &self.module, &mut self.instance)
303			{
304				break exec_result
305			}
306		};
307		let _ = self.runtime.ext().gas_meter_mut().sync_from_executor(self.instance.gas())?;
308		exec_result
309	}
310
311	/// The guest memory address at which the aux data is located.
312	#[cfg(feature = "runtime-benchmarks")]
313	pub fn aux_data_base(&self) -> u32 {
314		self.instance.module().memory_map().aux_data_address()
315	}
316
317	/// Copies `data` to the aux data at address `offset`.
318	///
319	/// It sets `a0` to the beginning of data inside the aux data.
320	/// It sets `a1` to the value passed.
321	///
322	/// Only used in benchmarking so far.
323	#[cfg(feature = "runtime-benchmarks")]
324	pub fn setup_aux_data(&mut self, data: &[u8], offset: u32, a1: u64) -> DispatchResult {
325		let a0 = self.aux_data_base().saturating_add(offset);
326		self.instance.write_memory(a0, data).map_err(|err| {
327			log::debug!(target: LOG_TARGET, "failed to write aux data: {err:?}");
328			Error::<E::T>::CodeRejected
329		})?;
330		self.instance.set_reg(polkavm::Reg::A0, a0.into());
331		self.instance.set_reg(polkavm::Reg::A1, a1);
332		Ok(())
333	}
334}

336impl<T: Config> ContractBlob<T> {
337	/// Compile and instantiate contract.
338	///
339	/// `aux_data_size` is only used for runtime benchmarks. Real contracts
340	/// don't make use of this buffer. Hence this should not be set to anything
341	/// other than `0` when not used for benchmarking.
342	pub fn prepare_call<E: Ext<T = T>>(
343		self,
344		mut runtime: Runtime<E, polkavm::RawInstance>,
345		entry_point: ExportedFunction,
346		aux_data_size: u32,
347	) -> Result<PreparedCall<E>, ExecError> {
348		let mut config = polkavm::Config::default();
349		config.set_backend(Some(polkavm::BackendKind::Interpreter));
350		config.set_cache_enabled(false);
351		#[cfg(feature = "std")]
352		if std::env::var_os("REVIVE_USE_COMPILER").is_some() {
353			log::warn!(target: LOG_TARGET, "Using PolkaVM compiler backend because env var REVIVE_USE_COMPILER is set");
354			config.set_backend(Some(polkavm::BackendKind::Compiler));
355		}
356		let engine = polkavm::Engine::new(&config).expect(
357			"on-chain (no_std) use of interpreter is hard coded.
358				interpreter is available on all platforms; qed",
359		);
360
361		let mut module_config = polkavm::ModuleConfig::new();
362		module_config.set_page_size(limits::PAGE_SIZE);
363		module_config.set_gas_metering(Some(polkavm::GasMeteringKind::Sync));
364		module_config.set_allow_sbrk(false);
365		module_config.set_aux_data_size(aux_data_size);
366		let module = polkavm::Module::new(&engine, &module_config, self.code.into_inner().into())
367			.map_err(|err| {
368			log::debug!(target: LOG_TARGET, "failed to create polkavm module: {err:?}");
369			Error::<T>::CodeRejected
370		})?;
371
372		let entry_program_counter = module
373			.exports()
374			.find(|export| export.symbol().as_bytes() == entry_point.identifier().as_bytes())
375			.ok_or_else(|| <Error<T>>::CodeRejected)?
376			.program_counter();
377
378		let gas_limit_polkavm: polkavm::Gas = runtime.ext().gas_meter_mut().engine_fuel_left()?;
379
380		let mut instance = module.instantiate().map_err(|err| {
381			log::debug!(target: LOG_TARGET, "failed to instantiate polkavm module: {err:?}");
382			Error::<T>::CodeRejected
383		})?;
384
385		instance.set_gas(gas_limit_polkavm);
386		instance.prepare_call_untyped(entry_program_counter, &[]);
387
388		Ok(PreparedCall { module, instance, runtime })
389	}
390}

392impl<T: Config> Executable<T> for ContractBlob<T>
393where
394	BalanceOf<T>: Into<U256> + TryFrom<U256>,
395{
396	fn from_storage(code_hash: H256, gas_meter: &mut GasMeter<T>) -> Result<Self, DispatchError> {
397		let code_info = <CodeInfoOf<T>>::get(code_hash).ok_or(Error::<T>::CodeNotFound)?;
398		gas_meter.charge(CodeLoadToken(code_info.code_len))?;
399		let code = <PristineCode<T>>::get(&code_hash).ok_or(Error::<T>::CodeNotFound)?;
400		Ok(Self { code, code_info, code_hash })
401	}
402
403	fn execute<E: Ext<T = T>>(
404		self,
405		ext: &mut E,
406		function: ExportedFunction,
407		input_data: Vec<u8>,
408	) -> ExecResult {
409		let prepared_call = self.prepare_call(Runtime::new(ext, input_data), function, 0)?;
410		prepared_call.call()
411	}
412
413	fn code(&self) -> &[u8] {
414		self.code.as_ref()
415	}
416
417	fn code_hash(&self) -> &H256 {
418		&self.code_hash
419	}
420
421	fn code_info(&self) -> &CodeInfo<T> {
422		&self.code_info
423	}
