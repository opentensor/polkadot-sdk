47/// The EVM gas price.
48/// This constant is used by the proxy to advertise it via the eth_gas_price RPC.
49///
50/// We use a fixed value for the gas price.
51/// This let us calculate the gas estimate for a transaction with the formula:
52/// `estimate_gas = substrate_fee / gas_price`.
53///
54/// The chosen constant value is:
55/// - Not too high, ensuring the gas value is large enough (at least 7 digits) to encode the
56///   ref_time, proof_size, and deposit into the less significant (6 lower) digits of the gas value.
57/// - Not too low, enabling users to adjust the gas price to define a tip.
58pub(crate) const GAS_PRICE: u64 = 1_000u64;

245/// EthExtra convert an unsigned [`crate::Call::eth_transact`] into a [`CheckedExtrinsic`].
246pub trait EthExtra {
247	/// The Runtime configuration.
248	type Config: Config + pallet_transaction_payment::Config;
249
250	/// The Runtime's transaction extension.
251	/// It should include at least:
252	/// - [`frame_system::CheckNonce`] to ensure that the nonce from the Ethereum transaction is
253	///   correct.
254	type Extension: TransactionExtension<CallOf<Self::Config>>;
255
256	/// Get the transaction extension to apply to an unsigned [`crate::Call::eth_transact`]
257	/// extrinsic.
258	///
259	/// # Parameters
260	/// - `nonce`: The nonce extracted from the Ethereum transaction.
261	/// - `tip`: The transaction tip calculated from the Ethereum transaction.
262	fn get_eth_extension(
263		nonce: <Self::Config as frame_system::Config>::Nonce,
264		tip: BalanceOf<Self::Config>,
265	) -> Self::Extension;
266
267	/// Convert the unsigned [`crate::Call::eth_transact`] into a [`CheckedExtrinsic`].
268	/// and ensure that the fees from the Ethereum transaction correspond to the fees computed from
269	/// the encoded_len, the injected gas_limit and storage_deposit_limit.
270	///
271	/// # Parameters
272	/// - `payload`: The RLP-encoded Ethereum transaction.
273	/// - `gas_limit`: The gas limit for the extrinsic
274	/// - `storage_deposit_limit`: The storage deposit limit for the extrinsic,
275	/// - `encoded_len`: The encoded length of the extrinsic.
276	fn try_into_checked_extrinsic(
277		payload: Vec<u8>,
278		encoded_len: usize,
279	) -> Result<
280		CheckedExtrinsic<AccountIdOf<Self::Config>, CallOf<Self::Config>, Self::Extension>,
281		InvalidTransaction,
282	>
283	where
284		<Self::Config as frame_system::Config>::Nonce: TryFrom<U256>,
285		BalanceOf<Self::Config>: Into<U256> + TryFrom<U256>,
286		MomentOf<Self::Config>: Into<U256>,
287		<Self::Config as frame_system::Config>::RuntimeCall: Dispatchable<Info = DispatchInfo>,
288		OnChargeTransactionBalanceOf<Self::Config>: Into<BalanceOf<Self::Config>>,
289		CallOf<Self::Config>: From<crate::Call<Self::Config>>,
290		<Self::Config as frame_system::Config>::Hash: frame_support::traits::IsType<H256>,
291	{
292		let tx = TransactionSigned::decode(&payload).map_err(|err| {
293			log::debug!(target: LOG_TARGET, "Failed to decode transaction: {err:?}");
294			InvalidTransaction::Call
295		})?;
296
297		let signer_addr = tx.recover_eth_address().map_err(|err| {
298			log::debug!(target: LOG_TARGET, "Failed to recover signer: {err:?}");
299			InvalidTransaction::BadProof
300		})?;
301
302		let signer = <Self::Config as Config>::AddressMapper::to_fallback_account_id(&signer_addr);
303		let GenericTransaction { nonce, chain_id, to, value, input, gas, gas_price, .. } =
304			GenericTransaction::from_signed(tx, crate::GAS_PRICE.into(), None);
305
306		let Some(gas) = gas else {
307			log::debug!(target: LOG_TARGET, "No gas provided");
308			return Err(InvalidTransaction::Call);
309		};
310
311		if chain_id.unwrap_or_default() != <Self::Config as Config>::ChainId::get().into() {
312			log::debug!(target: LOG_TARGET, "Invalid chain_id {chain_id:?}");
313			return Err(InvalidTransaction::Call);
314		}
315
316		let value = crate::Pallet::<Self::Config>::convert_evm_to_native(
317			value.unwrap_or_default(),
318			ConversionPrecision::Exact,
319		)
320		.map_err(|err| {
321			log::debug!(target: LOG_TARGET, "Failed to convert value to native: {err:?}");
322			InvalidTransaction::Call
323		})?;
324
325		let data = input.to_vec();
326
327		let (gas_limit, storage_deposit_limit) =
328			<Self::Config as Config>::EthGasEncoder::decode(gas).ok_or_else(|| {
329				log::debug!(target: LOG_TARGET, "Failed to decode gas: {gas:?}");
330				InvalidTransaction::Call
331			})?;
332
333		let call = if let Some(dest) = to {
334			if dest == RUNTIME_PALLETS_ADDR {
335				let call = CallOf::<Self::Config>::decode_all_with_depth_limit(
336					MAX_EXTRINSIC_DEPTH,
337					&mut &data[..],
338				)
339				.map_err(|_| {
340					log::debug!(target: LOG_TARGET, "Failed to decode data as Call");
341					InvalidTransaction::Call
342				})?;
343
344				if value != 0u32.into() {
345					log::debug!(target: LOG_TARGET, "Runtime pallets address cannot be called with value");
346					return Err(InvalidTransaction::Call)
347				}
348
349				call
350			} else {
351				crate::Call::call::<Self::Config> {
352					dest,
353					value,
354					gas_limit,
355					storage_deposit_limit,
356					data,
357				}
358				.into()
359			}
360		} else {
361			let blob = match polkavm::ProgramBlob::blob_length(&data) {
362				Some(blob_len) =>
363					blob_len.try_into().ok().and_then(|blob_len| (data.split_at_checked(blob_len))),
364				_ => None,
365			};
366
367			let Some((code, data)) = blob else {
368				log::debug!(target: LOG_TARGET, "Failed to extract polkavm code & data");
369				return Err(InvalidTransaction::Call);
370			};
371
372			crate::Call::eth_instantiate_with_code::<Self::Config> {
373				value,
374				gas_limit,
375				storage_deposit_limit,
376				code: code.to_vec(),
377				data: data.to_vec(),
378			}
379			.into()
380		};
381
382		let mut info = call.get_dispatch_info();
383		let nonce = nonce.unwrap_or_default().try_into().map_err(|_| {
384			log::debug!(target: LOG_TARGET, "Failed to convert nonce");
385			InvalidTransaction::Call
386		})?;
387		let gas_price = gas_price.unwrap_or_default();
388
389		let eth_fee = Pallet::<Self::Config>::evm_gas_to_fee(gas, gas_price)
390			.map_err(|_| InvalidTransaction::Call)?;
391
392		// Fees calculated from the extrinsic, without the tip.
393		info.extension_weight = Self::get_eth_extension(nonce, 0u32.into()).weight(&call);
394		let actual_fee: BalanceOf<Self::Config> =
395			pallet_transaction_payment::Pallet::<Self::Config>::compute_fee(
396				encoded_len as u32,
397				&info,
398				Default::default(),
399			)
400			.into();
401		log::debug!(target: LOG_TARGET, "try_into_checked_extrinsic: gas_price: {gas_price:?}, encoded_len: {encoded_len:?} actual_fee: {actual_fee:?} eth_fee: {eth_fee:?}");
402
403		// The fees from the Ethereum transaction should be greater or equal to the actual fees paid
404		// by the account.
405		if eth_fee < actual_fee {
406			log::debug!(target: LOG_TARGET, "eth fees {eth_fee:?} too low, actual fees: {actual_fee:?}");
407			return Err(InvalidTransaction::Payment.into())
408		}
409
410		let tip =
411			Pallet::<Self::Config>::evm_gas_to_fee(gas, gas_price.saturating_sub(GAS_PRICE.into()))
412				.unwrap_or_default()
413				.min(actual_fee);
414
415		crate::tracing::if_tracing(|tracer| {
416			tracer.watch_address(&Pallet::<Self::Config>::block_author().unwrap_or_default());
417			tracer.watch_address(&signer_addr);
418		});
419
420		log::debug!(target: LOG_TARGET, "Created checked Ethereum transaction with nonce: {nonce:?} and tip: {tip:?}");
421		Ok(CheckedExtrinsic {
422			format: ExtrinsicFormat::Signed(signer.into(), Self::get_eth_extension(nonce, tip)),
423			function: call,
424		})
425	}
426}
