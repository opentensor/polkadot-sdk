61/// Configuration for construction a `RuntimeInfo`.
62pub struct Config {
63	/// Needed for retrieval of `ValidatorInfo`
64	///
65	/// Pass `None` if you are not interested.
66	pub keystore: Option<KeystorePtr>,
67
68	/// How many sessions should we keep in the cache?
69	pub session_cache_lru_size: u32,
70}
71
72/// Caching of session info.
73///
74/// It should be ensured that a cached session stays live in the cache as long as we might need it.
75pub struct RuntimeInfo {
76	/// Get the session index for a given relay parent.
77	///
78	/// We query this up to a 100 times per block, so caching it here without roundtrips over the
79	/// overseer seems sensible.
80	session_index_cache: LruMap<Hash, SessionIndex>,
81
82	/// In the happy case, we do not query disabled validators at all. In the worst case, we can
83	/// query it order of `n_cores` times `n_validators` per block, so caching it here seems
84	/// sensible.
85	disabled_validators_cache: LruMap<Hash, Vec<ValidatorIndex>>,
86
87	/// Look up cached sessions by `SessionIndex`.
88	session_info_cache: LruMap<SessionIndex, ExtendedSessionInfo>,
89
90	/// Unpin handle of *some* block in the session.
91	/// Only blocks pinned explicitly by `pin_block` are stored here.
92	pinned_blocks: LruMap<SessionIndex, UnpinHandle>,
93
94	/// Key store for determining whether we are a validator and what `ValidatorIndex` we have.
95	keystore: Option<KeystorePtr>,
96}

98/// `SessionInfo` with additional useful data for validator nodes.
99pub struct ExtendedSessionInfo {
100	/// Actual session info as fetched from the runtime.
101	pub session_info: SessionInfo,
102	/// Contains useful information about ourselves, in case this node is a validator.
103	pub validator_info: ValidatorInfo,
104	/// Session executor parameters
105	pub executor_params: ExecutorParams,
106	/// Node features
107	pub node_features: NodeFeatures,
108}

110/// Information about ourselves, in case we are an `Authority`.
111///
112/// This data is derived from the `SessionInfo` and our key as found in the keystore.
113pub struct ValidatorInfo {
114	/// The index this very validator has in `SessionInfo` vectors, if any.
115	pub our_index: Option<ValidatorIndex>,
116	/// The group we belong to, if any.
117	pub our_group: Option<GroupIndex>,
118}

120impl Default for Config {
121	fn default() -> Self {
122		Self {
123			keystore: None,
124			// Usually we need to cache the current and the last session.
125			session_cache_lru_size: 2,
126		}
127	}

130impl RuntimeInfo {
131	/// Create a new `RuntimeInfo` for convenient runtime fetches.
132	pub fn new(keystore: Option<KeystorePtr>) -> Self {
133		Self::new_with_config(Config { keystore, ..Default::default() })
134	}
135
136	/// Create with more elaborate configuration options.
137	pub fn new_with_config(cfg: Config) -> Self {
138		Self {
139			// Usually messages are processed for blocks pointing to hashes from last finalized
140			// block to to best, so make this cache large enough to hold at least this amount of
141			// hashes, so that we get the benefit of caching even when finality lag is large.
142			session_index_cache: LruMap::new(ByLength::new(
143				cfg.session_cache_lru_size.max(2 * MAX_FINALITY_LAG),
144			)),
145			session_info_cache: LruMap::new(ByLength::new(cfg.session_cache_lru_size)),
146			disabled_validators_cache: LruMap::new(ByLength::new(100)),
147			pinned_blocks: LruMap::new(ByLength::new(cfg.session_cache_lru_size)),
148			keystore: cfg.keystore,
149		}
150	}

152	/// Returns the session index expected at any child of the `parent` block.
153	/// This does not return the session index for the `parent` block.
154	pub async fn get_session_index_for_child<Sender>(
155		&mut self,
156		sender: &mut Sender,
157		parent: Hash,
158	) -> Result<SessionIndex>
159	where
160		Sender: SubsystemSender<RuntimeApiMessage>,
161	{
162		match self.session_index_cache.get(&parent) {
163			Some(index) => Ok(*index),
164			None => {
165				let index =
166					recv_runtime(request_session_index_for_child(parent, sender).await).await?;
167				self.session_index_cache.insert(parent, index);
168				Ok(index)
169			},
170		}
171	}
172
173	/// Pin a given block in the given session if none are pinned in that session.
174	/// Unpinning will happen automatically when LRU cache grows over the limit.
175	pub fn pin_block(&mut self, session_index: SessionIndex, unpin_handle: UnpinHandle) {
176		self.pinned_blocks.get_or_insert(session_index, || unpin_handle);
177	}
178
179	/// Get the hash of a pinned block for the given session index, if any.
180	pub fn get_block_in_session(&self, session_index: SessionIndex) -> Option<Hash> {
181		self.pinned_blocks.peek(&session_index).map(|h| h.hash())
182	}

184	/// Get `ExtendedSessionInfo` by relay parent hash.
185	pub async fn get_session_info<'a, Sender>(
186		&'a mut self,
187		sender: &mut Sender,
188		relay_parent: Hash,
189	) -> Result<&'a ExtendedSessionInfo>
190	where
191		Sender: SubsystemSender<RuntimeApiMessage>,
192	{
193		let session_index = self.get_session_index_for_child(sender, relay_parent).await?;
194
195		self.get_session_info_by_index(sender, relay_parent, session_index).await
196	}

198	/// Get the list of disabled validators at the relay parent.
199	pub async fn get_disabled_validators<Sender>(
200		&mut self,
201		sender: &mut Sender,
202		relay_parent: Hash,
203	) -> Result<Vec<ValidatorIndex>>
204	where
205		Sender: SubsystemSender<RuntimeApiMessage>,
206	{
207		match self.disabled_validators_cache.get(&relay_parent).cloned() {
208			Some(result) => Ok(result),
209			None => {
210				let disabled_validators =
211					request_disabled_validators(relay_parent, sender).await.await??;
212				self.disabled_validators_cache.insert(relay_parent, disabled_validators.clone());
213				Ok(disabled_validators)
214			},
215		}
216	}

218	/// Get `ExtendedSessionInfo` by session index.
219	///
220	/// `request_session_info` still requires the parent to be passed in, so we take the parent
221	/// in addition to the `SessionIndex`.
222	pub async fn get_session_info_by_index<'a, Sender>(
223		&'a mut self,
224		sender: &mut Sender,
225		parent: Hash,
226		session_index: SessionIndex,
227	) -> Result<&'a ExtendedSessionInfo>
228	where
229		Sender: SubsystemSender<RuntimeApiMessage>,
230	{
231		if self.session_info_cache.get(&session_index).is_none() {
232			let session_info =
233				recv_runtime(request_session_info(parent, session_index, sender).await)
234					.await?
235					.ok_or(JfyiError::NoSuchSession(session_index))?;
236
237			let executor_params =
238				recv_runtime(request_session_executor_params(parent, session_index, sender).await)
239					.await?
240					.ok_or(JfyiError::NoExecutorParams(session_index))?;
241
242			let validator_info = self.get_validator_info(&session_info)?;
243
244			let node_features =
245				request_node_features(parent, session_index, sender).await.await??;
246			let last_set_index = node_features.iter_ones().last().unwrap_or_default();
247			if last_set_index >= FeatureIndex::FirstUnassigned as usize {
248				gum::warn!(target: LOG_TARGET, "Runtime requires feature bit {} that node doesn't support, please upgrade node version", last_set_index);
249			}
250
251			let full_info = ExtendedSessionInfo {
252				session_info,
253				validator_info,
254				executor_params,
255				node_features,
256			};
257
258			self.session_info_cache.insert(session_index, full_info);
259		}
260		Ok(self
261			.session_info_cache
262			.get(&session_index)
263			.expect("We just put the value there. qed."))
264	}

266	/// Convenience function for checking the signature of something signed.
267	pub async fn check_signature<Sender, Payload, RealPayload>(
268		&mut self,
269		sender: &mut Sender,
270		relay_parent: Hash,
271		signed: UncheckedSigned<Payload, RealPayload>,
272	) -> Result<
273		std::result::Result<Signed<Payload, RealPayload>, UncheckedSigned<Payload, RealPayload>>,
274	>
275	where
276		Sender: SubsystemSender<RuntimeApiMessage>,
277		Payload: EncodeAs<RealPayload> + Clone,
278		RealPayload: Encode + Clone,
279	{
280		let session_index = self.get_session_index_for_child(sender, relay_parent).await?;
281		let info = self.get_session_info_by_index(sender, relay_parent, session_index).await?;
282		Ok(check_signature(session_index, &info.session_info, relay_parent, signed))
283	}

285	/// Build `ValidatorInfo` for the current session.
286	///
287	///
288	/// Returns: `None` if not a parachain validator.
289	fn get_validator_info(&self, session_info: &SessionInfo) -> Result<ValidatorInfo> {
290		if let Some(our_index) = self.get_our_index(&session_info.validators) {
291			// Get our group index:
292			let our_group =
293				session_info.validator_groups.iter().enumerate().find_map(|(i, g)| {
294					g.iter().find_map(|v| {
295						if *v == our_index {
296							Some(GroupIndex(i as u32))
297						} else {
298							None
299						}
300					})
301				});
302			let info = ValidatorInfo { our_index: Some(our_index), our_group };
303			return Ok(info)
304		}
305		return Ok(ValidatorInfo { our_index: None, our_group: None })
306	}
307
308	/// Get our `ValidatorIndex`.
309	///
310	/// Returns: None if we are not a validator.
311	fn get_our_index(
312		&self,
313		validators: &IndexedVec<ValidatorIndex, ValidatorId>,
314	) -> Option<ValidatorIndex> {
315		let keystore = self.keystore.as_ref()?;
316		for (i, v) in validators.iter().enumerate() {
317			if Keystore::has_keys(&**keystore, &[(v.to_raw_vec(), ValidatorId::ID)]) {
318				return Some(ValidatorIndex(i as u32))
319			}
320		}
321		None
322	}
323}

325/// Convenience function for quickly checking the signature on signed data.
326pub fn check_signature<Payload, RealPayload>(
327	session_index: SessionIndex,
328	session_info: &SessionInfo,
329	relay_parent: Hash,
330	signed: UncheckedSigned<Payload, RealPayload>,
331) -> std::result::Result<Signed<Payload, RealPayload>, UncheckedSigned<Payload, RealPayload>>
332where
333	Payload: EncodeAs<RealPayload> + Clone,
334	RealPayload: Encode + Clone,
335{
336	let signing_context = SigningContext { session_index, parent_hash: relay_parent };
337
338	session_info
339		.validators
340		.get(signed.unchecked_validator_index())
341		.ok_or_else(|| signed.clone())
342		.and_then(|v| signed.try_into_checked(&signing_context, v))
343}

356/// Variant of `request_availability_cores` that only returns occupied ones.
357pub async fn get_occupied_cores<Sender>(
358	sender: &mut Sender,
359	relay_parent: Hash,
360) -> Result<Vec<(CoreIndex, OccupiedCore)>>
361where
362	Sender: overseer::SubsystemSender<RuntimeApiMessage>,
363{
364	let cores = get_availability_cores(sender, relay_parent).await?;
365
366	Ok(cores
367		.into_iter()
368		.enumerate()
369		.filter_map(|(core_index, core_state)| {
370			if let CoreState::Occupied(occupied) = core_state {
371				Some((CoreIndex(core_index as u32), occupied))
372			} else {
373				None
374			}
375		})
376		.collect())
377}

476/// A snapshot of the runtime claim queue at an arbitrary relay chain block.
477#[derive(Default)]
478pub struct ClaimQueueSnapshot(pub BTreeMap<CoreIndex, VecDeque<ParaId>>);
479
480impl From<BTreeMap<CoreIndex, VecDeque<ParaId>>> for ClaimQueueSnapshot {
481	fn from(claim_queue_snapshot: BTreeMap<CoreIndex, VecDeque<ParaId>>) -> Self {
482		ClaimQueueSnapshot(claim_queue_snapshot)
483	}
484}
485
486impl ClaimQueueSnapshot {
487	/// Returns the `ParaId` that has a claim for `core_index` at the specified `depth` in the
488	/// claim queue. A depth of `0` means the very next block.
489	pub fn get_claim_for(&self, core_index: CoreIndex, depth: usize) -> Option<ParaId> {
490		self.0.get(&core_index)?.get(depth).copied()
491	}
492
493	/// Returns an iterator over all claimed cores and the claiming `ParaId` at the specified
494	/// `depth` in the claim queue.
495	pub fn iter_claims_at_depth(
496		&self,
497		depth: usize,
498	) -> impl Iterator<Item = (CoreIndex, ParaId)> + '_ {
499		self.0
500			.iter()
501			.filter_map(move |(core_index, paras)| Some((*core_index, *paras.get(depth)?)))
502	}
503
504	/// Returns an iterator over all claims on the given core.
505	pub fn iter_claims_for_core(
506		&self,
507		core_index: &CoreIndex,
508	) -> impl Iterator<Item = &ParaId> + '_ {
509		self.0.get(core_index).map(|c| c.iter()).into_iter().flatten()
510	}
511
512	/// Returns an iterator over the whole claim queue.
513	pub fn iter_all_claims(&self) -> impl Iterator<Item = (&CoreIndex, &VecDeque<ParaId>)> + '_ {
514		self.0.iter()
515	}
516}
517
518/// Fetch the claim queue and wrap it into a helpful `ClaimQueueSnapshot`
519pub async fn fetch_claim_queue(
520	sender: &mut impl SubsystemSender<RuntimeApiMessage>,
521	relay_parent: Hash,
522) -> Result<ClaimQueueSnapshot> {
523	let cq = request_claim_queue(relay_parent, sender)
524		.await
525		.await
526		.map_err(Error::RuntimeRequestCanceled)??;
527
528	Ok(cq.into())
529}

531/// Returns the lookahead from the scheduler params if the runtime supports it,
532/// or default value if scheduling lookahead API is not supported by runtime.
533pub async fn fetch_scheduling_lookahead(
534	parent: Hash,
535	session_index: SessionIndex,
536	sender: &mut impl overseer::SubsystemSender<RuntimeApiMessage>,
537) -> Result<u32> {
538	let res = recv_runtime(
539		request_from_runtime(parent, sender, |tx| {
540			RuntimeApiRequest::SchedulingLookahead(session_index, tx)
541		})
542		.await,
543	)
544	.await;
545
546	if let Err(Error::RuntimeRequest(RuntimeApiError::NotSupported { .. })) = res {
547		gum::trace!(
548			target: LOG_TARGET,
549			?parent,
550			"Querying the scheduling lookahead from the runtime is not supported by the current Runtime API, falling back to default value of {}",
551			DEFAULT_SCHEDULING_LOOKAHEAD
552		);
553
554		Ok(DEFAULT_SCHEDULING_LOOKAHEAD)
555	} else {
556		res
557	}
558}
